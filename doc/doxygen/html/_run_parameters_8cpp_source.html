62 void NOMAD::RunParameters::init()
63 {
64  _typeName = "Run";
65 
66  try
67  {
68  #include "../Attribute/runAttributesDefinition.hpp"
69  registerAttributes( _definition );
70 
71  // Registered attributes using defined keywords (not in preprocessed special header file)
72  registerAttribute<NOMAD::Double>("EPSILON", NOMAD::DEFAULT_EPSILON, false,
73  " NOMAD precision for comparison of values ",
74  " \n \n . NOMAD precision for comparison of values \n "," advanced precision double(s) value(s) number(s) ");
75  registerAttribute<std::string>("UNDEF_STR", NOMAD::DEFAULT_UNDEF_STR, false,
76  "String for undefined values ",
77  " \n \n String for undefined values \n "," advanced undef(ined) value(s) string(s) ");
78  registerAttribute<std::string>("NOMAD_VERSION", NOMAD_VERSION_NUMBER, true,
79  " NOMAD version number (for runner) ",
80  " \n \n . NOMAD version number (optional) \n . If not compatible with current version will trigger exception \n " ," advanced nomad version(s) release(s) revision(s) " );
81  registerAttribute<std::string>("INF_STR", NOMAD::DEFAULT_INF_STR, false,
82  "String for infinite values",
83  " \n \n . String for infinite values \n "," advanced string(s) inf(inite) value(s) ");
84  registerAttribute<std::string>("PROBLEM_DIR", std::string(".") + NOMAD::DIR_SEP , false,
85  "Problem directory " ,
86  "\n \n . Problem directory \n . To complete \n "," problem dir(ectory) folder(s) ");
87  // Note: we cannot call checkAndComply() here, the default values
88  // are not valid.
89  }
90  catch (NOMAD::Exception & e)
91  {
92  std::string errorMsg = "Attribute registration failed: ";
93  errorMsg += e.what();
94  throw NOMAD::Exception(__FILE__,__LINE__, errorMsg);
95  }
96 }

100 /* check the parameters */
101 /*----------------------------------------*/
102 void NOMAD::RunParameters::checkAndComply(
103  const std::shared_ptr<NOMAD::EvaluatorControlGlobalParameters>& evaluatorControlGlobalParams,
104  const std::shared_ptr<NOMAD::PbParameters>& pbParams)
105 {
106  std::string err;
107  checkInfo();
108 
109  if (!toBeChecked())
110  {
111  // Early out
112  return;
113  }
114 
115  // check the non-interpreted parameters:
116  //const std::shared_ptr<NOMAD::ParameterEntry> pe = getNonInterpretedParamEntry();
117  std::vector<std::shared_ptr<NOMAD::ParameterEntry>> allNonInterp = getAllNonInterpretedParamEntries();
118  if (allNonInterp.size() > 0)
119  {
120  err = "Unrecognized parameters in file " + allNonInterp[0]->getParamFile() + ":\n";
121  for (auto pe : allNonInterp)
122  {
123  err += "line " + std::to_string(pe->getLine());
124  err += ": Unrecognized parameter: " + pe->getName() + "\n";
125  }
126  if (getAttributeValueProtected<bool>("REJECT_UNKNOWN_PARAMETERS", false))
127  {
128  throw NOMAD::Exception(__FILE__,__LINE__, err);
129  }
130  else
131  {
132  if (!_warningUnknownParamShown)
133  {
134  std::cerr << "Warning: " << err << "Ignoring unknown parameters." << std::endl;
135  }
136  }
137  }
138 
139  auto problemDir = getAttributeValueProtected<std::string>("PROBLEM_DIR", false);
140 
141  auto seed = getAttributeValueProtected<int>("SEED" ,false);
142  if ( seed < 0)
143  {
144  throw NOMAD::Exception(__FILE__,__LINE__, "Parameters check: SEED must be non-negative" );
145  }
146 
147  auto version_number = getAttributeValueProtected<std::string>("NOMAD_VERSION", false);
148  if ( version_number != NOMAD_VERSION_NUMBER )
149  {
150  err = "Parameters check: NOMAD_VERSION is not compatible with registered version " + std::string(NOMAD_VERSION_NUMBER) + "\n";
151  throw NOMAD::Exception(__FILE__,__LINE__, err );
152  }
153 
154  auto anisotropyFactor = getAttributeValueProtected<NOMAD::Double>("ANISOTROPY_FACTOR", false);
155  if ( anisotropyFactor <= 0)
156  {
157  throw NOMAD::Exception(__FILE__,__LINE__, "Parameters check: ANISOTROPY_FACTOR must be positive" );
158  }
159 
160  setStaticParameters();
161 
162  /*-------------------*/
163  /* Disable parameter */
164  /*-------------------*/
165  // Convert all disabled entries (MODELS, EVAL_SORT, ...) to upper case.
166  auto disabledConst = getAttributeValueProtected<NOMAD::ArrayOfString>("DISABLE", false);
167  NOMAD::ArrayOfString disabled;
168  for (size_t i = 0; i < disabledConst.size(); i++)
169  {
170  std::string disabledI = disabledConst[i];
171  NOMAD::toupper(disabledI);
172  disabled.add(disabledI);
173  }
174  setAttributeValue("DISABLE", disabled);
175 
176  /*---------------------------*/
177  /* Sgtelib Search parameters */
178  /*---------------------------*/
179  const size_t bigDim = 50;
180  size_t n = pbParams->getAttributeValue<size_t>("DIMENSION");
181 #ifdef USE_SGTELIB
182  bool showDisableWarn = true;
183 #endif
184  // If dimension is too large, disable models.
185  disabled = getAttributeValueProtected<NOMAD::ArrayOfString>("DISABLE", false);
186  if (n >= bigDim)
187  {
188  if (-1 == disabled.find("MODELS"))
189  {
190  disabled.add(std::string("MODELS"));
191  setAttributeValue("DISABLE", disabled);
192  std::cerr << "Warning: Dimension " << n << " is greater than (or equal to) " << bigDim << ". Models are disabled." << std::endl;
193 #ifdef USE_SGTELIB
194  showDisableWarn = false;
195 #endif
196  }
197  }
198 #ifdef USE_SGTELIB
199  // If models are disabled, set SGTELIB_SEARCH to false.
200  disabled = getAttributeValueProtected<NOMAD::ArrayOfString>("DISABLE", false);
201  if (disabled.find("MODELS") >= 0)
202  {
203  if (getAttributeValueProtected<bool>("SGTELIB_SEARCH", false))
204  {
205  if (showDisableWarn)
206  {
207  std::cerr << "Warning: Models are disabled. SGTELIB_SEARCH set to false." << std::endl;
208  }
209  setAttributeValue("SGTELIB_SEARCH", false);
210  }
211  if (getAttributeValueProtected<bool>("QUAD_MODEL_SEARCH", false))
212  {
213  if (showDisableWarn)
214  {
215  std::cerr << "Warning: Models are disabled. QUAD_MODEL_SEARCH set to false." << std::endl;
216  }
217  setAttributeValue("QUAD_MODEL_SEARCH", false);
218  }
219  }
220 #endif
221 
222  // Set default value, if the parameter is not set.
223  // Default value: TYPE LOWESS DEGREE 1 KERNEL_SHAPE OPTIM KERNEL_COEF OPTIM RIDGE 0 METRIC AOECV
224  auto sgtelibModelDefinition = getAttributeValueProtected<NOMAD::ArrayOfString>("SGTELIB_MODEL_DEFINITION",false);
225  if (sgtelibModelDefinition.size() == 0)
226  {
227  //NOMAD::ArrayOfString aos("TYPE LOWESS DEGREE 1 KERNEL_SHAPE OPTIM KERNEL_COEF OPTIM RIDGE 0 METRIC AOECV");
228  NOMAD::ArrayOfString aos("TYPE PRS DEGREE 2");
229  setAttributeValue("SGTELIB_MODEL_DEFINITION", aos );
230  }
231 
232  auto sgtelibModelTrials = getAttributeValueProtected<size_t>("SGTELIB_MODEL_TRIALS", false);
233  if (0 == sgtelibModelTrials)
234  {
235  throw NOMAD::Exception(__FILE__, __LINE__, "Parameter SGTELIB_MODEL_TRIALS must be positive");
236  }
237 
238  /*--------------------------------*/
239  /* Parallelism related parameters */
240  /*--------------------------------*/
241  // Ensure we can get value for BB_MAX_BLOCK_SIZE without throwing an exception.
242  if (evaluatorControlGlobalParams->toBeChecked())
243  {
244  evaluatorControlGlobalParams->checkAndComply();
245  }
246  auto bbBlockSize = evaluatorControlGlobalParams->getAttributeValue<size_t>("BB_MAX_BLOCK_SIZE");
247  if (0 == bbBlockSize)
248  {
249  throw NOMAD::Exception(__FILE__, __LINE__, "Parameter BB_MAX_BLOCK_SIZE must be positive");
250  }
251  auto sgteBlockSize = evaluatorControlGlobalParams->getAttributeValue<size_t>("SGTE_MAX_BLOCK_SIZE");
252  if (0 == sgteBlockSize)
253  {
254  throw NOMAD::Exception(__FILE__, __LINE__, "Parameter SGTE_MAX_BLOCK_SIZE must be positive");
255  }
256 
257 #ifndef USE_SGTELIB
258  // Look for SgtelibModel parameters that are set but cannot be used
259  if (getAttributeValueProtected<bool>("SGTELIB_MODEL_EVAL", false))
260  {
261  err = "Sgtelib Model sampling cannot be used. Either set parameter ";
262  err += "SGTELIB_MODEL_EVAL to false, or recompile NOMAD using option USE_SGTELIB=1.";
263  throw NOMAD::Exception(__FILE__,__LINE__, err);
264  }
265  if (getAttributeValueProtected<bool>("SGTELIB_SEARCH", false))
266  {
267  err = "Warning: Parameter SGTELIB_SEARCH is set to true, but ";
268  err += "Sgtelib Model sampling cannot be used. To be able to use Sgtelib Model ";
269  err += "search method, NOMAD must be recompiled using option USE_SGTELIB=1.";
270  std::cerr << err << std::endl;
271  }
272  if (getAttributeValueProtected<bool>("QUAD_MODEL_SEARCH", false))
273  {
274  err = "Warning: Parameter QUAD_MODEL_SEARCH is set to true, but ";
275  err += "Quad Model sampling cannot be used. To be able to use Quad Model Search ";
276  err += "search method, NOMAD must be recompiled using option USE_SGTELIB=1.";
277  std::cerr << err << std::endl;
278  }
279 #endif
280 
281  // PSD-Mads and SSD-Mads parameters
282  bool useAlgoPSDMads = getAttributeValueProtected<bool>("PSD_MADS_OPTIMIZATION", false);
283 
284 #ifndef _OPENMP
285  if (useAlgoPSDMads)
286  {
287  err = "Error: PSD_MADS_OPTIMIZATION can only be used when OpenMP is available. If that is not the case, use SSD_MADS_OPTIMIZATION.";
288  throw NOMAD::Exception(__FILE__,__LINE__, err);
289  }
290 #endif // _OPENMP
291 
292  bool useAlgoSSDMads = getAttributeValueProtected<bool>("SSD_MADS_OPTIMIZATION", false);
293  if (useAlgoPSDMads || useAlgoSSDMads)
294  {
295  std::string nbVarParamName = (useAlgoPSDMads ? "PSD_MADS_NB_VAR_IN_SUBPROBLEM" : "SSD_MADS_NB_VAR_IN_SUBPROBLEM");
296  const size_t nbVariablesInSubproblem = getAttributeValueProtected<size_t>(nbVarParamName, false);
297  if (0 == nbVariablesInSubproblem || nbVariablesInSubproblem > n)
298  {
299  err = "Parameter " + nbVarParamName + " must be between 1 and " + NOMAD::itos(n);
300  err += ". Value provided: " + NOMAD::itos(nbVariablesInSubproblem);
301  throw NOMAD::InvalidParameter(__FILE__,__LINE__, err);
302  }
303 
304  size_t nbMadsSubproblem = getAttributeValueProtected<size_t>((useAlgoPSDMads ? "PSD_MADS_NB_SUBPROBLEM" : "SSD_MADS_NB_SUBPROBLEM"), false);
305  // Distribute all the variables between subproblems of reduced dimension.
306  bool nbMadsSubproblemSetByUser = true;
307  if (nbMadsSubproblem == INF_SIZE_T)
308  {
309  nbMadsSubproblem = std::round(n/nbVariablesInSubproblem)+1; // Add an additional mads for the pollster
310  nbMadsSubproblemSetByUser = false;
311  }
312  if (useAlgoPSDMads)
313  {
314  // Cannot have more subproblems than the number of threads
315  size_t nbThreads = (size_t)getAttributeValueProtected<int>("NB_THREADS_OPENMP", false);
316  if (nbMadsSubproblem > nbThreads)
317  {
318  nbMadsSubproblem = nbThreads;
319  if (nbMadsSubproblemSetByUser)
320  {
321  // Warn the user
322  std::cerr << "Warning: parameter PSD_MADS_NB_SUBPROBLEM reset to number of available threads (" << nbThreads << ")" << std::endl;
323  }
324  }
325  }
326  setAttributeValue((useAlgoPSDMads ? "PSD_MADS_NB_SUBPROBLEM" : "SSD_MADS_NB_SUBPROBLEM"), nbMadsSubproblem);
327 
328  // Check parameter for coverage
329  if (useAlgoPSDMads)
330  {
331  std::string covParamName = "PSD_MADS_SUBPROBLEM_PCT_COVERAGE";
332  auto coverage = getAttributeValueProtected<NOMAD::Double>(covParamName, false);
333  if (coverage < 0.0 || coverage > 100.0)
334  {
335  err = "Parameter " + covParamName + " must be between 0.0 and 100.0";
336  throw NOMAD::InvalidParameter(__FILE__,__LINE__, err);
337  }
338  }
339  }
340 
341  // Algorithm parameters: use an algorithm other than MADS.
342  // They are mutually-exclusive.
343  bool useAlgoLH = (getAttributeValueProtected<size_t>("LH_EVAL", false) > 0);
344  bool useAlgoNM = getAttributeValueProtected<bool>("NM_OPTIMIZATION", false);
345  bool useAlgoQuadOpt = getAttributeValueProtected<bool>("QUAD_MODEL_OPTIMIZATION", false);
346  bool useAlgoSgtelibModel = getAttributeValueProtected<bool>("SGTELIB_MODEL_EVAL", false);
347  int totalAlgoSet = (int)useAlgoLH + (int)useAlgoNM + (int)useAlgoQuadOpt
348  + (int)useAlgoPSDMads + (int)useAlgoSgtelibModel + (int)useAlgoSSDMads;
349 
350  if (totalAlgoSet >= 2)
351  {
352  err = "Multiple parameters for algorithms are set. ";
353  err += "These parameters are mutually exclusive:";
354  if (useAlgoLH)
355  {
356  err += " LH_EVAL";
357  }
358  if (useAlgoNM)
359  {
360  err += " NM_OPTIMIZATION";
361  }
362  if (useAlgoPSDMads)
363  {
364  err += " PSD_MADS_OPTIMIZATION";
365  }
366  if (useAlgoQuadOpt)
367  {
368  err += " QUAD_MODEL_OPTIMIZATION";
369  }
370  if (useAlgoSgtelibModel)
371  {
372  err += " SGTELIB_MODEL_EVAL";
373  }
374  if (useAlgoSSDMads)
375  {
376  err += " SSD_MADS_OPTIMIZATION";
377  }
378  err += ". Please review parameters settings and choose only one algorithm.";
379  throw NOMAD::Exception(__FILE__,__LINE__, err);
380  }
381 
382  /*------------------------*/
383  /* Hot restart parameters */
384  /* Update file names */
385  /*------------------------*/
386  std::string hotRestartFileName = getAttributeValueProtected<std::string>("HOT_RESTART_FILE",false);
387  if (!hotRestartFileName.empty())
388  {
389  NOMAD::completeFileName(hotRestartFileName, problemDir);
390  setAttributeValue("HOT_RESTART_FILE", hotRestartFileName);
391  }
392 
393 
394  auto hMax = getAttributeValueProtected<NOMAD::Double>("H_MAX_0", false);
395  if (hMax <= 0)
396  {
397  throw NOMAD::Exception(__FILE__,__LINE__, "Parameters check: H_MAX_0 must be positive");
398  }
399 
400  _warningUnknownParamShown = true;
401 
402  _toBeChecked = false;
403 }

407 void NOMAD::RunParameters::setStaticParameters()
408 {
409  // Sub-method of checkAndComply() to set static variables of some classes.
410  NOMAD::RNG::setSeed ( getAttributeValueProtected<int>("SEED",false) );
411  NOMAD::Double::setEpsilon ( getAttributeValueProtected<NOMAD::Double>("EPSILON",false).todouble() );
412  NOMAD::Double::setUndefStr ( getAttributeValueProtected<std::string>("UNDEF_STR",false) );
413  NOMAD::Double::setInfStr ( getAttributeValueProtected<std::string>("INF_STR",false) );
414 
415  // Reset parameter values from these static values, to ensure coherence.
416  // This is bad because we have twice the same value for some parameters.
417  setAttributeValue ( "SEED", NOMAD::RNG::getSeed() );
418  setAttributeValue ( "EPSILON", NOMAD::Double(NOMAD::Double::getEpsilon()) );
419  setAttributeValue ( "UNDEF_STR", NOMAD::Double::getUndefStr() );
420  setAttributeValue ( "INF_STR", NOMAD::Double::getInfStr() );
421 }
