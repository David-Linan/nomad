61 // Evaluator type
62 enum class ComputeType
63 {
64  STANDARD, ///< f is OBJ. h is the squared sum of violations of
65  ///< all constraints. EB constraint violation result
66  ///< in h being INF.
67  PHASE_ONE, ///< f is computed based on the violation of EB
68  ///< constraints only. h is always 0. OBJ and PB
69  ///< constraints are ignored.
70  USER, ///< f and h are computed by a user-provided function.
71  UNDEFINED ///< Undefined: This value may be used when the
72  ///< ComputeType is not mandatory
73 };
