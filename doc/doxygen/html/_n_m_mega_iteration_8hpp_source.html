54 /// Class for Nelder Mead mega iteration.
55 /**
56  * Manager for Nelder Mead iterations starts, runs and ends.
57  * Steps:
58  * - Generate points over simplices (start).
59  * - Evaluate points (run)
60  * - Post-processing (end)
61  */
62 class NMMegaIteration: public MegaIteration
63 {
64 protected:
65 
66  std::shared_ptr<NMIteration> _nmIteration;
67 
68 private:
69 
70  void init();
71 
72 public:
73  /// Constructor
74  /**
75  \param parentStep The parent step of this step -- \b IN.
76  \param k The main iteration counter -- \b IN.
77  \param barrier The barrier for constraints handling -- \b IN.
78  \param success Success type of the previous MegaIteration. -- \b IN.
79  */
80  explicit NMMegaIteration(const Step* parentStep,
81  size_t k,
82  std::shared_ptr<Barrier> barrier,
83  SuccessType success)
84  : MegaIteration(parentStep, k,barrier,success), _nmIteration(nullptr)
85  {
86  init();
87  }
88  // No Destructor needed - keep defaults.
89 
90 
91  void read( std::istream& is ) override;
92  void display( std::ostream& os ) const override ;
93 
94 private:
95 
96  /// Implementation of start task.
97  /**
98  Create a Nelder Mead iteration for a current incumbent: use xFeas or xInf from the barrier. \n
99  \note Running the algorithm requires a single iteration object with several start, run, end for the various iterations of the algorithm. This allows to easily maintain a proper simplex in a single NMIteration object.
100  */
101  virtual void startImp() override ;
102 
103  /// Implementation of run task.
104  /**
105  The algorithm iterations are started, ran and ended sequentially until a stop reason to terminate is obtained. \n
106  We have a success if either a better xFeas or
107  a dominating or partial success for xInf was found.
108  See Algorithm 12.2 from DFBO.
109  */
110  virtual bool runImp() override;
111 
112 };
