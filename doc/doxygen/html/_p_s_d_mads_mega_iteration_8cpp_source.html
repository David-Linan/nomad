54 void NOMAD::PSDMadsMegaIteration::startImp()
55 {
56  auto madsStopReasons = std::make_shared<NOMAD::AlgoStopReasons<NOMAD::MadsStopType>>();
57  bool isPollster = (0 == NOMAD::getThreadNum());
58 
59  // Set parameters for subproblems
60  auto subProblemPbParams = std::make_shared<NOMAD::PbParameters>(*_pbParams);
61  auto subProblemRunParams = std::make_shared<NOMAD::RunParameters>(*_runParams);
62  setupSubproblemParams(subProblemPbParams, subProblemRunParams, isPollster);
63 
64  // Create Mads for this subproblem
65  _madsOnSubPb = std::make_shared<NOMAD::Mads>(this, madsStopReasons, subProblemRunParams, subProblemPbParams);
66  std::string madsName = "Mads ";
67  if (isPollster)
68  {
69  madsName += "pollster";
70  }
71  else
72  {
73  if (_fixedVariable.size() <= 10)
74  {
75  madsName += "with fixed variable ";
76  madsName += _fixedVariable.display();
77  }
78  else
79  {
80  madsName += "with ";
81  madsName += NOMAD::itos(_fixedVariable.size() - _fixedVariable.nbDefined());
82  madsName += " fixed variables";
83  }
84  }
85  _madsOnSubPb->setName(madsName);
86 }

90 bool NOMAD::PSDMadsMegaIteration::runImp()
91 {
92  // Run Mads
93  // Note: Pollster is always run whenever thread 0 is available.
94  // However, contrary to the NOMAD 3 version, mesh is not updated at each pollster run.
95 
96  auto evc = NOMAD::EvcInterface::getEvaluatorControl();
97  int mainThreadNum = NOMAD::getThreadNum();
98 
99  std::string s = "Running " + _madsOnSubPb->getName();
100  s += " on thread " + NOMAD::itos(mainThreadNum);
101  AddOutputInfo(s, NOMAD::OutputLevel::LEVEL_NORMAL);
102  _madsOnSubPb->start();
103  bool madsSuccessful = _madsOnSubPb->run(); // If this run is successful, barrier will be updated.
104  _madsOnSubPb->end();
105 
106  s = "Done running " + _madsOnSubPb->getName();
107  s += " on thread " + NOMAD::itos(mainThreadNum) + ". ";
108  s += "Number of evaluations: " + NOMAD::itos(evc->getBbEvalInSubproblem()) + ". ";
109  s += "Found a new success: " + NOMAD::boolToString(madsSuccessful) + ".";
110  AddOutputInfo(s, NOMAD::OutputLevel::LEVEL_NORMAL);
111  evc->resetBbEvalInSubproblem();
112 
113  return madsSuccessful;
114 }

117 void NOMAD::PSDMadsMegaIteration::setupSubproblemParams(std::shared_ptr<NOMAD::PbParameters> &subProblemPbParams,
118  std::shared_ptr<NOMAD::RunParameters> &subProblemRunParams,
119  const bool isPollster)
120 {
121  auto mainFrameSize = _mainMesh->getDeltaFrameSize();
122  auto evc = NOMAD::EvcInterface::getEvaluatorControl();
123 
124  // Note: If n >= 50, models are disabled. They could be re-enabled on
125  // subproblems with lesser dimension. See issue #370.
126 
127  subProblemPbParams->doNotShowWarnings();
128  if (isPollster)
129  {
130  subProblemRunParams->setAttributeValue("DIRECTION_TYPE", NOMAD::DirectionType::SINGLE);
131  subProblemPbParams->setAttributeValue("INITIAL_FRAME_SIZE", mainFrameSize);
132 
133  // Disable all searches
134  subProblemRunParams->setAttributeValue("LH_SEARCH", NOMAD::LHSearchType("0 0"));
135  subProblemRunParams->setAttributeValue("NM_SEARCH", false);
136  subProblemRunParams->setAttributeValue("QUAD_MODEL_SEARCH", false);
137  subProblemRunParams->setAttributeValue("SGTELIB_MODEL_SEARCH", false);
138  subProblemRunParams->setAttributeValue("SPECULATIVE_SEARCH", false);
139  }
140  else
141  {
142  auto initialFrameSize = _mainMesh->getDeltaFrameSizeCoarser();
143  subProblemPbParams->setAttributeValue("INITIAL_FRAME_SIZE", initialFrameSize);
144 
145  // The main frame size is used as minFrameSize for the subproblem.
146  // Initial and min must be compatible -> adjust.
147  for (size_t i = 0; i < initialFrameSize.size(); i++)
148  {
149  if (initialFrameSize[i] < mainFrameSize[i])
150  {
151  OUTPUT_INFO_START
152  AddOutputInfo("Set initial frame size to main frame size.");
153  OUTPUT_INFO_END
154  subProblemPbParams->setAttributeValue("INITIAL_FRAME_SIZE", mainFrameSize);
155  break;
156  }
157  }
158 
159  subProblemPbParams->setAttributeValue("FIXED_VARIABLE", _fixedVariable);
160  subProblemPbParams->setAttributeValue("X0", _x0);
161  subProblemPbParams->setAttributeValue("MIN_FRAME_SIZE", mainFrameSize);
162  subProblemRunParams->setAttributeValue("PSD_MADS_NB_VAR_IN_SUBPROBLEM", _fixedVariable.size() - _fixedVariable.nbDefined());
163  }
164 
165  // Set max number of bb evals per subproblem.
166  // Strategy to be refined.
167  if (isPollster)
168  {
169  evc->setMaxBbEvalInSubproblem(1);
170  }
171  else
172  {
173  size_t totalBudget = evc->getEvaluatorControlGlobalParams()->getAttributeValue<size_t>("MAX_BB_EVAL");
174  size_t nbSubproblem = _runParams->getAttributeValue<size_t>("PSD_MADS_NB_SUBPROBLEM");
175  size_t budgetPerSubproblem = (totalBudget-1) / nbSubproblem;
176  size_t maxBbEvalInSubproblem = evc->getMaxBbEvalInSubproblem();
177  if (budgetPerSubproblem < maxBbEvalInSubproblem)
178  {
179  maxBbEvalInSubproblem = budgetPerSubproblem;
180  }
181  evc->setMaxBbEvalInSubproblem(maxBbEvalInSubproblem);
182  }
183 
184  subProblemPbParams->checkAndComply();
185  auto evcParams = evc->getEvaluatorControlGlobalParams();
186  subProblemRunParams->checkAndComply(evcParams, subProblemPbParams);
187 }
