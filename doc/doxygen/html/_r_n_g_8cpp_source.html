90 uint32_t NOMAD::RNG::rand()
91 {
92  // http://madrabbit.org/~ray/code/xorshf96.c //period 2^96-1
93 
94  uint32_t t;
95  _x ^= _x << 16;
96  _x ^= _x >> 5;
97  _x ^= _x << 1;
98 
99  t = _x;
100  _x = _y;
101  _y = _z;
102  _z = t ^ _x ^ _y;
103 
104  return _z;
105 }

111 double NOMAD::RNG::normalRand(double mean, double var)
112 {
113  // Box-Muller transformation~\cite{BoMu58}
114 
115  double x1, w;
116 
117  do
118  {
119  x1 = NOMAD::RNG::rand(-1.0,1.0);
120  double x2 = NOMAD::RNG::rand(-1.0,1.0);
121  w = x1 * x1 + x2 * x2;
122  } while ( w >= 1.0 );
123 
124  w = sqrt( (-2.0 * log( w ) ) / w );
125 
126  return pow(var,0.5) * x1 * w + mean;
127 }

130 double NOMAD::RNG::normalRandMean0(double Var, int Nsample)
131 {
132  double sum = 0.0;
133  double a = pow(3.0 * Var, 0.5);
134  for (int i = 0 ; i < Nsample ; i++)
135  {
136  sum += NOMAD::RNG::rand(-a,a);
137  }
138  return sum / pow(Nsample, 0.5);
