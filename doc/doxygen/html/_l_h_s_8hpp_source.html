57 /// \brief Latin Hypercube Sampling class.
58 /**
59  * Input:
60  * n dimension
61  * p number of desired samples
62  * lowerBound and upperBound of type ArrayOfDouble indicating lower and upper bounds. Must be completely defined.
63  * seed (optional)
64  *
65  * Output:
66  * p points of dimension n, distributed in the l x u hyper-rectangle of R^n
67  */

69 {

71  size_t _n; ///< dimension
72  size_t _p; ///< number of samples
73  ArrayOfDouble _lowerBound; ///< lower bounds
74  ArrayOfDouble _upperBound; ///< upper bounds
75

77  /// Constructor
78  /**
79  \param n Dimension -- \b IN.
80  \param p Number of samples -- \b IN.
81  \param lowerBound Lower bounds -- \b IN.
82  \param upperBound Upper bounds -- \b IN.
83  */
84  explicit LHS(const size_t n,
85  const size_t p,
86  const ArrayOfDouble& lowerBound,
87  const ArrayOfDouble& upperBound,
88  const NOMAD::Point& frameCenter = NOMAD::Point(),
