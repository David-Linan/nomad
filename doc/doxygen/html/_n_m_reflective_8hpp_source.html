55 /// Class to perform the Reflect, Expansion, Inside_Contraction, Outside_Contraction steps of a Nelder Mead algorithm.
56 /**
57  The function NMReflective::getNextNMStepType decides which Nelder Mead step comes next. The next step type is updated when calling run (NMReflective::runImp).\n
58 
59  The class maintains the lists of undominated (Y0) and dominated points (Yn) extracted from simplex. See the NM-Mads paper: https://link.springer.com/article/10.1007/s10589-018-0016-0 for details.
60 
61  To compare points we use a dominance operator from EvalPoint::dominates.
62 
63  * \note The name "reflective" is because all those steps are reflections with different delta.
64  */
65 class NMReflective: public Step, public NMIterationUtils
66 {
67 private:
68 
69 
70  NMStepType _nextStepType;
71  Double _delta, _deltaE, _deltaOC, _deltaIC;
72 
73  EvalPoint _xr, _xe, _xoc, _xic;
74 
75  std::vector<EvalPoint> _nmY0; ///< Vector of undominated points extracted from simplex (the simplex has a loose ordering-->tied points can exist).
76  std::vector<EvalPoint> _nmYn; ///< Vector of dominated points extracted from simplex (the simplex has a loose ordering-->tied points can exist).
77 
78 
79 public:
80  /// Constructor
81  /**
82  \param parentStep The parent of this NM step
83  */
84  explicit NMReflective(const Step* parentStep)
85  : Step(parentStep),
86  NMIterationUtils(parentStep)
87  {
88 
89 
90  init();
91  }
92  virtual ~NMReflective() {}
93 
94  /// Generate new points to evaluate
95  /**
96  A new point is obtained using the simplex. xt = yc + dela*d. Delta is the reflective factor. The value depends the step type (REFLECT, EXPAND, INSIDE_CONTRACTION, OUTSIDE_CONTRACTION). yc is the barycenter of the simplex. We have d=yc-yn where yn is the last point of Y. \n
97 
98  The point is snapped to bounds and projected on the mesh.
99  */
100  void generateTrialPoints() override;
101 
102  /// Set the parameters for a given step type.
103  /**
104  The name of the step and the value of delta are changed according to stepType.
105  */
106  void setCurrentNMStepType(NMStepType stepType);
107 
108 
109  NMStepType getNextNMStepType() const { return _nextStepType ; }
110 
111 private:
112 
113  /**
114  The delta parameter used to create the trial point is different for EXPANSION, INSIDE_CONTRACTION, OUTSIDE_CONTRACTION. The possible delta parameters are obtained from _runParams. The validity of the parameters are checked. \n
115  The flag to perform a standalone Nelder Mead optimization is also set.
116  */
117  void init();
118 
119  /// Implementation of the start task.
120  /**
121  Call NMReflective::generateTrialPoints and update the trial points.
122  */
123  virtual void startImp() override ;
124 
125  /// Implementation of the run task.
126  /**
127  Evaluate the trial point and store it locally. Call IterationUtils::postProcessing.
128 
129  \return \c true if a better point is obtained \c false otherwise.
130  */
131  virtual bool runImp() override ;
132 
133  /// No end task is required
134  virtual void endImp() override {}
135 
136  /// Helper for NMReflective::runImp
137  void setNextNMStepType();
138 
139  /**
140  Reflect is always the first step of Nelder Mead iteration. The reflect point is xr. In NM-Mads paper, depending on which zone xr belongs to, we perform another step:
141  - If xr dominates Y0 -> EXPAND
142  - If Yn dominates xr -> INSIDE_CONTRACTION
143  - If xr dominates 2 points or more in Y -> iteration completed
144  - If xr dominates 0 or 1 point in Y -> OUTSIDE_CONTRACTION
145  */
146  void setAfterReflect() ;
147 
148 
149  /**
150  EXPAND follows REFLECT. The expand point is xe. \n
151  In NM-Mads paper: xr belongs to the expansion zone. xe has been evaluated. The best point between xr and xe is inserted in the simplex Y. If a proper simplex Y is obtained the iteration is completed, if not the next step can be a SHRINK.
152  */
153  void setAfterExpand();
154 
155  /**
156  OUTSIDE_CONTRACTION follows REFLECT. The outside contraction point is xoc. \n
157  In NM-Mads paper: xr belongs to the outside contraction zone. The best point between xr and xoc is inserted in Y. If a proper simplex Y is obtained the iteration is completed, if not the next step can be a SHRINK.
158  */
159  void setAfterOutsideContract();
160 
161  /**
162  INSIDE_CONTRACTION follows REFLECT. The inside contraction point is xic. \n
163  In NM-Mads paper: xr belongs to the inside contraction zone. If xic belongs to the inside contraction zone (that is Yn dominates xic), iteration is completed. Otherwise insert xic in Y. If a proper simplex Y is obtained, the iteration is completed, if not the next step can be a SHRINK.
164  */
165  void setAfterInsideContract();
166 
167  /**
168  Insert a point in the simplex Y. If a point is inserted, the last point of Y is removed, return \c true. If Y is unchanged, the insertion failed, return \c false. \n
169  Update Y0, Yn and the simplex characteristic if necessary.
170  */
171  bool insertInY(const EvalPoint& x);
172 
173  /**
174  Try to insert the best of two points in the simplex.
175 
176  \return \c true if the simplex has changed and \c false otherwise.
177  */
178  bool insertInYBest(const EvalPoint& x1, const EvalPoint& x2);
179 
180  /// Helper for the setAfterXXXX functions
181  bool pointDominatesY0(const EvalPoint& x) const ;
182 
183  /// Helper for the setAfterXXXX functions
184  bool YnDominatesPoint(const EvalPoint& x) const ;
185 
186  /// Helper for the setAfterXXXX functions
187  bool pointDominatesPtsInY(const EvalPoint& x, size_t nb) const ;
188 
189  bool makeListY0(); ///< Create the undominated list of points from Y
190  bool makeListYn(); ///< Create the dominated list of points from Y
191 
192  void displayY0nInfo() const ;
193 };
