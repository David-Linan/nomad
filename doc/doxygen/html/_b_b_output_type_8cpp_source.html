60 // Convert a string (ex "OBJ", "EB", "PB"...)
61 // to a NOMAD::BBOutputType.
62 NOMAD::BBOutputType NOMAD::stringToBBOutputType(const std::string &sConst)
63 {
64  NOMAD::BBOutputType ret = NOMAD::BBOutputType::BBO_UNDEFINED;
65  std::string s = sConst;
66  NOMAD::toupper(s);
67 
68  if (s == "OBJ")
69  {
70  ret = NOMAD::BBOutputType::OBJ;
71  }
72  else if (s == "EB")
73  {
74  ret = NOMAD::BBOutputType::EB;
75  }
76  else if (s == "PB" || s == "CSTR")
77  {
78  ret = NOMAD::BBOutputType::PB;
79  }
80  else if (s == "CNT_EVAL")
81  {
82  ret = NOMAD::BBOutputType::CNT_EVAL;
83  }
84  else if (s == "EXTRA_O" || s == "NOTHING" || s == "-" || s == "BBO_UNDEFINED")
85  {
86  ret = NOMAD::BBOutputType::BBO_UNDEFINED;
87  }
88  else
89  {
90  throw NOMAD::Exception(__FILE__, __LINE__, "Unrecognized string for NOMAD::BBOutputType: " + s);
91  }
92 
93  return ret;
94 }

96 // test if a BBOutputType is a constraint (PB, EB, ....). Add new constraint type as they appear.
97 bool NOMAD::BBOutputTypeIsConstraint( const BBOutputType & bboType )
98 {
99  bool ret = false;
100  switch ( bboType )
101  {
102  case NOMAD::BBOutputType::EB:
103  ret = true;
104  break;
105  case NOMAD::BBOutputType::PB:
106  ret = true;
107  break;
108  default:
109  ret = false;
110  }
111  return ret;
112 }

115 // Convert a string containing multiple BBOutputTypes (ex "OBJ EB PB PB")
116 // to a NOMAD::BBOutputTypeList.
117 NOMAD::BBOutputTypeList NOMAD::stringToBBOutputTypeList(const std::string &s)
118 {
119  NOMAD::BBOutputTypeList bbOutputType;
120  NOMAD::ArrayOfString aos(s);
121  for (size_t i = 0; i < aos.size(); i++)
122  {
123  bbOutputType.push_back(NOMAD::stringToBBOutputType(aos[i]));
124  }
125  return bbOutputType;
126 }
