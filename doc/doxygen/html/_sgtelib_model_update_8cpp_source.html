80 // Update the SGTELIB::TrainingSet and SGTELIB::Surrogate contained in the
81 // ancestor SgtelibModel (modelAlgo).
82 //
83 // 1- Get relevant points in cache, around current frame centers.
84 // 2- Add points to training set, and build new model.
85 // 3- Assess if model is ready. Update its bounds.
86 //
87 // Note: Update uses blackbox values
88 bool NOMAD::SgtelibModelUpdate::runImp()
89 {
90  bool updateDone = true;
91  std::string s; // Used for output
92 
93  auto modelFeasibility = _runParams->getAttributeValue<NOMAD::SgtelibModelFeasibilityType>("SGTELIB_MODEL_FEASIBILITY");
94  auto modelFormulation = _runParams->getAttributeValue<NOMAD::SgtelibModelFormulationType>("SGTELIB_MODEL_FORMULATION");
95 
96  auto modelAlgo = getParentOfType<NOMAD::SgtelibModel*>();
97  if (nullptr == modelAlgo)
98  {
99  s = "Error: In SgtelibModelUpdate, need a SgtelibModel parent.";
100  throw NOMAD::Exception(__FILE__, __LINE__, s);
101  }
102 
103  auto n = _pbParams->getAttributeValue<size_t>("DIMENSION");
104  const auto bbot = NOMAD::SgtelibModel::getBBOutputType();
105  size_t nbConstraints = NOMAD::getNbConstraints(bbot);
106  size_t nbModels = NOMAD::SgtelibModel::getNbModels(modelFeasibility, nbConstraints);
107 
108  if (NOMAD::SgtelibModelFormulationType::EXTERN == modelFormulation)
109  {
110  // Extern model. Early out
111  OUTPUT_INFO_START
112  AddOutputInfo("FORMULATION: EXTERN.", _displayLevel);
113  OUTPUT_INFO_END
114  return updateDone;
115  }
116 
117  // row_X and row_Z are one-liner matrices to stock current point.
118  SGTELIB::Matrix row_X("row_X", 1, static_cast<int>(n));
119  SGTELIB::Matrix row_Z("row_Z", 1, static_cast<int>(nbModels));
120 
121  // Matrices to stock all points to send to the model
122  auto add_X = std::make_shared<SGTELIB::Matrix>("add_X", 0, static_cast<int>(n));
123  auto add_Z = std::make_shared<SGTELIB::Matrix>("add_Z", 0, static_cast<int>(nbModels));
124 
125  // Go through cache points
126  OUTPUT_INFO_START
127  AddOutputInfo("Review of the cache", _displayLevel);
128  OUTPUT_INFO_END
129  int k = 0;
130  NOMAD::Double v;
131 
132  //
133  // 1- Get relevant points in cache, around current frame centers.
134  //
135  std::vector<NOMAD::EvalPoint> evalPointList;
136  if (NOMAD::EvcInterface::getEvaluatorControl()->getUseCache())
137  {
138  // Get valid points: notably, they have a BB evaluation.
139  // Use CacheInterface to ensure the points are converted to subspace
140  NOMAD::CacheInterface cacheInterface(this);
141  cacheInterface.find(validForUpdate, evalPointList);
142  }
143 
144  // Minimum and maximum number of valid points to build a model
145  const size_t minNbPoints = _runParams->getAttributeValue<size_t>("SGTELIB_MIN_POINTS_FOR_MODEL");
146  // Not using SGTELIB_MAX_POINTS_FOR_MODEL, see below.
147  //const size_t maxNbPoints = _runParams->getAttributeValue<size_t>("SGTELIB_MAX_POINTS_FOR_MODEL");
148 
149  // Select valid points that are close enough to frame centers
150  // Compute distances that must not be violated for each variable.
151  // Get a Delta (frame size) if available.
152  NOMAD::ArrayOfDouble radius(n, 1.0);
153  if (nullptr != modelAlgo->getMesh())
154  {
155  radius = modelAlgo->getMesh()->getDeltaFrameSize();
156  }
157  // Multiply by radius parameter
158  auto radiusFactor = _runParams->getAttributeValue<NOMAD::Double>("SGTELIB_MODEL_RADIUS_FACTOR");
159  radius *= radiusFactor;
160 
161  // Get all frame centers
162  auto megaIter = getParentOfType<NOMAD::SgtelibModelMegaIteration*>();
163  auto allCenters = megaIter->getBarrier()->getAllPoints();
164  size_t nbCenters = allCenters.size();
165  std::vector<NOMAD::EvalPoint> evalPointListWithinRadius;
166  for (auto evalPoint : evalPointList)
167  {
168  bool pointAdded = false;
169  for (size_t centerIndex = 0;
170  centerIndex < nbCenters && !pointAdded;
171  centerIndex++)
172  {
173  auto center = allCenters[centerIndex];
174  auto distances = NOMAD::Point::vectorize(*center.getX(), evalPoint);
175  distances = distances.abs();
176  if (distances <= radius)
177  {
178  // This evalPoint is within radius. Keep it.
179  evalPointListWithinRadius.push_back(evalPoint);
180  pointAdded = true;
181  }
182  }
183  }
184  evalPointList = evalPointListWithinRadius;
185  size_t nbValidPoints = evalPointList.size();
186 
187  /*
188  // This is not a safe way to select a subset of points.
189  // First, we should ensure that the frame center is part of that subset.
190  // Second, how to select the other points? The points closest to
191  // the frame center seem like an interesting option.
192  if (nbValidPoints > maxNbPoints)
193  {
194  OUTPUT_INFO_START
195  s = "SgtelibModel found " + std::to_string(nbValidPoints);
196  s += " valid points to build model. Keep only ";
197  s += std::to_string(maxNbPoints);
198  AddOutputInfo(s);
199  OUTPUT_INFO_END
200  // First, sort evalPointList by dominance
201  std::sort(evalPointList.begin(), evalPointList.end());
202  // Next, get first maxNbPoints points
203  std::vector<NOMAD::EvalPoint> evalPointListShort;
204  for (size_t i = 0; i < maxNbPoints; i++)
205  {
206  evalPointListShort.push_back(evalPointList[i]);
207  }
208  evalPointList = evalPointListShort;
209  nbValidPoints = evalPointList.size();
210  }
211  */
212 
213  if (nbValidPoints < minNbPoints)
214  {
215  // If no points available, it is impossible to build a model.
216  auto sgtelibModelStopReason = NOMAD::AlgoStopReasons<NOMAD::ModelStopType>::get(_stopReasons);
217  sgtelibModelStopReason->set(NOMAD::ModelStopType::NOT_ENOUGH_POINTS);
218 
219  OUTPUT_INFO_START
220  AddOutputInfo("Sgtelib Model has not enough points to build model");
221  OUTPUT_INFO_END
222  }
223  OUTPUT_INFO_START
224  s = "Found " + std::to_string(nbValidPoints);
225  s += " point";
226  if (nbValidPoints > 1)
227  {
228  s += "s";
229  }
230  s += " in cache of size " + std::to_string(NOMAD::CacheBase::getInstance()->size());
231  AddOutputInfo(s);
232  OUTPUT_INFO_END
233 
234  for (auto evalPoint : evalPointList)
235  {
236  NOMAD::Point x = *evalPoint.getX();
237  OUTPUT_INFO_START
238  s = "xNew = " + evalPoint.display();
239  AddOutputInfo(s, _displayLevel);
240  OUTPUT_INFO_END
241 
242  for (size_t j = 0; j < n; j++)
243  {
244  // X
245  row_X.set(0, static_cast<int>(j), x[j].todouble());
246  }
247  add_X->add_rows(row_X);
248 
249  // Objective
250  // Update uses blackbox values
251  row_Z.set(0, 0, evalPoint.getF(NOMAD::EvalType::BB, NOMAD::EvcInterface::getEvaluatorControl()->getComputeType()).todouble()); // 1st column: constraint model
252 
253  NOMAD::ArrayOfDouble bbo = evalPoint.getEval(NOMAD::EvalType::BB)->getBBOutput().getBBOAsArrayOfDouble();
254  // Constraints
255  switch (modelFeasibility)
256  {
257  case NOMAD::SgtelibModelFeasibilityType::C:
258  k = 1;
259  for (size_t j = 0; j < bbo.size(); j++)
260  {
261  if (NOMAD::isConstraint(bbot[j]))
262  {
263  row_Z.set(0, k, bbo[j].todouble());
264  k++;
265  }
266  }
267  break;
268 
269  case NOMAD::SgtelibModelFeasibilityType::H:
270  row_Z.set(0, 1, v.todouble()); // 2nd column: constraint model
271  break;
272 
273  case NOMAD::SgtelibModelFeasibilityType::B:
274  row_Z.set(0, 1, evalPoint.isFeasible(NOMAD::EvalType::BB, NOMAD::EvcInterface::getEvaluatorControl()->getComputeType())); // 2nd column: constraint model
275  break;
276 
277  case NOMAD::SgtelibModelFeasibilityType::M:
278  v = -NOMAD::INF;
279  for (size_t j = 0; j < bbot.size(); j++)
280  {
281  if (NOMAD::isConstraint(bbot[j]))
282  {
283  v = max(v, bbo[j]);
284  }
285  }
286  row_Z.set(0, 1, v.todouble()); // 2nd column: constraint model
287  break;
288 
289  case NOMAD::SgtelibModelFeasibilityType::UNDEFINED:
290  default:
291  AddOutputInfo("UNDEFINED", _displayLevel);
292  break;
293  } // end switch
294 
295  add_Z->add_rows(row_Z);
296 
297  if (evalPoint.isFeasible(NOMAD::EvalType::BB, NOMAD::EvcInterface::getEvaluatorControl()->getComputeType()))
298  {
299  if (!modelAlgo->getFoundFeasible())
300  {
301  AddOutputInfo(" (feasible)", _displayLevel);
302  modelAlgo->setFoundFeasible(true);
303  }
304  }
305  } // Done going through valid points
306 
307  //
308  // 2- Add points to training set, and build new model.
309  //
310  std::shared_ptr<SGTELIB::TrainingSet> trainingSet = modelAlgo->getTrainingSet();
311  std::shared_ptr<SGTELIB::Surrogate> model = modelAlgo->getModel();
312 
313  OUTPUT_INFO_START
314  s = "Current nb of points: " + std::to_string(trainingSet->get_nb_points());
315  AddOutputInfo(s, NOMAD::OutputLevel::LEVEL_INFO);
316  OUTPUT_INFO_END
317 
318  if (add_X->get_nb_rows() > 0)
319  {
320  // Build the model
321  OUTPUT_INFO_START
322  AddOutputInfo("Add points...", _displayLevel);
323  OUTPUT_INFO_END
324 
325  trainingSet->add_points(*add_X, *add_Z);
326 
327  OUTPUT_INFO_START
328  AddOutputInfo("OK", _displayLevel);
329  AddOutputInfo("Build model...", _displayLevel);
330  OUTPUT_INFO_END
331 
332  model->build();
333  OUTPUT_INFO_START
334  AddOutputInfo("OK.", _displayLevel);
335  OUTPUT_INFO_END
336  }
337 
338  //
339  // 3- Assess if model is ready. Update its bounds.
340  //
341  // Check if the model is ready
342  bool ready = model->is_ready();
343  modelAlgo->setReady(ready);
344 
345  OUTPUT_INFO_START
346  s = "New nb of points: " + std::to_string(trainingSet->get_nb_points());
347  AddOutputInfo(s, NOMAD::OutputLevel::LEVEL_INFO);
348  s = "Ready: " + NOMAD::boolToString(ready);
349  AddOutputInfo(s, NOMAD::OutputLevel::LEVEL_INFO);
350  OUTPUT_INFO_END
351 
352  // Update the bounds of the model
353  modelAlgo->setModelBounds(add_X);
354 
355  return updateDone;
356 }

364 bool NOMAD::SgtelibModelUpdate::validForUpdate(const NOMAD::EvalPoint& evalPoint)
365 {
366  // Verify that the point is valid
367  // - Not a NaN
368  // - Not a fail
369  // - All outputs defined
370  // - Blackbox OBJ available (Not MODEL)
371  bool validPoint = true;
372  NOMAD::ArrayOfDouble bbo;
373 
374  auto computeType = NOMAD::EvcInterface::getEvaluatorControl()->getComputeType();
375  auto eval = evalPoint.getEval(NOMAD::EvalType::BB);
376  if (nullptr == eval)
377  {
378  // Blackbox Eval must be available
379  validPoint = false;
380  }
381  else
382  {
383  bbo = eval->getBBOutput().getBBOAsArrayOfDouble();
384 
385  // Note: it could be discussed if points that have h > hMax should still be used
386  // to build the model. We validate them to comply with Nomad 3.
387  // If f or h greater than MODEL_MAX_OUTPUT (default=1E10) the point is not valid (same as Nomad 3)
388  if ( ! eval->isBBOutputComplete()
389  || !(NOMAD::EvalStatusType::EVAL_OK == eval->getEvalStatus())
390  || !eval->getF(computeType).isDefined()
391  || !eval->getH(computeType).isDefined()
392  || eval->getF(computeType) > NOMAD::MODEL_MAX_OUTPUT
393  || eval->getH(computeType) > NOMAD::MODEL_MAX_OUTPUT)
394  {
395  validPoint = false;
396  }
397  }
398 
399  return validPoint;
400 }
