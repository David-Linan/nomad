96 void NOMAD::MegaIteration::computeMaxXFeasXInf(size_t &maxXFeas, size_t &maxXInf)
97 {
98  const size_t maxIter = _runParams->getAttributeValue<size_t>("MAX_ITERATION_PER_MEGAITERATION");
99  const size_t maxXFeas0 = maxXFeas;
100  const size_t maxXInf0 = maxXInf;
101 
102  // If maxXFeas + maxXInf does not exceed maxIter, do nothing.
103  if (maxXFeas + maxXInf > maxIter)
104  {
105  if (maxXFeas <= maxIter / 2)
106  {
107  // Use all xFeas, and the remaining in xInf.
108  maxXInf = maxIter - maxXFeas;
109  }
110  else if (maxXInf < maxIter / 2)
111  {
112  // Use all xInf, and the remaining in xFeas.
113  maxXFeas = maxIter - maxXInf;
114  }
115  else
116  {
117  // Both the number of xFeas and xInf is over.
118  // Use half of maxIter for xFeas and half for xInf.
119  maxXInf = maxIter / 2;
120  maxXFeas = maxIter - maxXInf;
121  }
122  if (maxXFeas + maxXInf > maxIter)
123  {
124  // This case should not happen and should be debugged.
125  std::cerr << "Warning: Bad computation in computeMaxXFeasXInf. maxIter = " << maxIter << " maxXFeas = " << maxXFeas << " (was " << maxXFeas0 << ") maxXInf = " << maxXInf << " (was " << maxXInf0 << ")" << std::endl;
126  }
127  }
128 }

131 void NOMAD::MegaIteration::read(std::istream& is)
132 {
133  // Set up structures to gather member info
134  size_t k = 0;
135 
136  // Read line by line
137  std::string name;
138  while (is >> name && is.good() && !is.eof())
139  {
140  if ("ITERATION_COUNT" == name)
141  {
142  is >> k;
143  }
144  else if ("BARRIER" == name)
145  {
146  if (nullptr != _barrier)
147  {
148  is >> *_barrier;
149  }
150  else
151  {
152  std::string err = "Error: Reading a Barrier onto a NULL pointer";
153  std::cerr << err << std::endl;
154  }
155  }
156  else
157  {
158  for (size_t i = 0; i < name.size(); i++)
159  {
160  is.unget();
161  }
162  break;
163  }
164  }
165 
166  setK(k);
167 
168 }
