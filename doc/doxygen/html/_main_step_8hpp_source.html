61 /// Main step to manage an algorithm execution.
62 /**
63 * MainStep is the root step of an executable running an algorithm.
64 * Think of it as a wrapper around the algorithm.
65 * MainStep manages all algorithm needs: Parameters, Evaluator
66 * and has utility functions (ex. displayUsage).
67 * MainStep takes care of the OpenMP parallelism.
68 * An algorithm can call other algorithms during its execution.
69 */
70 class MainStep: public Step

76  std::vector<std::shared_ptr<Algorithm>> _algos;

82 #endif // TIME_STATS
