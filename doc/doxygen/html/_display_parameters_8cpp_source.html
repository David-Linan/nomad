78 void NOMAD::DisplayParameters::checkAndComply(
79  const std::shared_ptr<NOMAD::RunParameters> &runParams,
80  const std::shared_ptr<NOMAD::PbParameters> &pbParams)
81 {
82 
83  checkInfo();
84 
85  if (!toBeChecked())
86  {
87  // Early out
88  return;
89  }
90 
91  // Pb params must be checked before accessing its value
92  size_t n = pbParams->getAttributeValue<size_t>("DIMENSION");
93  if (n == 0)
94  {
95  throw NOMAD::Exception(__FILE__,__LINE__, "Parameters check: DIMENSION must be positive" );
96  }
97 
98 
99  // SOL_FORMAT (internal)
100  auto solFormat = getAttributeValueProtected<NOMAD::ArrayOfDouble>("SOL_FORMAT",false);
101  if ( !solFormat.isDefined() )
102  {
103  solFormat.reset(n, NOMAD::DISPLAY_PRECISION_STD);
104  setAttributeValue("SOL_FORMAT", solFormat);
105  }
106 
107  if ( ! pbParams->isAttributeDefaultValue<NOMAD::ArrayOfDouble>("GRANULARITY") )
108  {
109 
110  // Update SOL_FORMAT.
111  // We want to remember the number of decimals in
112  // GRANULARITY arguments, to be used later for formatting.
113  //
114  // Default is DISPLAY_PRECISION_STD.
115 
116  auto newSolFormat = setFormatFromGranularity( pbParams->getAttributeValue<NOMAD::ArrayOfDouble>("GRANULARITY") );
117  setAttributeValue("SOL_FORMAT", newSolFormat);
118  }
119 
120  // The default value is empty: set a basic display stats: BBE OBJ
121  auto displayStats = getAttributeValueProtected<NOMAD::ArrayOfString>("DISPLAY_STATS",false);
122  if ( displayStats.size() == 0 )
123  {
124  NOMAD::ArrayOfString aos("BBE OBJ");
125  setAttributeValue("DISPLAY_STATS", aos );
126  }
127 
128 
129  /*------------------------------------------------------*/
130  /* Stats files */
131  /*------------------------------------------------------*/
132 
133  auto statsFileParam = getAttributeValueProtected<NOMAD::ArrayOfString>("STATS_FILE",false) ;
134  std::string statsFileName;
135  if (statsFileParam.size() > 0)
136  {
137  statsFileName = statsFileParam[0];
138  if (statsFileParam.size() == 1)
139  {
140  // Default stats: BBE OBJ.
141  statsFileParam.add("BBE");
142  statsFileParam.add("OBJ");
143  }
144  }
145 
146 
147  // Update stats file name
148  auto addSeedToFileNames = runParams->getAttributeValue<bool>("ADD_SEED_TO_FILE_NAMES");
149  auto problemDir = runParams->getAttributeValue<std::string>("PROBLEM_DIR");
150  if (!statsFileName.empty())
151  {
152  auto seed = runParams->getAttributeValue<int>("SEED");
153  NOMAD::completeFileName(statsFileName, problemDir, addSeedToFileNames, seed);
154  statsFileParam.replace(0, statsFileName);
155  setAttributeValue("STATS_FILE", statsFileParam);
156  }
157 
158  auto evalStatsFileName = getAttributeValueProtected<std::string>("EVAL_STATS_FILE",false) ;
159  if (!evalStatsFileName.empty() && evalStatsFileName.compare("-") != 0 )
160  {
161  auto seed = runParams->getAttributeValue<int>("SEED");
162  NOMAD::completeFileName(evalStatsFileName, problemDir, addSeedToFileNames, seed);
163  setAttributeValue("EVAL_STATS_FILE", evalStatsFileName);
164  }
165 
166  /*------------------------------------------------------*/
167  /* History file */
168  /*------------------------------------------------------*/
169  auto historyFileName = getAttributeValueProtected<std::string>("HISTORY_FILE",false) ;
170  if (!historyFileName.empty())
171  {
172  auto seed = runParams->getAttributeValue<int>("SEED");
173  NOMAD::completeFileName(historyFileName, problemDir, addSeedToFileNames, seed);
174  setAttributeValue("HISTORY_FILE", historyFileName);
175  }
176 
177  /*------------------------------------------------------*/
178  /* Solution file */
179  /*------------------------------------------------------*/
180  auto solutionFileName = getAttributeValueProtected<std::string>("SOLUTION_FILE",false) ;
181  if (!historyFileName.empty())
182  {
183  auto seed = runParams->getAttributeValue<int>("SEED");
184  NOMAD::completeFileName(solutionFileName, problemDir, addSeedToFileNames, seed);
185  setAttributeValue("SOLUTION_FILE", solutionFileName);
186  }
187 
188  _toBeChecked = false;
189 
190 }
191 // End checkAndComply()

196 NOMAD::ArrayOfDouble NOMAD::DisplayParameters::setFormatFromGranularity( const NOMAD::ArrayOfDouble & aod )
197 {
198  size_t n = aod.size();
199  NOMAD::ArrayOfDouble solFormat(n, NOMAD::DISPLAY_PRECISION_STD);
200 
201  // Use GRANULARITY as an ArrayOfDouble.
202  size_t nbDecimals;
203  for ( size_t i=0 ; i < n ; i++ )
204  {
205  if ( aod[i] > 0 )
206  {
207  nbDecimals = aod[i].nbDecimals( );
208  solFormat.set(i, nbDecimals);
209  }
210  }
211  return solFormat;
212 
213 }
