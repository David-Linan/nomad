93 bool NOMAD::SgtelibModelOptimize::runImp()
94 {
95  bool optimizeOk = false;
96  std::string s;
97 
98  auto modelFormulation = _runParams->getAttributeValue<NOMAD::SgtelibModelFormulationType>("SGTELIB_MODEL_FORMULATION");
99  auto evc = NOMAD::EvcInterface::getEvaluatorControl();
100 
101  if (NOMAD::SgtelibModelFormulationType::EXTERN == modelFormulation)
102  {
103  std::string modelDefinition = _runParams->getAttributeValue<std::string>("MODEL_DEFINITION");
104  evc->getEvalParams()->setAttributeValue("BB_EXE", modelDefinition);
105  }
106  else
107  {
108  // Enforce no opportunism and use no cache.
109  auto previousOpportunism = evc->getOpportunisticEval();
110  auto previousUseCache = evc->getUseCache();
111  evc->setOpportunisticEval(false);
112  evc->setUseCache(false);
113 
114  auto modelDisplay = _runParams->getAttributeValue<std::string>("SGTELIB_MODEL_DISPLAY");
115  auto diversification = _runParams->getAttributeValue<NOMAD::Double>("SGTELIB_MODEL_DIVERSIFICATION");
116  auto modelFeasibility = _runParams->getAttributeValue<NOMAD::SgtelibModelFeasibilityType>("SGTELIB_MODEL_FEASIBILITY");
117  double tc = _runParams->getAttributeValue<NOMAD::Double>("SGTELIB_MODEL_SEARCH_EXCLUSION_AREA").todouble();
118 
119  if (nullptr == _modelAlgo)
120  {
121  s = "Error: In SgtelibModelOptimize, need a SgtelibModel parent.";
122  throw NOMAD::Exception(__FILE__, __LINE__, s);
123  }
124 
125  auto ev = std::make_shared<NOMAD::SgtelibModelEvaluator>(
126  evc->getEvalParams(), _modelAlgo, modelDisplay,
127  diversification, modelFeasibility, tc,
128  NOMAD::SubproblemManager::getInstance()->getSubFixedVariable(this));
129 
130  // Replace the EvaluatorControl's evaluator with this one
131  // we just created
132  auto previousEvaluator = evc->setEvaluator(ev);
133  if (nullptr == previousEvaluator)
134  {
135  std::cerr << "Warning: QuadModelOptimize: Could not set MODEL Evaluator" << std::endl;
136  return false;
137  }
138 
139  auto madsStopReasons = std::make_shared<NOMAD::AlgoStopReasons<NOMAD::MadsStopType>>();
140 
141  // Create a Mads step
142  // Parameters for mads (_optRunParams and _optPbParams) are already updated.
143  _mads = std::make_shared<NOMAD::Mads>(this, madsStopReasons, _optRunParams, _optPbParams);
144  _mads->setName(_mads->getName() + " (SgtelibModelOptimize)");
145  _mads->setEndDisplay(false);
146  evc->resetModelEval();
147  _mads->start();
148  optimizeOk = _mads->run();
149  _mads->end();
150  evc->resetModelEval();
151  evc->setEvaluator(previousEvaluator);
152 
153  // Note: No need to check the Mads stop reason: It is not a stop reason
154  // for SgtelibModel.
155 
156  // Get the solutions
157  updateOraclePoints();
158 
159  // Reset opportunism to previous values.
160  evc->setOpportunisticEval(previousOpportunism);
161  evc->setUseCache(previousUseCache);
162  }
163 
164  if (!optimizeOk)
165  {
166  auto sgetlibModelStopReasons = NOMAD::AlgoStopReasons<NOMAD::ModelStopType>::get(_stopReasons);
167  sgetlibModelStopReasons->set(NOMAD::ModelStopType::MODEL_OPTIMIZATION_FAIL);
168  }
169 
170  return optimizeOk;
171 }

179 void NOMAD::SgtelibModelOptimize::setupRunParameters()
180 {
181  _optRunParams = std::make_shared<NOMAD::RunParameters>(*_refRunParams);
182 
183  // Ensure there is no model used in model optimization.
184  _optRunParams->setAttributeValue("SGTELIB_MODEL_SEARCH", false);
185  _optRunParams->setAttributeValue("QUAD_MODEL_SEARCH", false);
186 
187  // Set direction type to Ortho 2n
188  _optRunParams->setAttributeValue("DIRECTION_TYPE",NOMAD::DirectionType::ORTHO_2N);
189 
190  // Use isotropic mesh
191  _optRunParams->setAttributeValue("ANISOTROPIC_MESH", false);
192 
193  auto evcParams = NOMAD::EvcInterface::getEvaluatorControl()->getEvaluatorControlGlobalParams();
194  std::string lhstr = std::to_string(int(evcParams->getAttributeValue<size_t>("SGTELIB_MODEL_MAX_EVAL") * 0.3));
195  lhstr += " 0";
196  NOMAD::LHSearchType lhSearch(lhstr);
197  _optRunParams->setAttributeValue("LH_SEARCH", lhSearch);
198 
199  // No hMax in the context of SgtelibModel
200  _optRunParams->setAttributeValue("H_MAX_0", NOMAD::Double(NOMAD::INF));
201 
202  // Disable user calls
203  _optRunParams->setAttributeValue("USER_CALLS_ENABLED", false);
204 
205  _optRunParams->checkAndComply(evcParams, _pbParams);
206 }

209 void NOMAD::SgtelibModelOptimize::setupPbParameters(const NOMAD::ArrayOfDouble& lowerBound,
210  const NOMAD::ArrayOfDouble& upperBound)
211 {
212  _optPbParams = std::make_shared<NOMAD::PbParameters>(*_refPbParams);
213 
214  _optPbParams->setAttributeValue("LOWER_BOUND", lowerBound);
215  _optPbParams->setAttributeValue("UPPER_BOUND", upperBound);
216 
217  // Reset initial mesh and frame sizes
218  // The initial mesh and frame sizes will be calculated from bounds and X0
219  _optPbParams->resetToDefaultValue("INITIAL_MESH_SIZE");
220  _optPbParams->resetToDefaultValue("INITIAL_FRAME_SIZE");
221  // Use default min mesh and frame sizes
222  _optPbParams->resetToDefaultValue("MIN_MESH_SIZE");
223  _optPbParams->resetToDefaultValue("MIN_FRAME_SIZE");
224 
225  // Granularity is set to 0 and bb_input_type is set to all continuous variables. Candidate points are projected on the mesh before evaluation.
226  _optPbParams->resetToDefaultValue("GRANULARITY");
227  _optPbParams->resetToDefaultValue("BB_INPUT_TYPE");
228 
229  // No variable groups are considered for suboptimization
230  _optPbParams->resetToDefaultValue("VARIABLE_GROUP");
231 
232  // Find best points (MODEL evals) and use them as X0s to optimize models.
233  NOMAD::CacheInterface cacheInterface(this);
234  std::vector<NOMAD::EvalPoint> evalPointFeasList;
235  std::vector<NOMAD::EvalPoint> evalPointInfList;
236  NOMAD::Double hMax = _modelAlgo->getHMax();
237 
238  // Only looking into model evaluations here
239  cacheInterface.findBestFeas(evalPointFeasList,
240  NOMAD::EvalType::MODEL,
241  NOMAD::ComputeType::STANDARD,
242  nullptr);
243  cacheInterface.findBestInf(evalPointInfList,
244  hMax,
245  NOMAD::EvalType::MODEL,
246  NOMAD::ComputeType::STANDARD,
247  nullptr);
248 
249  NOMAD::ArrayOfPoint x0s;
250  for (auto evalPointX0 : evalPointFeasList)
251  {
252  if (evalPointX0.inBounds(lowerBound, upperBound))
253  {
254  x0s.push_back(*(evalPointX0.getX()));
255  }
256  }
257  for (auto evalPointX0 : evalPointInfList)
258  {
259  if (evalPointX0.inBounds(lowerBound, upperBound))
260  {
261  x0s.push_back(*(evalPointX0.getX()));
262  }
263  }
264  // Fallback: No MODEL points found. Use points from an upper Mads barrier
265  // (_barrierForX0s).
266  if (0 == x0s.size())
267  {
268  // Get best points from upper Mads
269  for (auto evalPointX0 : _modelAlgo->getX0s())
270  {
271  if (evalPointX0.inBounds(lowerBound, upperBound))
272  {
273  x0s.push_back(*(evalPointX0.getX()));
274  }
275  }
276  }
277  _optPbParams->setAttributeValue("X0", x0s);
278 
279  // We do not want certain warnings appearing in sub-optimization.
280  _optPbParams->doNotShowWarnings();
281 
282  _optPbParams->checkAndComply();
283 
284 }

287 // Oracle points are the best feasible and infeasible points found by
288 // running Mads member. Get them using the barrier.
289 void NOMAD::SgtelibModelOptimize::updateOraclePoints()
290 {
291  _oraclePoints.clear();
292 
293  std::shared_ptr<NOMAD::Barrier> barrier;
294  if (nullptr != _mads && nullptr != _mads->getMegaIteration())
295  {
296  barrier = _mads->getMegaIteration()->getBarrier();
297  }
298 
299  if (barrier)
300  {
301  auto allBestPoints = barrier->getAllPoints();
302 
303  for (auto evalPoint : allBestPoints)
304  {
305  _oraclePoints.insert(evalPoint);
306  }
307  }
308 
309 }
