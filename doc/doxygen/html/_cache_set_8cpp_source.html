181 // Get EvalPoint evalPoint at Point x from the cache
182 // Returns the number of EvalPoints found
183 size_t NOMAD::CacheSet::find(const NOMAD::Point& x, NOMAD::EvalPoint &evalPoint,
184  const NOMAD::EvalType evalType) const
185 {
186  size_t nbFound = 0;
187 
188  NOMAD::EvalPointSet::const_iterator it;
189 #ifdef _OPENMP
190  omp_set_lock(&_cacheLock);
191 #endif // _OPENMP
192  it = _cache.find(NOMAD::EvalPoint(x));
193 #ifdef _OPENMP
194  omp_unset_lock(&_cacheLock);
195 #endif // _OPENMP
196  if (it != _cache.end())
197  {
198 #ifdef _OPENMP
199  // Wait for evaluation:
200  // If using OpenMP, the EvalPoint may be updated by another thread.
201  // Otherwise, do not wait.
202  if ((NOMAD::EvalType::UNDEFINED != evalType) && (omp_get_num_threads() > 1))
203  {
204  auto evalStatus = it->getEvalStatus(evalType);
205  while (!_stopWaiting
206  && (NOMAD::EvalStatusType::EVAL_IN_PROGRESS == evalStatus
207  || NOMAD::EvalStatusType::EVAL_NOT_STARTED == evalStatus
208  || NOMAD::EvalStatusType::EVAL_STATUS_UNDEFINED == evalStatus))
209  {
210  usleep(10);
211  evalStatus = it->getEvalStatus(evalType);
212  }
213  }
214 #endif // _OPENMP
215  nbFound = 1;
216  evalPoint = *it;
217  }
218  return nbFound;
219 }

222 // Insert evalPoint in cache.
223 // Return a boolean indicating if we should eval this point.
224 // If insertion worked, the point was not in the cache before. Return true.
225 // If insertion did not work, the point was in the cache before.
226 // Depending on its EvalStatus, return true if it should be evaluated again,
227 // false otherwise.
228 bool NOMAD::CacheSet::smartInsert(const NOMAD::EvalPoint &evalPoint,
229  const short maxNumberEval,
230  const NOMAD::EvalType& evalType)
231 {
232  verifyPointComplete(evalPoint);
233  verifyPointSize(evalPoint);
234 
235  // First insert sets n (even if insert fails)
236  if (0 == _cache.size())
237  {
238  _n = evalPoint.size();
239  }
240 
241  bool inserted = false;
242  bool doEval;
243  std::pair<NOMAD::EvalPointSet::iterator,bool> ret; // Return of the insert()
244 #ifdef _OPENMP
245  omp_set_lock(&_cacheLock);
246 #endif // _OPENMP
247  ret = _cache.insert(evalPoint);
248 #ifdef _OPENMP
249  omp_unset_lock(&_cacheLock);
250 #endif // _OPENMP
251  inserted = ret.second;
252  bool canEval = (*ret.first).toEval(maxNumberEval, evalType);
253  if (inserted && -1 == evalPoint.getTag())
254  {
255  ret.first->updateTag();
256  }
257  evalPoint.setTag(ret.first->getTag());
258 
259  if (inserted && canEval)
260  {
261  doEval = true;
262  }
263  else if (nullptr == (ret.first)->getEval(evalType))
264  {
265  // Point already inserted, but not evaluated.
266  // NOTE: We do not know if this point is in the evaluation queue yet, or not.
267  // We do not know here if the evaluation queue is cleared between runs.
268  // If doEval is set to true, the point could be evaluated twice.
269  // If doEval is set to false, there might be cases where it is not evaluated at all.
270 
271  // Only warn when in blackbox context.
272  if (NOMAD::EvalType::BB == evalType)
273  {
274  OUTPUT_INFO_START
275  std::string s = "Point already inserted in cache, but not evaluated: ";
276  s += ret.first->display();
277  NOMAD::OutputQueue::Add(s, NOMAD::OutputLevel::LEVEL_INFO);
278  OUTPUT_INFO_END
279 
280  // Avoid re-evaluating BB.
281  //doEval = false;
282  // TODO This will have to be re-assessed.
283  doEval = canEval;
284  }
285  else
286  {
287  // It is ok to re-evaluate MODEL points.
288  doEval = true;
289  }
290  }
291  else
292  {
293  // Cache hit.
294  doEval = canEval;
295 
296  // Only count as cache hit when using Blackbox Eval.
297  if (!inserted && NOMAD::EvalType::BB == evalType)
298  {
299  _nbCacheHits++;
300  OUTPUT_INFO_START
301  std::string s = "Cache hit: ";
302  s += ret.first->display();
303  NOMAD::OutputQueue::Add(s, NOMAD::OutputLevel::LEVEL_INFO);
304  OUTPUT_INFO_END
305  }
306  if (doEval)
307  {
308  std::cerr << "Warning: CacheSet: smartInsert: New evaluation of point found in cache " << (*ret.first).display() << std::endl;
309  }
310  }
311 
312  return doEval;
313 }

332 size_t NOMAD::CacheSet::find(const NOMAD::Eval &refeval,
333  std::function<bool(const NOMAD::Eval&, const NOMAD::Eval&, const NOMAD::ComputeType&)> comp,
334  std::vector<NOMAD::EvalPoint> &evalPointList,
335  const NOMAD::EvalType& evalType,
336  const NOMAD::ComputeType& computeType) const
337 {
338  evalPointList.clear();
339  NOMAD::EvalPointSet::const_iterator it;
340  for (it = _cache.begin(); it != _cache.end(); ++it)
341  {
342  const NOMAD::Eval* eval = it->getEval(evalType);
343  if (nullptr == eval)
344  {
345  continue;
346  }
347  if (comp(*eval, refeval, computeType))
348  {
349  NOMAD::EvalPoint evalPoint(*it);
350  evalPointList.push_back(evalPoint);
351  }
352  }
353 
354  return evalPointList.size();
355 }

358 // Get best eval points, using comp()
359 size_t NOMAD::CacheSet::findBest(std::function<bool(const NOMAD::Eval&, const NOMAD::Eval&, const NOMAD::ComputeType&)> comp,
360  std::vector<NOMAD::EvalPoint> &evalPointList,
361  const bool findFeas,
362  const NOMAD::Double& hMax,
363  const NOMAD::Point& fixedVariable,
364  const NOMAD::EvalType& evalType,
365  const NOMAD::ComputeType& computeType,
366  const NOMAD::Eval* refevalIn) const
367 {
368  evalPointList.clear();
369  NOMAD::EvalPointSet::const_iterator it;
370  std::shared_ptr<NOMAD::Eval> refeval = nullptr;
371  if (nullptr != refevalIn)
372  {
373  refeval = std::make_shared<NOMAD::Eval>(*refevalIn);
374  }
375 
376  for (it = _cache.begin(); it != _cache.end(); ++it)
377  {
378  NOMAD::EvalPoint evalPoint(*it);
379  const NOMAD::Eval* eval = evalPoint.getEval(evalType);
380  if (nullptr == eval || NOMAD::EvalStatusType::EVAL_OK != eval->getEvalStatus())
381  {
382  continue;
383  }
384  if (findFeas != eval->isFeasible(computeType))
385  {
386  continue;
387  }
388  if (eval->getH(computeType) > hMax)
389  {
390  continue;
391  }
392  // Must be in the sub-space defined by fixedVariable
393  if (!evalPoint.hasFixed(fixedVariable))
394  {
395  continue;
396  }
397 
398  if (nullptr == refeval)
399  {
400  // Found first point
401  refeval = std::make_shared<NOMAD::Eval>(*eval);
402  evalPointList.push_back(evalPoint);
403  }
404  else if (*eval == *refeval)
405  {
406  // Found a point with eval == refeval
407  evalPointList.push_back(evalPoint);
408  }
409  else if (comp(*eval, *refeval, computeType))
410  {
411  // Found a better point
412  *refeval = *eval;
413  // Reset list with new best
414  evalPointList.clear();
415  evalPointList.push_back(evalPoint);
416  }
417  }
418 
419  return evalPointList.size();
420 }

472 size_t NOMAD::CacheSet::find(const NOMAD::Point & X,
473  std::function<bool(const NOMAD::Point&, const NOMAD::EvalPoint &)> crit,
474  std::vector<NOMAD::EvalPoint> &evalPointList,
475  int maxEvalPoints) const
476 {
477  verifyPointComplete(X);
478  verifyPointSize(X);
479  evalPointList.clear();
480 
481  bool stopWhenMaxFound = (maxEvalPoints > 0);
482  bool errSizeDisplayed = false; // Error about size to be displayed only once.
483  NOMAD::EvalPointSet::const_iterator it;
484  for (it = _cache.begin(); it != _cache.end(); ++it)
485  {
486  if (X.size() != it->size())
487  {
488  if (!errSizeDisplayed)
489  {
490  std::string err = "CacheSet: find: Looking for a point of size ";
491  err += X.size();
492  err += " but the cache points are of size ";
493  err += it->size();
494  std::cerr << "Warning: CacheSet: find: Looking for a point of size " << X.size() << " but found cache point of size " << it->size() << std::endl;
495  errSizeDisplayed = true;
496  }
497  continue; // Points are in different dimensions -skip.
498  }
499 
500  if (crit(X, *it))
501  {
502  NOMAD::EvalPoint evalPoint(*it);
503  evalPointList.push_back(evalPoint);
504  if (stopWhenMaxFound && evalPointList.size() >= (size_t)maxEvalPoints)
505  {
506  break;
507  }
508  }
509  }
510  return evalPointList.size();
511 }

550 // Update EvalPoint in cache.
551 // Look for Point and update the Eval part.
552 // If the point is not found, it is a serious issue.
553 // Don't throw an exception, to avoid a crash, but write a warning.
554 // Returns true if update succeded, false if there was an error.
555 bool NOMAD::CacheSet::update(const NOMAD::EvalPoint& evalPoint, const NOMAD::EvalType& evalType)
556 {
557  bool updateOk = false;
558 
559  if (nullptr == evalPoint.getEval(evalType))
560  {
561  // Cannot update to a null Eval. Warn the user.
562  std::string err = "Warning: CacheSet: Update: Cannot update to a NULL Eval for Point ";
563  err += evalPoint.displayAll();
564  std::cerr << err << std::endl;
565  return false;
566  }
567 
568  NOMAD::EvalPointSet::const_iterator it;
569 #ifdef _OPENMP
570  omp_set_lock(&_cacheLock);
571 #endif // _OPENMP
572  it = _cache.find(evalPoint);
573 #ifdef _OPENMP
574  omp_unset_lock(&_cacheLock);
575 #endif // _OPENMP
576  if (it == _cache.end())
577  {
578  std::string err = "Warning: CacheSet: Update: Did not find EvalPoint to update in cache: " + evalPoint.displayAll();
579  std::cerr << err << std::endl;
580  NOMAD::OutputQueue::Add(err, NOMAD::OutputLevel::LEVEL_WARNING);
581  }
582  else
583  {
584  // Update EvalPoint in cache directly.
585  // Since we are not changing the Point part, which is the only part
586  // used for sorting, the cache should remain coherent.
587  auto cacheEvalPoint = const_cast<NOMAD::EvalPoint*>(&*it);
588  cacheEvalPoint->setEval(*evalPoint.getEval(evalType), evalType);
589  cacheEvalPoint->setNumberEval(evalPoint.getNumberEval());
590  updateOk = true;
591  }
592 
593  return updateOk;
594 }

625 // Purge the cache for space.
626 //
627 // Current implementation:
628 // For now, we only keep EvalPoints which have an f under the mean f.
629 // We should also keep points that are cache hits, or that were recent cache hits.
630 // It could be related to the iteration where the point was added,
631 // or to the number of purges it survived, etc.
632 // We might need a CachePoint class to keep that information.
633 // Etc.
634 //
635 // General idea: We want to keep points that might be hit again, to
636 // avoid having to recompute them.
637 // We also want to keep points that are good enough to be interesting to the
638 // user.
639 //
640 void NOMAD::CacheSet::purge()
641 {
642  std::cerr << "Warning: Calling Cache purge. Size is " << _cache.size() << " max is " << _maxSize << ". Some points will be removed from the cache." << std::endl;
643  if ( _maxSize== NOMAD::INF_SIZE_T || _cache.size() < _maxSize)
644  {
645  // Do nothing
646  return;
647  }
648  size_t nbRemovedLast = 1;
649 
650 #ifdef _OPENMP
651  omp_set_lock(&_cacheLock);
652 #endif // _OPENMP
653 
654  while (_cache.size() >= _maxSize)
655  {
656  NOMAD::EvalPointSet tmpCache;
657  NOMAD::Double meanF;
658  size_t nbElemWithF = computeMeanF(meanF);
659  //std::cout << "Debug: purge: meanF = " << meanF << " nb elem = " << nbElemWithF << std::endl;
660 
661  if (nbElemWithF > 0 && nbRemovedLast > 0)
662  {
663  // Remove all EvalPoints for which f is over or equal to the mean.
664  // For this, use a temporary set/cache, because we
665  // cannot iterate over a set and erase items at the same time.
666  NOMAD::EvalPointSet::const_iterator it;
667  for (it = _cache.begin(); it != _cache.end(); ++it)
668  {
669  if (!it->getF(NOMAD::EvalType::BB, NOMAD::ComputeType::STANDARD).isDefined())
670  {
671  continue;
672  }
673  if (it->getF(NOMAD::EvalType::BB, NOMAD::ComputeType::STANDARD) < meanF)
674  {
675  //std::cout << "Debug: purge: insert EvalPoint with f = " << it->getF(NOMAD::EvalType::BB, NOMAD::ComputeType::STANDARD) << " to tmpCache" << std::endl;
676  tmpCache.insert(*it);
677  }
678  else
679  {
680  //std::cout << "Debug: purge: Do not insert EvalPoint with f = " << it->getF(NOMAD::EvalType::BB, NOMAD::ComputeType::STANDARD) << " to tmpCache" << std::endl;
681  }
682  }
683  }
684  else
685  {
686  // Remove arbitrarly half the elements of cache.
687  // Keep the first half.
688  size_t i = 0;
689  NOMAD::EvalPointSet::const_iterator it;
690  for (it = _cache.begin(); i < _cache.size() / 2; ++it, i++)
691  {
692  tmpCache.insert(*it);
693  }
694  }
695 
696  // If tmpCache is empty, set nbRemovedLast to 0 and the next loop
697  // will go to the "else" case.
698  if (0 == tmpCache.size())
699  {
700  nbRemovedLast = 0;
701  }
702  else
703  {
704  //std::cout << "Debug: Cache has " << _cache.size() << " elements." << std::endl;
705  //std::cout << "Debug: Tmp cache has " << tmpCache.size() << " elements." << std::endl;
706  nbRemovedLast = _cache.size() - tmpCache.size();
707  _cache.clear();
708  _cache = std::move(tmpCache);
709  }
710  }
711 #ifdef _OPENMP
712  omp_unset_lock(&_cacheLock);
713 #endif // _OPENMP
714 }

717 // Naive way to compute the mean f for all points in the cache.
718 size_t NOMAD::CacheSet::computeMeanF(NOMAD::Double &mean) const
719 {
720  size_t nbElem = 0;
721  NOMAD::Double total = 0;
722  mean.reset();
723  NOMAD::EvalPointSet::const_iterator it;
724  for (it = _cache.begin(); it != _cache.end(); ++it)
725  {
726  NOMAD::Double f = it->getF(NOMAD::EvalType::BB, NOMAD::ComputeType::STANDARD);
727  if (f.isDefined())
728  {
729  total += f;
730  nbElem++;
731  }
732  }
733  if (nbElem > 0)
734  {
735  mean = total / nbElem;
736  }
737 
738  return nbElem;
