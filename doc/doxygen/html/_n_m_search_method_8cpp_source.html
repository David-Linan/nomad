54 void NOMAD::NMSearchMethod::init()
55 {
56  if ( _runParams->getAttributeValue<bool>("MEGA_SEARCH_POLL") )
57  {
58  _name = "Search (Nelder Mead single pass)";
59  }
60  else
61  {
62  _name = "Search (Nelder Mead optimization)";
63  }
64  //setComment("(NMSearch)");
65 
66  auto nmSearch = _runParams->getAttributeValue<bool>("NM_SEARCH");
67  setEnabled(nmSearch);
68 
69  if (nmSearch)
70  {
71  // Set the lap counter
72  auto nmFactor = _runParams->getAttributeValue<size_t>("NM_SEARCH_MAX_TRIAL_PTS_NFACTOR");
73  auto dim = _pbParams->getAttributeValue<size_t>("DIMENSION");
74  if (nmFactor < NOMAD::INF_SIZE_T)
75  {
76  NOMAD::EvcInterface::getEvaluatorControl()->setLapMaxBbEval( dim*nmFactor );
77  }
78  }
79 }

82 bool NOMAD::NMSearchMethod::runImp()
83 {
84  // NM is an algorithm with its own stop reasons.
85  auto nmStopReasons = std::make_shared<NOMAD::AlgoStopReasons<NOMAD::NMStopType>>();
86 
87  // Create the NM algorithm with its own stop reason
88  auto nm = std::make_shared<NOMAD::NM>(this,
89  nmStopReasons ,
90  _runParams,
91  _pbParams);
92  nm->setEndDisplay(false);
93 
94  nm->start();
95  bool foundBetter = nm->run();
96  nm->end();
97 
98  return foundBetter;
99 }

102 void NOMAD::NMSearchMethod::generateTrialPointsImp()
103 {
104  // The trial points of one iteration of NM reflective steps are generated (not evaluated).
105  // The trial points are Reflect, Expansion, Inside and Outside Contraction NM points
106 
107  auto madsIteration = getParentOfType<MadsIteration*>();
108 
109  // Note: Use first point of barrier as simplex center.
110  NOMAD::NMAllReflective allReflective(this,
111  std::make_shared<NOMAD::EvalPoint>(getMegaIterationBarrier()->getFirstPoint()),
112  madsIteration->getMesh());
113  allReflective.start();
114  allReflective.end();
115 
116  // Pass the generated trial pts to this
117  auto trialPtsNM = allReflective.getTrialPoints();
118  for (auto point : trialPtsNM)
119  {
120  insertTrialPoint(point);
121  }
122 
123 }
