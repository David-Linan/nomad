57 void NOMAD::SSDMadsMegaIteration::startImp()
58 {
59 
60  // Update manager mesh and barrier.
61  NOMAD::MadsUpdate update( this );
62  update.start();
63  update.run();
64  update.end();
65 
66  // Now that update has used the previous MegaIteration success type, reset it
67  setSuccessType(NOMAD::SuccessType::NOT_EVALUATED);
68 
69  // Verify mesh stop conditions.
70  _mainMesh->checkMeshForStopping( _stopReasons );
71 
72  OUTPUT_DEBUG_START
73  AddOutputDebug("Mesh Stop Reason: " + _stopReasons->getStopReasonAsString());
74  OUTPUT_DEBUG_END
75  if ( ! _stopReasons->checkTerminate() )
76  {
77  auto bestEvalPoint = _barrier->getFirstXFeas();
78 
79  if (bestEvalPoint == nullptr)
80  bestEvalPoint = _barrier->getFirstXInf();
81 
82  if (bestEvalPoint == nullptr)
83  throw NOMAD::Exception(__FILE__, __LINE__, "No best eval point");
84 
85  auto nbMadsSubproblem = _runParams->getAttributeValue<size_t>("SSD_MADS_NB_SUBPROBLEM");
86 
87  for (size_t nbMads = 0; nbMads < nbMadsSubproblem; nbMads++)
88  {
89  // The stop reasons for a subproblem mads
90  auto stopReasons = std::make_shared<NOMAD::AlgoStopReasons<NOMAD::MadsStopType>>();
91 
92  auto subProblemPbParams = std::make_shared<NOMAD::PbParameters>(*_pbParams);
93 
94  auto subProblemRunParams = std::make_shared<NOMAD::RunParameters>(*_runParams);
95 
96  bool isPollsterWorker = (nbMads==0)? true:false;
97 
98  setupSubproblemParams (subProblemPbParams, subProblemRunParams, *(bestEvalPoint->getX()), isPollsterWorker );
99 
100  // Second check, first check is done during setup
101  subProblemPbParams->checkAndComply();
102 
103  auto evcParams = NOMAD::EvcInterface::getEvaluatorControl()->getEvaluatorControlGlobalParams();
104 
105  subProblemRunParams->checkAndComply(evcParams, subProblemPbParams);
106 
107  auto madsOnASubProblem = std::make_shared<NOMAD::Mads>(this , stopReasons, subProblemRunParams, subProblemPbParams );
108  _madsList.push_back(madsOnASubProblem);
109  }
110 
111 
112  OUTPUT_INFO_START
113  AddOutputInfo(_name + " has " + std::to_string(nbMadsSubproblem) + " subproblem mads.");
114  OUTPUT_INFO_END
115  }
116 }

120 bool NOMAD::SSDMadsMegaIteration::runImp()
121 {
122  NOMAD::SuccessType bestSuccessYet = NOMAD::SuccessType::NOT_EVALUATED;
123  NOMAD::SuccessType subPbSuccess = SuccessType::NOT_EVALUATED;
124 
125  std::string s;
126 
127  if ( _stopReasons->checkTerminate() )
128  {
129  OUTPUT_DEBUG_START
130  s = "SSDMadsMegaIteration: stopReason = " + _stopReasons->getStopReasonAsString() ;
131  AddOutputDebug(s);
132  OUTPUT_DEBUG_END
133  return false;
134  }
135 
136  if (_madsList.empty())
137  {
138  throw NOMAD::Exception(__FILE__, __LINE__, "No mads on subproblem to run");
139  }
140 
141  // Get Mads ancestor to call terminate(k)
142  NOMAD::SSDMads* ssdmads = getParentOfType<NOMAD::SSDMads*>();
143  if (nullptr == ssdmads)
144  {
145  throw NOMAD::Exception(__FILE__, __LINE__, "SSDMads Iteration without SSDMads ancestor");
146  }
147 
148  // Reset counter for bb eval in subproblem for this sub-optimization
149  auto evc = NOMAD::EvcInterface::getEvaluatorControl();
150  evc->resetBbEvalInSubproblem();
151 
152  for (size_t i = 0; i < _madsList.size(); i++)
153  {
154  if (_stopReasons->checkTerminate()
155  || evc->testIf(NOMAD::EvalMainThreadStopType::OPPORTUNISTIC_SUCCESS)
156  || ssdmads->terminate(i))
157  {
158  break;
159  }
160 
161  // downcast from Iteration to MadsIteration
162  std::shared_ptr<NOMAD::Mads> madsOnSubPb = std::dynamic_pointer_cast<NOMAD::Mads>(_madsList[i]);
163 
164  if (madsOnSubPb == nullptr)
165  {
166  throw NOMAD::Exception(__FILE__, __LINE__, "Invalid shared pointer cast");
167  }
168 
169  madsOnSubPb->start();
170 
171  bool iterSuccessful = madsOnSubPb->run(); // Is this iteration successful
172  madsOnSubPb->end();
173 
174  // Compute succes
175  NOMAD::EvalPointPtr newBestFeas,newBestInf;
176 
177  // Use mega iteration barrier to get the reference best points (should work for opportunistic or not)
178  auto refBestFeas = _barrier->getFirstXFeas();
179  auto refBestInf = _barrier->getFirstXInf();
180 
181  // Set the iter success based on best feasible and best infeasible points found compared to initial point.
182  if (iterSuccessful && (nullptr != refBestFeas || nullptr != refBestInf))
183  {
184  // Use the mads barrier to access the new best points
185  auto barrier = madsOnSubPb->getMegaIterationBarrier();
186  if (nullptr != barrier)
187  {
188  newBestFeas = barrier->getFirstXFeas();
189  newBestInf = barrier->getFirstXInf();
190 
191  // Compute success
192  // Get which of newBestFeas and newBestInf is improving
193  // the solution. Check newBestFeas first.
194  NOMAD::ComputeSuccessType computeSuccess(evc->getEvalType(), evc->getComputeType());
195  subPbSuccess = computeSuccess(newBestFeas, refBestFeas);
196  if (subPbSuccess >= NOMAD::SuccessType::PARTIAL_SUCCESS)
197  {
198  // newBestFeas is the improving point.
199  OUTPUT_DEBUG_START
200  // Output Warning: When using '\n', the computed indentation for the
201  // Step will be ignored. Leaving it like this for now. Using an
202  // OutputInfo with AddMsg() would resolve the output layout.
203  s = "Update: improving feasible point";
204  if (refBestFeas)
205  {
206  s += " from\n " + refBestFeas->display() + "\n";
207  }
208  s += " to " + newBestFeas->display();
209  AddOutputDebug(s);
210  OUTPUT_DEBUG_END
211  }
212  else
213  {
214  // Check newBestInf
215  NOMAD::SuccessType success2 = computeSuccess(newBestInf, refBestInf);
216  if (success2 > subPbSuccess)
217  {
218  subPbSuccess = success2;
219  }
220  if (subPbSuccess >= NOMAD::SuccessType::PARTIAL_SUCCESS)
221  {
222  OUTPUT_DEBUG_START
223  s = "Update: improving infeasible point";
224  if (refBestInf)
225  {
226  s+= " from\n " + refBestInf->display() + "\n";
227  }
228  s += " to " + newBestInf->display();
229  AddOutputDebug(s);
230  OUTPUT_DEBUG_END
231  }
232  }
233  if (subPbSuccess == NOMAD::SuccessType::UNSUCCESSFUL)
234  {
235  OUTPUT_DEBUG_START
236  s = "Update: no success found";
237  AddOutputDebug(s);
238  OUTPUT_DEBUG_END
239  }
240  }
241  }
242  if (subPbSuccess > bestSuccessYet)
243  {
244  bestSuccessYet = subPbSuccess;
245  }
246 
247  //
248  // Transfer the Mads subproblem barrier into the mega iteration barrier
249  //
250  // The fixed variables of the mads subproblem
251  auto fixedVariable = NOMAD::SubproblemManager::getInstance()->getSubFixedVariable((madsOnSubPb.get()));
252  auto evalPointList = madsOnSubPb->getMegaIterationBarrier()->getAllPoints();
253  // Convert into full dimension
254  NOMAD::convertPointListToFull(evalPointList, fixedVariable);
255  _barrier->updateWithPoints(evalPointList, NOMAD::EvalType::BB, evc->getComputeType(), true);
256 
257  // Reset counter will reset the EvalMainThreadStopReason if the max bb is reached for this sub optimization
258  evc->resetBbEvalInSubproblem();
259 
260  // Set the stop reason for opportunistic success of a subproblem mads
261  const bool opportunisticItStop = _runParams->getAttributeValue<bool>("SSD_MADS_ITER_OPPORTUNISTIC");
262  if (opportunisticItStop && (bestSuccessYet == NOMAD::SuccessType::FULL_SUCCESS))
263  {
264  evc->setStopReason(NOMAD::getThreadNum(), NOMAD::EvalMainThreadStopType::OPPORTUNISTIC_SUCCESS);
265  }
266 
267 
268 // Display MegaIteration's stop reason
269  if (_stopReasons->checkTerminate())
270  {
271  OUTPUT_DEBUG_START
272  s = _name + " stop reason set to: " + _stopReasons->getStopReasonAsString();
273  AddOutputDebug(s);
274  OUTPUT_DEBUG_END
275  }
276 
277  if (_userInterrupt)
278  {
279  hotRestartOnUserInterrupt();
280  }
281 
282  }
283 
284 
285  // Set success of the mega iteration
286  setSuccessType(bestSuccessYet);
287 
288  // return true if we have a partial or full success.
289  return (bestSuccessYet >= NOMAD::SuccessType::PARTIAL_SUCCESS);
290 }
