120 void NOMAD::trim(std::string &s)
121 {
122  // Trim extra spaces at the beginning
123  size_t space_index = s.find(' ');
124  while (s.size() > 0 && 0 == space_index)
125  {
126  s.replace(0, 1, "");
127  space_index = s.find(' ');
128  }
129 
130  // Trim extra spaces at the end
131  size_t space_rindex = s.rfind(' ');
132  while (s.size() > 0 && s.size()-1 == space_rindex)
133  {
134  s.replace(space_rindex, 1, "");
135  space_rindex = s.rfind(' ');
136  }
137 }

143 bool NOMAD::atoi ( const std::string & s , int & i )
144 {
145  i = -1;
146  if ( s.empty() )
147  {
148  return false;
149  }
150 
151  size_t n = s.size();
152 
153  if ( s[0] == '-' )
154  {
155  if ( n > 1 && s[1] == '-' )
156  return false;
157  std::string ss = s;
158  ss.erase(ss.begin());
159  if ( NOMAD::atoi ( ss , i ) )
160  {
161  i = -i;
162  return true;
163  }
164  return false;
165  }
166 
167  std::string ts = s;
168  NOMAD::toupper(ts);
169  if ( ts == "INF" || ts == "+INF" )
170  {
171  i = NOMAD::P_INF_INT;
172  return true;
173  }
174  if ( ts == "-INF" )
175  {
176  i = NOMAD::M_INF_INT;
177  return true;
178  }
179 
180  for ( size_t k = 0 ; k < n ; ++k )
181  {
182  if ( !isdigit(s[k]) )
183  {
184  return false;
185  }
186  }
187  i = std::atoi(s.c_str());
188  return true;
189 }

291

300  std::istringstream in (s);

302

308  size_t k , n1 = s1.size();
309 
310  if ( n1 >= s.size() - 1 )

312  for ( k = 0 ; k < n1 ; ++k )
313  if (!isdigit(s1[k]))
314  return false;
315  if ( ! atost ( s1 , i ) )
316  return false;
317  if ( n1 == s.size() )
318  {
319  j = i;
320  return true;
321  }
322  return false;
323  }
324 
325  std::string s2;
326  getline (in, s2);
327 
328  if (in.fail())
329  return false;
330 
331  size_t n2 = s2.size();
332  for ( k = 0 ; k < n2 ; ++k )
333  if ( !isdigit(s2[k]) )
334  return false;
335 
336  if ( ! atost ( s1, i ) || ! atost ( s2 , j ) )
337  return false;
338 
339  return !check_order || i <= j;

396 // s is a string representing a double.
397 // prec = number of digits to show after the point.
398 // spacePadding = recommended space padding after s.
399 // width = total width
400 // Ex. 5.123456789 with a prec of 6 shows as 5.123456.
401 // Ex. 5.12 with a prec of 6 will return as "5.12 " (with padding).
402 void NOMAD::getFormat(const std::string &s, const size_t prec, size_t &width, size_t &spacePadding)
403 {
404  // These values non modifiable for now.
405  const size_t nbDigitsBeforePoint = NOMAD::NB_DIGITS_BEFORE_POINT;
406  const size_t intWidth = NOMAD::INT_DISPLAY_WIDTH;
407 
408  if (0 == prec)
409  {
410  // Integer.
411  width = intWidth;
412  }
413  else
414  {
415  // Double
416  width = nbDigitsBeforePoint + 1 + prec;
417  size_t pointPos = s.find(".");
418  spacePadding = 1 + prec; // Including one space for the decimal point
419  if (pointPos != std::string::npos)
420  {
421  // Compute padding as needed.
422  // Ex. if precision is 6 but the string number is "27.03" -> pad to "27.03 ".
423  spacePadding -= (s.size() - pointPos);
424  if (spacePadding >= width)
425  {
426  spacePadding = 0;
427  }
428  }
429  }
430 }

433 bool NOMAD::separateFormat(const std::string &s, std::string &format, std::string &tag)
434 {
435  // NOTE: format found here is used in StatsInfo, for display of OBJ, CONS_H
436  // and H_MAX (or any Double). It is not used for values of type ArrayOfDouble
437  // or its derived classes.
438 
439  format = "";
440  tag = s;
441  const std::string formatLetters = "eEfgGdi";
442  const std::string allLetters = "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ";
443  bool doSeparate = false;
444 
445  if ('%' == s[0])
446  {
447  std::size_t firstLetter = s.find_first_of(allLetters, 1);
448  std::size_t firstFormatLetter = s.find_first_of(formatLetters, 1);
449  std::size_t cutIndex = firstLetter;
450  if (std::string::npos != firstFormatLetter
451  && firstLetter == firstFormatLetter)
452  {
453  cutIndex += 1;
454  }
455  if (std::string::npos != cutIndex)
456  {
457  std::string tempFormat = s.substr(0, cutIndex);
458  doSeparate = validFormat(tempFormat);
459  if (doSeparate)
460  {
461  format = tempFormat;
462  tag = s.substr(cutIndex, s.length() - cutIndex);
463  }
464  }
465  }
466 
467  return doSeparate;
468 }

480 // e Scientific notation (mantise/exponent) using e character 3.9265e+2
481 // E Scientific notation (mantise/exponent) using E character 3.9265E+2
482 // f Decimal floating point 392.65

485 // d or i Integer rounded value 393

487 // %4.2 is also valid (no format letter). Append f.
488 bool NOMAD::validFormat(std::string &s)
489 {
490  const std::string formatLetters = "eEfgGdi";
491  bool isValid = true;
492 
493  // Verify size
494  if (s.length() < 2)
495  {
496  isValid = false;
497  }
498 
499  else
500  {
501  // Append f if last char is a digit.
502  if (std::isdigit(s[s.length()-1]))
503  {
504  s = s + "f";
505  }
506  size_t indexFormatLetter = s.find_first_of(formatLetters, 1);
507 
508  // Verify we do have a format letter
509  if (std::string::npos == indexFormatLetter)
510  {
511  isValid = false;
512  }
513  // Verify first character is '%'
514  else if (s[0] != '%')
515  {
516  isValid = false;
517  }
518  // Verify last character is a formatting character
519  else if (indexFormatLetter < s.length()-1)
520  {
521  isValid = false;
522  }
523  // Verify in between we have a decimal number
524  else
525  {
526  bool pointEncountered = false;
527  for (size_t i = 1; i < indexFormatLetter; i++)
528  {
529  if (std::isdigit(s[i]))
530  {
531  // ok
532  }
533  else if (s[i] == '.')
534  {
535  if (!pointEncountered)
536  {
537  pointEncountered = true;
538  }
539  else
540  {
541  // more than one point
542  isValid = false;
543  }
544  }
545  else
546  {
547  isValid = false;
548  }
549  }
550  }
551  }
552 
553  return isValid;
554 }
