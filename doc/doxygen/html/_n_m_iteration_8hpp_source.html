55 /// Class for Nelder Mead (NM) iterations
56 /**
57  The start function for a Nelder Mead iteration creates the initial simplex using a frame ("simplex") center and the points in cache. \n
58  The run function of this class iterates between the different reflective step
59  (REFLECT, EXPAND, INSIDE_CONTRACTION, OUTSIDE_CONTRACTION) and the SHRINK step
60  if it is required. The function also updates the type of success to pass to the NMMegaIteration (if it exists) and manages the stop reason.
61  */
62 class NMIteration: public Iteration
63 {
64 private:
65  /// Helper for constructor
66  void init ();
67 
68  /**
69  The simplex is shared among Nelder Mead components. The simplex is initially built by NMInitializeSimplex::run
70  */
71  std::shared_ptr<NMSimplexEvalPointSet> _nmY ;
72 
73  /**
74  The simplex "center" at the creation of this iteration.
75  The initial simplex is built around this point.
76  The frame center of MADS is used when Nelder Mead is used for a search method of MADS.
77  */
78  const std::shared_ptr<EvalPoint> _simplexCenter;
79 
80  /**
81  The Mads mesh can be available if Nelder Mead is used as a Search method. If not, it is set to \c nullptr. When available, trials can be projected on it.
82  */
83  const std::shared_ptr<MeshBase> _madsMesh;
84 
85  SuccessType _bestSuccess; ///< The best success obtained during the Nelder Mead iterations.
86 
87 public:
88  /// Constructor
89  /**
90  \param parentStep The parent of this step -- \b IN.
91  \param frameCenter The frame center -- \b IN.
92  \param k The iteration number -- \b IN.
93  \param madsMesh Mads Mesh for trial point projection (can be null) -- \b IN.
94  */
95  explicit NMIteration(const Step *parentStep,
96  const std::shared_ptr<EvalPoint> &frameCenter,
97  const size_t k,
98  std::shared_ptr<MeshBase> madsMesh)
99  : Iteration(parentStep, k),
100  _simplexCenter(frameCenter),
101  _madsMesh(madsMesh)
102  {
103  init();
104 
105  // Create an empty simplex to be shared among Nelder Mead components
106  _nmY = std::make_shared<NMSimplexEvalPointSet>();
107  }
108 
109 
110  // Get/Set
111 
112  const std::shared_ptr<EvalPoint> getSimplexCenter() const { return _simplexCenter ; }
113 
114  const std::shared_ptr<MeshBase> getMesh() const override { return _madsMesh; }
115 
116  const std::shared_ptr<NMSimplexEvalPointSet> getY( void ) const { return _nmY; }
117 
118 protected:
119 
120  /// Implementation of run task.
121  /**
122  Sequential run of Nelder Mead steps among INITIAL, ( REFLECT, EXPANSION, INSIDE_CONTRACTION, OUTSIDE_CONTRACTION ), SHRINK.
123  Set the stop reason and also updates the Nelder Mead mega iteration success.
124  */
125  virtual bool runImp() override ;
126 
127  /// Implementation of start task.
128  /**
129  Update the barrier and create the initial simplex if it is empty.
130  */
131  virtual void startImp() override;
132 
133 
134 };
