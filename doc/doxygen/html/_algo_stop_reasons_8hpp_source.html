57 /// Template class for algorithm stop reasons.
58 /**
59 
60  The class is templated with a StopType defined according to which algorithm is considered. For example, we have ::MadsStopType, ::LHStopType and ::NMStopType. \n
61  At some point during an algorithm a stop reason is set. It can be specific to the algorithm or generic (that is be an AllStopReasons stop type). \n
62  The stop reasons in AllStopReasons are private and not directly accessible. But the AlgoStopReasons::checkTerminate() function, checks both AllStopReasons and AlgoStopReasons.
63  */
64 template <typename StopType>
65 class AlgoStopReasons : public AllStopReasons

91  std::string getStopReasonAsString() const override
92  {
93  std::string stopReasonStr = AllStopReasons::getStopReasonAsString();
94 
95  if (!_algoStopReason.isStarted())
96  {
97  stopReasonStr += _algoStopReason.getStopReasonAsString() + " (Algo)";
98  }
99 
100  auto evc = NOMAD::EvcInterface::getEvaluatorControl();
101  if (nullptr != evc)
102  {
103  int mainThreadNum = NOMAD::getThreadNum(); // The code in Algos/ is always called from a main thread.
104  auto evalStopReason = evc->getStopReason(mainThreadNum);
105  if (!evalStopReason.isStarted())
106  {
107  stopReasonStr += (stopReasonStr.empty() ? "" : " ") + evalStopReason.getStopReasonAsString();
108  }
109  }
110 
111  return stopReasonStr;
112  }
113 
114 
115  /// Check among generic stop reasons and algo stop reason if the algorithm must terminate
116  bool checkTerminate() const override
117  {
118  auto evc = NOMAD::EvcInterface::getEvaluatorControl();
119  return (NOMAD::AllStopReasons::checkTerminate()
120  || _algoStopReason.checkTerminate()
121  || ((nullptr != evc) && evc->getStopReason(NOMAD::getThreadNum()).checkTerminate()));
122  }
123 
124 
125  /// Access to the AlgoStopReasons
126  static std::shared_ptr<AlgoStopReasons<StopType>> get(std::shared_ptr<AllStopReasons> allStopReasons)
127  {
128  std::shared_ptr<AlgoStopReasons<StopType>> stopReasons = std::dynamic_pointer_cast<AlgoStopReasons<StopType>>( allStopReasons );
129 
130  if ( stopReasons == nullptr )
131  throw Exception(__FILE__, __LINE__, "Invalid shared pointer cast");
132  return stopReasons;
133  }
134 
135 
136  /// Test for a specific algorithm stop type.
137  /**
138  Used to pass a sub-algorithm stop reason to a parent algorithm stop reason.
139  */
140  bool testIf (const StopType& s) const
141  {
142  return ( _algoStopReason.get() == s );
143  }
144 
145 
146  /// Reset stop reasons to their default STARTED state.
147  void setStarted() override
148  {
149  _algoStopReason.setStarted();
150  AllStopReasons::setStarted();
151  auto evc = NOMAD::EvcInterface::getEvaluatorControl();
152  if (nullptr != evc)
153  {
154  evc->setStopReason(NOMAD::getThreadNum(), NOMAD::EvalMainThreadStopType::STARTED);
155  }
156  }
157 
158 };
