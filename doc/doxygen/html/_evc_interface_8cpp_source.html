96 // For each point, look if it is in the cache.
97 // If it is, remove it from the EvalPointSet.
98 // If not, add it to EvaluatorControl's Queue.
99 void NOMAD::EvcInterface::keepPointsThatNeedEval(const NOMAD::EvalPointSet &trialPoints, bool useMesh)
100 {
101  // Create EvalPoints and send them to EvaluatorControl
102  if (nullptr == _evaluatorControl)
103  {
104  throw NOMAD::StepException(__FILE__,__LINE__, _step->getName() + ": EvaluatorControl not found", _step);
105  }
106 
107  NOMAD::EvalType evalType = _evaluatorControl->getEvalType();
108 
109  // Currently, this method may be used inside an Iteration (Search or Poll, NM, ...),
110  // or inside a MegaSearchPoll.
111  auto iteration = _step->getParentOfType<NOMAD::Iteration*>();
112  auto megaSearchPoll = dynamic_cast<const NOMAD::MegaSearchPoll*>(_step);
113 
114  if (useMesh && nullptr == iteration && nullptr == megaSearchPoll)
115  {
116  throw NOMAD::StepException(__FILE__,__LINE__, _step->getName() + ": In keepPointsThatNeedEval: need a parent of type Iteration or MegaSearchPoll", _step);
117  }
118 
119  if (trialPoints.size() > 0)
120  {
121  OUTPUT_INFO_START
122  _step->AddOutputInfo("Add points to eval queue for step " + _step->getName(), true, false);
123  OUTPUT_INFO_END
124  OUTPUT_DEBUG_START
125  _step->AddOutputDebug(NOMAD::itos(trialPoints.size()) + " points to add to eval queue");
126  OUTPUT_DEBUG_END
127  }
128 
129  for (auto trialPoint : trialPoints)
130  {
131  // Try to insert the trialPoint in the cache.
132  // If it is not in the cache, it will be added.
133  // If the point is already in the cache, depending on its
134  // EvalStatus we might want to evaluate it.
135 
136  // First, convert trial point to full dimension, since we are
137  // now only working with the cache and the EvaluatorControl.
138  auto trialPointSub = trialPoint; // Used to get iteration
139  trialPoint = trialPoint.makeFullSpacePointFromFixed(_fixedVariable);
140 
141  // Compute if we should evaluate, maybe re-evaluate, this point
142  bool doEval = true;
143  // maxNumberEval is used to compute if we should re-evaluate this point.
144  // Default value is 1.
145  // This will be a parameter in the future. Currently not implemented.
146  const int maxNumberEval = 1;
147  if (_evaluatorControl->getUseCache())
148  {
149  doEval = NOMAD::CacheBase::getInstance()->smartInsert(trialPoint, maxNumberEval, evalType);
150  }
151  else
152  {
153  // smartInsert would have taken care of updating tag, but since
154  // cache is not used, update tag here.
155  trialPoint.updateTag();
156  // Look in EvaluatorControl's Barrier if the point is already evaluated.
157  // Only do this when EvalType is BB. If it is MODEL, always reevaluate.
158  if (NOMAD::EvalType::BB == evalType)
159  {
160  auto barrier = _evaluatorControl->getBarrier();
161  if (nullptr != barrier)
162  {
163  NOMAD::EvalPoint foundEvalPoint;
164  // Either point is not in barrier, or
165  // point was evaluated, but not with the current eval type.
166  doEval = !findInList(*trialPoint.getX(), barrier->getAllPoints(), foundEvalPoint)
167  || (nullptr == foundEvalPoint.getEval(evalType));
168  }
169  }
170  }
171 
172  if (doEval)
173  {
174  NOMAD::EvalQueuePointPtr evalQueuePoint(new NOMAD::EvalQueuePoint(trialPoint, evalType));
175  if (useMesh && nullptr == iteration)
176  {
177  std::string s = _step->getName();
178  s += ": In keepPointsThatNeedEval: Could not determine iteration for point ";
179  s += trialPoint.display();
180  throw NOMAD::StepException(__FILE__,__LINE__, s, _step);
181  }
182  if ( useMesh )
183  {
184  auto mesh = iteration->getMesh();
185  if ( mesh != nullptr )
186  {
187  evalQueuePoint->setMeshSize(mesh->getdeltaMeshSize());
188  evalQueuePoint->setFrameSize(mesh->getDeltaFrameSize());
189  evalQueuePoint->setK(iteration->getK());
190  }
191  }
192 
193  // Set a flag in evalQueuePoint according to algo -> this is for stats
194  const Algorithm * algo = _step->getRootAlgorithm();
195  auto algoConstPhaseOne = dynamic_cast<const PhaseOne*>(algo);
196  if(nullptr != algoConstPhaseOne)
197  {
198  evalQueuePoint->setGenByPhaseOne(true);
199  }
200  else
201  {
202  evalQueuePoint->setGenByPhaseOne(false);
203  }
204 
205  evalQueuePoint->setComment(_step->getAlgoComment());
206  evalQueuePoint->setGenStep(_step->getName());
207 
208  if (_evaluatorControl->addToQueue(evalQueuePoint))
209  {
210  OUTPUT_DEBUG_START
211  _step->AddOutputDebug("New point added to eval queue: " + trialPoint.display());
212  OUTPUT_DEBUG_END
213  }
214  else
215  {
216  OUTPUT_DEBUG_START
217  _step->AddOutputDebug("Point not added to eval queue: " + trialPoint.display());
218  OUTPUT_DEBUG_END
219  }
220  }
221  else
222  {
223  // Point already evaluated
224  OUTPUT_DEBUG_START
225  _step->AddOutputDebug("Point not re-evaluated: " + trialPoint.display());
226  OUTPUT_DEBUG_END
227  }
228  }
229 
230  OUTPUT_DEBUG_START
231  size_t evcNbPoints = _evaluatorControl->getQueueSize();
232  if (evcNbPoints > 0)
233  {
234  // Not conditional to trialPoints size:
235  // There could be leftover points to evaluate from previous evaluations.
236  _step->AddOutputDebug("Eval queue now has " + NOMAD::itos(evcNbPoints) + " points.");
237  }
238  OUTPUT_DEBUG_END
239 
240  OUTPUT_INFO_START
241  if (trialPoints.size() > 0)
242  {
243  _step->AddOutputInfo("Add points to eval queue for step " + _step->getName(), false, true);
244  }
245 
246  NOMAD::OutputQueue::Flush();
247  OUTPUT_INFO_END
248 }

251 void NOMAD::EvcInterface::setBarrier(const std::shared_ptr<NOMAD::Barrier>& subBarrier)
252 {
253  if (subBarrier == nullptr)
254  {
255  return;
256  }
257 
258  // Input is the barrier from MegaIteration, which may belong to a subspace.
259  // EvaluatorControl's barrier must be in full dimension.
260  auto fullBarrier = std::make_shared<NOMAD::Barrier>(*subBarrier);
261 
262  // Clear xFeas and xInf lists and recompute them
263  fullBarrier->clearXFeas();
264  fullBarrier->clearXInf();
265  auto evalType = _evaluatorControl->getEvalType();
266  for (auto xFeas : subBarrier->getAllXFeas())
267  {
268  auto xFeasFull = xFeas.makeFullSpacePointFromFixed(_fixedVariable);
269  fullBarrier->addXFeas(xFeasFull, evalType, _evaluatorControl->getComputeType());
270  }
271  for (auto xInf : subBarrier->getAllXInf())
272  {
273  auto xInfFull = xInf.makeFullSpacePointFromFixed(_fixedVariable);
274  fullBarrier->addXInf(xInfFull, evalType);
275  }
276  auto refBestFeas = subBarrier->getRefBestFeas();
277  auto refBestInf = subBarrier->getRefBestInf();
278  if (nullptr != refBestFeas)
279  {
280  fullBarrier->setRefBestFeas(std::make_shared<NOMAD::EvalPoint>(refBestFeas->makeFullSpacePointFromFixed(_fixedVariable)));
281  }
282  if (nullptr != refBestInf)
283  {
284  fullBarrier->setRefBestInf(std::make_shared<NOMAD::EvalPoint>(refBestInf->makeFullSpacePointFromFixed(_fixedVariable)));
285  }
286 
287  _evaluatorControl->setBarrier(fullBarrier);
288 }

291 bool NOMAD::EvcInterface::findInBarrier(const NOMAD::Point& x, NOMAD::EvalPoint& evalPoint) const
292 {
293  bool pointFound = false;
294 
295  auto barrier = _evaluatorControl->getBarrier();
296  if (nullptr != barrier)
297  {
298  auto xFull = x.makeFullSpacePointFromFixed(_fixedVariable);
299  NOMAD::EvalPoint evalPointFull(evalPoint);
300  pointFound = findInList(xFull, barrier->getAllPoints(), evalPointFull);
301  if (pointFound)
302  {
303  // Put found point back in sub-dimension
304  evalPoint = evalPointFull.makeSubSpacePointFromFixed(_fixedVariable);
305  }
306  }
307 
308  return pointFound;
309 }

329 NOMAD::SuccessType NOMAD::EvcInterface::startEvaluation()
330 {
331  OUTPUT_INFO_START
332  _step->AddOutputInfo("Evaluate points for " + _step->getName(), true, false);
333  OUTPUT_INFO_END
334 
335  NOMAD::SuccessType success = NOMAD::SuccessType::UNSUCCESSFUL;
336  std::shared_ptr<NOMAD::AllStopReasons> stopReasons = _step->getAllStopReasons();
337 
338  // Evaluate points
339  // Note: do not use checkTerminate() here. If it is time to terminate, EvaluatorControl will take
340  // care of clearing the queue.
341  success = _evaluatorControl->run();
342 
343  OUTPUT_DEBUG_START
344  std::string s = _step->getName() + ": " + NOMAD::enumStr(success);
345  s += ". Stop reasons: " + stopReasons->getStopReasonAsString() ;
346  _step->AddOutputDebug(s);
347  OUTPUT_DEBUG_END
348 
349  OUTPUT_INFO_START
350  NOMAD::OutputQueue::Flush();
351 
352  _step->AddOutputInfo("Evaluate points for " + _step->getName(), false, true);
353  OUTPUT_INFO_END
354 
355  return success;
356 }
