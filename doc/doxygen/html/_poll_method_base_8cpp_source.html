59 void NOMAD::PollMethodBase::generateTrialPoints()
60 {
61  // Groups of variables.
62  auto varGroups = _pbParams->getAttributeValue<NOMAD::ListOfVariableGroup>("VARIABLE_GROUP");;
63 
64  auto n = _pbParams->getAttributeValue<size_t>("DIMENSION");
65 
66  std::list<NOMAD::Direction> directionsSubSpace, directionsFullSpace;
67 
68  if (varGroups.size() == 0)
69  {
70  // Creation of the poll directions in the full space
71  generateUnitPollDirections(directionsFullSpace,n);
72  }
73  else
74  {
75  for (auto vg : varGroups)
76  {
77  size_t nVG = vg.size();
78 
79  // Creation of the poll directions in the sub space of the variable group
80  generateUnitPollDirections(directionsSubSpace,nVG);
81 
82  // Convert sub space (in a group of variable) directions to full space directions (all variables)
83  if (varGroups.size() > 1)
84  {
85  size_t vgIndex = 0; // For Output debug only
86  for (std::list<NOMAD::Direction>::iterator it = directionsSubSpace.begin(); it != directionsSubSpace.end() ; ++it)
87  {
88  // In full space, the direction for an index outside the group of variables is null
89  NOMAD::Direction fullSpaceDirection(n,0.0);
90 
91  // Copy the the sub space direction elements to full space
92  size_t i = 0;
93  for (auto index: vg)
94  {
95  fullSpaceDirection[index] = (*it)[i++];
96  }
97  directionsFullSpace.push_back(fullSpaceDirection);
98  OUTPUT_DEBUG_START
99  AddOutputDebug("Unit poll direction for Variable Group " + std::to_string(vgIndex) + ": "+ fullSpaceDirection.display());
100  OUTPUT_DEBUG_END
101  }
102  vgIndex++;
103  }
104  else
105  {
106  directionsFullSpace = directionsSubSpace;
107 
108  OUTPUT_DEBUG_START
109  for (auto dir : directionsFullSpace)
110  {
111  AddOutputDebug("Unit poll direction: " + dir.display());
112  }
113  OUTPUT_DEBUG_END
114 
115  }
116  }
117  }
118 
119  // Scale and project directions on the mesh
120  scaleAndProjectOnMesh(directionsFullSpace);
121 
122  OUTPUT_DEBUG_START
123  for (auto dir : directionsFullSpace)
124  {
125  AddOutputDebug("Scaled and mesh projected poll direction: " + dir.display());
126  }
127  OUTPUT_DEBUG_END
128 
129  OUTPUT_INFO_START
130  AddOutputInfo("Generate points for " + _name, true, false);
131  OUTPUT_INFO_END
132 
133  // We need a frame center to start with.
134  if (!_frameCenter.ArrayOfDouble::isDefined() || _frameCenter.size() != n)
135  {
136  std::string err = "Invalid frame center: " + _frameCenter.display();
137  throw NOMAD::Exception(__FILE__, __LINE__, err);
138  }
139 
140  OUTPUT_DEBUG_START
141  AddOutputDebug("Frame center: " + _frameCenter.display());
142  OUTPUT_DEBUG_END
143 
144  for (std::list<NOMAD::Direction>::iterator it = directionsFullSpace.begin(); it != directionsFullSpace.end() ; ++it)
145  {
146  NOMAD::Point pt(n);
147 
148  // pt = frame center + direction
149  for (size_t i = 0 ; i < n ; ++i )
150  {
151  pt[i] = _frameCenter[i] + (*it)[i];
152  }
153 
154  auto evalPoint = NOMAD::EvalPoint(pt);
155  evalPoint.setPointFrom(std::make_shared<NOMAD::EvalPoint>(_frameCenter), NOMAD::SubproblemManager::getInstance()->getSubFixedVariable(this));
156  // Snap the points and the corresponding direction to the bounds
157  if (snapPointToBoundsAndProjectOnMesh(evalPoint,
158  _pbParams->getAttributeValue<NOMAD::ArrayOfDouble>("LOWER_BOUND"),
159  _pbParams->getAttributeValue<NOMAD::ArrayOfDouble>("UPPER_BOUND")))
160  {
161  if (*evalPoint.getX() != *_frameCenter.getX())
162  {
163  // New EvalPoint to be evaluated.
164  // Add it to the list.
165  evalPoint.setGenStep(getName());
166  bool inserted = insertTrialPoint(evalPoint);
167 
168  OUTPUT_INFO_START
169  std::string s = "Generated point";
170  s += (inserted) ? ": " : " not inserted: ";
171  s += evalPoint.display();
172  AddOutputInfo(s);
173  OUTPUT_INFO_END
174  }
175  }
176  }
177 
178  OUTPUT_INFO_START
179  AddOutputInfo("Generated " + NOMAD::itos(getTrialPointsCount()) + " points");
180  AddOutputInfo("Generate points for " + _name, false, true);
181  OUTPUT_INFO_END
182 }

185 void NOMAD::PollMethodBase::scaleAndProjectOnMesh(std::list<Direction> & dirs)
186 {
187  // Scale the directions and project on the mesh
188 
189  std::shared_ptr<NOMAD::MeshBase> mesh = getIterationMesh();
190  if (nullptr == mesh)
191  {
192  std::string err("Iteration or Mesh not found.");
193  throw NOMAD::Exception(__FILE__, __LINE__, err);
194  }
195 
196  std::list<NOMAD::Direction>::iterator itDir;
197  auto n = _pbParams->getAttributeValue<size_t>("DIMENSION");
198  for (itDir = dirs.begin(); itDir != dirs.end(); ++itDir)
199  {
200  Direction scaledDir(n,0.0);
201 
202  // Compute infinite norm for direction pointed by itDir.
203  NOMAD::Double infiniteNorm = (*itDir).infiniteNorm();
204  if (0 == infiniteNorm)
205  {
206  std::string err("Cannot handle an infinite norm of zero");
207  throw NOMAD::Exception(__FILE__, __LINE__, err);
208  }
209 
210  for (size_t i = 0; i < n; ++i)
211  {
212  // Scaling and projection on the mesh
213  scaledDir[i] = mesh->scaleAndProjectOnMesh(i, (*itDir)[i] / infiniteNorm);
214  }
215 
216  *itDir = scaledDir;
217  }
218 }
