65 /// Class for random number generator
66 /**
67 This class is used to set a seed for the random number generator and get a random integer or a random double between two values. \n
68  http://madrabbit.org/~ray/code/xorshf96.c with period 2^96-1
69  */
70 class RNG {
71 
72 public:
73  typedef uint32_t result_type;
74 
75  static constexpr result_type min() { return 0; }
76  static constexpr result_type max() { return UINT32_MAX; }
77 
78  /// Get current seed
79  /**
80  \return An integer in [0,UINT32_MAX].
81  */
82  static int getSeed()
83  {
84  return _s;
85  }
86 
87  /// Set seed
88  /**
89  * The set seed works like a reset. The private seed used by RNG is always reset.
90  \param s The seed -- \b IN.
91  */
92  static void setSeed(int s);
93 
94  /// Get a random integer
95  /**
96  \return An integer in the interval [0,UINT32_MAX].
97  */
98  static result_type rand();
99 
100  /// Functor to get a random integer
101  /**
102  \return An integer in the interval [0,UINT32_MAX].
103  */
104  result_type operator()() { return rand(); }
105 
106  /// Get a random number having a normal distribution as double
107  /**
108  \param a Lower bound -- \b IN.
109  \param b Upper bound -- \b IN.
110  \return A double in the interval [a,b].
111  */
112  static double rand(double a, double b)
113  {
114  return a+((b-a)*RNG::rand())/UINT32_MAX;
115  }
116 
117  /// Get a random number using a normal distribution centered on 0
118  /**
119  * Get a random number approaching a normal distribution (N(0,Var)) as double
120  *
121  *
122  \param Var Variance of the target normal distribution -- \b IN.
123  \param Nsample Number of samples for averaging -- \b IN.
124  \return A double in the interval [-sqrt(3*Var);+sqrt(3*Var)].
125  */
126  static double normalRandMean0(double Var = 1, int Nsample = 12);
127 
128 
129  /// Get a random number approaching a normal distribution N(Mean,Var) as double.
130  /**
131  A series of Nsample random numbers Xi in the interval [-sqrt(3*Var);+sqrt(3*Var)] is used -> E[Xi] = 0, Var(Xi) = var. \n
132  See http://en.wikipedia.org/wiki/Central_limit_theorem
133 
134  \param Mean Mean of the target normal distribution -- \b IN.
135  \param Var Variance of the target normal distribution -- \b IN.
136  \return A random number.
137  */
138  static double normalRand(double Mean = 0, double Var = 1);
139 
140  /// Reset seed to its default value
141  static void resetPrivateSeedToDefault()
142  {
143  _x = x_def;
144  _y = y_def;
145  _z = z_def;
146  }
147 
148  /// Get private values
149  static void getPrivateSeed(uint32_t &x, uint32_t &y, uint32_t &z)
150  {
151  x = _x;
152  y = _y;
153  z = _z;
154  }
155 
156  /// Reset seed to given values.
157  /**
158  Used to set back RNG in a known state.
159  */
160  static void setPrivateSeed(uint32_t x, uint32_t y, uint32_t z)
161  {
162  _x = x;
163  _y = y;
164  _z = z;
165  }
166 
167 private:
168 
169  static uint32_t x_def, y_def, z_def; ///< Initial values for the random number generator
170  static uint32_t _x, _y, _z; ///< Current values for the random number generator
171 
172  static int _s;
173 
174 
175 };
