79 /*------------------------------------------------*/
80 /* Norm of this direction X, viewed as a vector. */
81 /* Norm type may be L1, L2, or LINF. */
82 /*------------------------------------------------*/
83 const NOMAD::Double NOMAD::Direction::norm(NOMAD::NormType normType) const
84 {
85  NOMAD::Double retNorm = 0;
86 
87  switch (normType)
88  {
89  case NOMAD::NormType::L1:
90  for (size_t i = 0; i < size(); i++)
91  {
92  retNorm += _array[i].abs();
93  }
94  break;
95  case NOMAD::NormType::L2:
96  default:
97  retNorm = this->squaredL2Norm();
98  retNorm = sqrt(retNorm.todouble());
99  break;
100  case NOMAD::NormType::LINF:
101  for (size_t i = 0; i < size(); i++)
102  {
103  retNorm = NOMAD::max(retNorm, _array[i].abs());
104  }
105  break;
106  }
107 
108  return retNorm;
109 }

142 const NOMAD::Double NOMAD::Direction::cos(const NOMAD::Direction& dir1,
143  const NOMAD::Direction& dir2)
144 {
145  NOMAD::Double cos = 0.0;
146 
147  double norm1 = dir1.norm().todouble();
148  double norm2 = dir2.norm().todouble();
149  if (0.0 == norm1 || 0.0 == norm2)
150  {
151  std::string err = "Cosine: a vector is of size 0";
152  throw NOMAD::Exception(__FILE__, __LINE__, err);
153  }
154 
155  cos = dotProduct(dir1, dir2) / (norm1 * norm2);
156 
157  return cos;
158 }

160 /*-----------------------------------------------------------*/
161 /* computation of the angle with another direction */
162 /*-----------------------------------------------------------*/
163 const NOMAD::Double NOMAD::Direction::angle(const NOMAD::Direction& dir1,
164  const NOMAD::Direction& dir2)
165 {
166  if (dir1.size() != dir2.size())
167  {
168  return NOMAD::Double();
169  }
170 
171  NOMAD::Double innerProduct = 0.0, norm1 = 0.0, norm2 = 0.0;
172 
173  for (size_t i = 0; i < dir1.size(); i++)
174  {
175  norm1 += dir1[i] * dir1[i];
176  norm2 += dir2[i] * dir2[i];
177  innerProduct += dir1[i] * dir2[i];
178  }
179 
180  if (norm1 == 0.0 || norm2 == 0.0)
181  {
182  return NOMAD::Double();
183  }
184 
185  return std::acos((innerProduct / (norm1.sqrt() * norm2.sqrt())).todouble());
186 }

192 /*--------------------------------------------------*/
193 void NOMAD::Direction::computeDirOnUnitSphere(NOMAD::Direction &randomDir)
194 {
195  size_t i;
196  NOMAD::Double norm, d;
197  size_t n = randomDir.size();
198 
199  for (i = 0; i < n; ++i)
200  {
201  randomDir[i] = NOMAD::RNG::normalRand(0,1);
202  }
203 
204  norm = randomDir.norm();
205 
206  if (0 == norm)
207  {
208  std::string err("Cannot compute a random direction on unit sphere");
209  throw NOMAD::Exception(__FILE__, __LINE__, err);
210  }
211 
212  for (i = 0; i < n; ++i)
213  {
214  randomDir[i] /= norm;
215  }
216 
217 }

220 /*----------------------------------------------------------------*/
221 /* - Householder transformation to generate _nc directions from */
222 /* a given direction. */
223 /* - Compute also H[i+nc] = -H[i] (completion to 2n directions). */
224 /* - Private method. */
225 /*----------------------------------------------------------------*/
226 void NOMAD::Direction::householder(const NOMAD::Direction &dir,
227  bool completeTo2n,
228  NOMAD::Direction ** H)
229 {
230  size_t n = dir.size();
231 
232  const NOMAD::Double norm2 = dir.squaredL2Norm();
233  NOMAD::Double v, h2i;
234 
235  for (size_t i = 0 ; i < n ; ++i)
236  {
237  h2i = 2 * dir[i];
238  for (size_t j = 0 ; j < n ; ++j)
239  {
240  // H[i]:
241  (*H[i])[j] = v = (i == j) ? norm2 - h2i * dir[j] : - h2i * dir[j];
242 
243  // -H[i]:
244  if ( completeTo2n )
245  {
246  (*H[i+n])[j] = -v;
247  }
248  }
249  }
250 }
