54 /// Class combining all stop reasons that are not algorithmic stop reasons.
55 /**
56 
57  Several stop reasons are members of this class. The stop reasons are templated on stop type. Several stop types are available in this class:
58  - a ::BaseStopType for high level stop reasons.
59  - an ::EvalGlobalStopType for global evaluation stop reasons.
60  - an ::IterStopType for stop reasons during iteration of an algorithm (for example, maximum iteration number reached).
61 
62  The static stop reasons ::BaseStopType and ::EvalGlobalStopType are shared.
63  */
64 class AllStopReasons
65 {

67  /// Constructor

70  }

79  StopReason<IterStopType> _iterStopReason; ///< An iteration stop reason.
