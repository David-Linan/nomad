60 /// Class to define an optimization subproblem
61 /**
62 * Subproblem of lesser dimension than the original problem
63 *
64 * - Sets up the new parameters
65 * - Keeps the necessary information to bridge the gap between subproblem and
66  original problem
67 */
68 class Subproblem
69 {
70 private:
71  /**
72  * The elements of this point that have defined values are fixed value
73  * "variables". The elements that are undefined are for true variables.
74  * This Point is always in full dimension.
75  */
76  const Point _fixedVariable;
77  size_t _dimension; ///< Dimension of the subproblem.
78 
79  /**
80  Reference to the original problem's PbParameters.
81  */
82  const std::shared_ptr<PbParameters> _refPbParams;
83 
84  /**
85  PbParameters converted to subdimension
86  */
87  std::shared_ptr<PbParameters> _subPbParams;
88 
89 public:
90  /// Constructor
91  /**
92  Pb parameters will be recomputed as dimension has changed.
93  */
94  explicit Subproblem(const std::shared_ptr<PbParameters> refPbParams,
95  const Point& fullFixedVariable)
96  : _fixedVariable(fullFixedVariable),
97  _dimension(refPbParams->getAttributeValue<size_t>("DIMENSION")),
98  _refPbParams(refPbParams),
99  _subPbParams(nullptr)
100  {
101  init();
102  }
103 
104  /// Destructor
105  virtual ~Subproblem();
106 
107  // Get/Set
108 
109  const Point& getFixedVariable() const { return _fixedVariable; }
110  const std::shared_ptr<PbParameters>& getPbParams() const { return _subPbParams; }
111 
112 private:
113  /// Helper for constructor calls to Subproblem::setupProblemParameters
114  void init();
115 
116  /// Helper for constructor
117  /**
118  Construct the subproblem parameters (X0, LB, UB, mesh sizes, variable groups...) based on Subproblem::_fixedVariable
119  \note If a new parameter with dimension (ex. a parameter of type ArrayOfDouble, Point, or Dimension)
120  is added to the class PbParameters, this method will break.
121  Currently supported parameters: X0 LOWER_BOUND UPPER_BOUND BB_INPUT_TYPE INITIAL_MESH_SIZE INITIAL_FRAME_SIZE MIN_MESH_SIZE MIN_FRAME_SIZE GRANULARITY VARIABLE_GROUP
122  */
123  void setupProblemParameters();
124 
125 
126  /// Helper for setupProblemParameters()
127  void resetVariableGroupsAgainstFixedVariables(ListOfVariableGroup& lvg, const Point& fixedVar) const;
128 };
