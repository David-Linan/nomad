187 /// Template class for the stop reason of a stop type.
188 /**
189  The possible stop types can be generic: ::IterStopType,
190  ::EvalGlobalStopType, ::EvalMainThreadStopType, BaseStopType, or specific to
191  an algorithm: ::MadsStopType, ::PhaseOneStopType, ::NMStopType ,....
192 
193  The default stop type is STARTED (no stop). A stop reason different than STARTED indicates what is the cause of termination. Some stop reasons indicate a normal termination that do not need propagation and others must be propagated to stop an algorithm (see StopReason::checkTerminate()).
194 
195  */
196 template<typename T >
197 class StopReason
198 {
199 
200 private:
201 
202  T _stopReason; ///< The stop reason stored as a stop type
203 
204  /// Dictionnary to translate a stop type into a string.
205  /**
206  We have template specializations of this function for each stop type.
207  This function is called to display the stop reason.
208  */
209  std::map<T,std::string> & dict() const;
210 
211  /// Helper for constructor (check sanity)
212  void testValidity() const
213  {
214  if ( ! std::is_enum<T>::value )
215  throw Exception(__FILE__,__LINE__,"The templated stop reason is not an enum.");
216 
217  if ( dict().size() == 0 )
218  throw Exception(__FILE__,__LINE__,"Dictionary not filled.");
219 
220 
221  if ( (int) T::STARTED != 0 )
222  throw Exception(__FILE__,__LINE__,"First StopType in enum must be STARTED.");
223 
224  if ( (int) T::LAST != dict().size() )
225  {
226  std::string s = "Not enough elements in enum dictionary (";
227  s += std::to_string(dict().size()) + "), expecting " + std::to_string((int)T::LAST);
228  throw Exception(__FILE__,__LINE__,s);
229  }
230 
231  for (int i = (int) T::STARTED; i < (int ) T::LAST; i++)
232  {
233  typename std::map<T,std::string>::iterator it = dict().find( (T) i );
234 
235  if ( it== dict().end() )
236  throw Exception(__FILE__,__LINE__,"All enum elements must be in dictionary.");
237 
238  }
239 
240  }
241 
242 public:
243  /// Constructor
244  /**
245  Upon construction, the validity of the stop reason is verified (sanity check). \n
246  By default the stop reason is set to STARTED.
247  */
248  explicit StopReason ()
249  {
250  testValidity();
251 
252  _stopReason = T::STARTED;
253 
254  }
255 
256  /// Destructor
257  virtual ~StopReason() {}
258 
259  /// The stop reason
260  T get() const
261  {
262  return _stopReason;
263  }
264 
265  /// Set the stop reason if it is listed in dictionnary
266  void set (T s)
267  {
268  typename std::map<T,std::string>::iterator it = dict().find(s);
269 
270  if ( it==dict().end() )
271  throw Exception(__FILE__,__LINE__,"Stop reason not found.");
272 
273  _stopReason = s;
274  }
275 
276  /// Reset the stop reason to the default STARTED state.
277  void setStarted ()
278  {
279  _stopReason = T::STARTED;
280  }
281 
282  /// Check if it is in STARTED state.
283  bool isStarted () const
284  {
285  return ( _stopReason == T::STARTED );
286  }
287 
288  /// Translate the stop reason into a string for display
289  std::string getStopReasonAsString() const
290  {
291  typename std::map<T,std::string>::iterator it = dict().find( _stopReason );
292  return it->second;
293  }
294 
295  /// Check if the stop reason requires a termination.
296  /**
297  This is implemented for each stop type (template specialization of the function). \n
298  Except for ::EvalMainThreadStopType, a stop reason different than STARTED indicates that an algorithm or a sub-algorithm must terminate.
299 
300  \return \c true if a termination is required, \c false otherwise.
301  */
302  bool checkTerminate () const ;
303 
304 };
