53 /// Class for barrier following algorithm 12.2 of DFBO.
54 class Barrier
55 {
56 private:
57 
58  std::vector<EvalPoint> _xFeas; ///< Current feasible incumbent solutions
59  std::vector<EvalPoint> _xInf; ///< Current infeasible incumbent solutions
60 
61  EvalPointPtr _refBestFeas; ///< Previous first feasible incumbent
62  EvalPointPtr _refBestInf; ///< Previous first infeasible incumbent
63 
64  Double _hMax; ///< Maximum acceptable value for h
65 
66  /// Dimension of the points in the barrier.
67  /**
68  * Used for verification only.
69  * To be reviewed when we address category variables.
70  /see _n in CacheBase.
71  */
72  size_t _n;
73 
74 public:
75  /// Constructor
76  /**
77  * hMax will be updated during optimization.
78  \param hMax The max of h to keep a point in the barrier -- \b IN.
79  \param fixedVariable The fixed variables have a fixed value -- \b IN.
80  \param evalType Type of evaluation (BB or MODEL) -- \b IN.
81  \param evalPointList Additional points to consider in building the barrier -- \b IN.
82  */
83  Barrier(const Double& hMax = INF,
84  const Point& fixedVariable = Point(),
85  const EvalType& evalType = EvalType::BB,
86  const ComputeType& computeType = ComputeType::STANDARD,
87  const std::vector<EvalPoint>& evalPointList = std::vector<EvalPoint>())
88  : _xFeas(),
89  _xInf(),
90  _refBestFeas(nullptr),
91  _refBestInf(nullptr),
92  _hMax(hMax),
93  _n(0)
94  {
95  init(fixedVariable, evalType, evalPointList, computeType);
96  }
97 
98  /*-----------------*/
99  /* Feasible points */
100  /*-----------------*/
101  /// Get all feasible points in the barrier
102  /**
103  \return All the eval points that are feasible.
104  */
105  const std::vector<EvalPoint>& getAllXFeas() const { return _xFeas; }
106 
107  /// Update ref best feasible and ref best infeasible values.
108  void updateRefBests();
109 
110  /// Get the first feasible point in the barrier.
111  /**
112  * If there is no feasible point, return a \c nullptr
113  \return A single feasible eval point.
114  */
115  EvalPointPtr getFirstXFeas() const;
116 
117  /// Get the point that was previously the first feasible point in the barrier.
118  /**
119  * If there is no feasible point, return a \c nullptr
120  \return A single feasible eval point.
121  */
122  EvalPointPtr getRefBestFeas() const { return _refBestFeas; }
123  void setRefBestFeas(const EvalPointPtr refBestFeas) { _refBestFeas = refBestFeas; }
124 
125  /// Number of feasible points in the barrier.
126  size_t nbXFeas() const { return _xFeas.size(); }
127 
128  /// Add a feasible point in the barrier.
129  /**
130  * If the point is feasible it is added, if not an exception is triggered.
131  \param xFeas The eval point to add -- \b IN.
132  \param evalType Which eval (Blackbox or Model) of the EvalPoint to use to verify feasibility -- \b IN.
133  */
134  void addXFeas(const EvalPoint &xFeas, const EvalType& evalType, const ComputeType& computeType = ComputeType::STANDARD);
135 
136  /// Remove feasible points from the barrier.
137  void clearXFeas();
138 
139  /*-------------------*/
140  /* Infeasible points */
141  /*-------------------*/
142  /// Get all infeasible points in the barrier
143  /**
144  \return All the eval points that are infeasible.
145  */
146  const std::vector<EvalPoint>& getAllXInf() const { return _xInf; }
147 
148  /// Get the first infeasible point in the barrier.
149  /**
150  * If there is no infeasible point, return a \c nullptr
151  \return A single infeasible eval point.
152  */
153  EvalPointPtr getFirstXInf() const;
154 
155  /// Get the point that was previously the first infeasible point in the barrier.
156  /**
157  * If there is no feasible point, return a \c nullptr
158  \return A single feasible eval point.
159  */
160  EvalPointPtr getRefBestInf() const { return _refBestInf; }
161  void setRefBestInf(const EvalPointPtr refBestInf) { _refBestInf = refBestInf; }
162 
163  /// Number of infeasible points in the barrier.
164  size_t nbXInf() const { return _xInf.size(); }
165 
166  /// Add an infeasible point in the barrier.
167  /**
168  * If the point is nullptr an exception is triggered.
169  \param xInf The eval point to add -- \b IN.
170  */
171  void addXInf(const EvalPoint &xInf, const EvalType& evalType);
172 
173  /// Remove infeasible points from the barrier.
174  void clearXInf();
175 
176  /*---------------*/
177  /* Other methods */
178  /*---------------*/
179  /// Get all feasible and infeasable points
180  std::vector<EvalPoint> getAllPoints() const;
181 
182  /// Get first of all feasible and infeasible points.
183  /** If there are feasible points, returns first feasible point.
184  * else, returns first infeasible point. */
185  const EvalPoint& getFirstPoint() const;
186 
187  /// Get the current hMax of the barrier.
188  Double getHMax() const { return _hMax; }
189 
190  /// Set the hMax of the barrier
191  /**
192  \param hMax The hMax -- \b IN.
193  */
194  void setHMax(const Double &hMax);
195 
196  /// Update xFeas and xInf according to given points.
197  /* \param evalPointList vector of EvalPoints -- \b IN.
198  * \param keepAllPoints keep all good points, or keep just one point as in NOMAD 3 -- \b IN.
199  * \return true if the Barrier was updated, false otherwise
200  * \note Input EvalPoints are already in subproblem dimention
201  */
202  bool updateWithPoints(const std::vector<EvalPoint>& evalPointList,
203  const EvalType& evalType,
204  const ComputeType& computeType,
205  const bool keepAllPoints = false);
206 
207  /// Return the barrier as a string.
208  /* May be used for information, or for saving a barrier. In the former case,
209  * it may be useful to set parameter max to a small value (e.g., 4). In the
210  * latter case, INF_SIZE_T should be used so that all points are saved.
211  * \param max Maximum number of feasible and infeasible points to display
212  * \return A string describing the barrier
213  */
214  std::string display(const size_t max = INF_SIZE_T) const;
215 
216 private:
217 
218  /**
219  * \brief Helper function for constructor.
220  *
221  * Will throw exceptions or output error messages if something is wrong. Will remain silent otherwise.
222  \param fixedVariable The fixed variables have a fixed value -- \b IN.
223  \param evalType Which eval (Blackbox or Model) to use to verify feasibility -- \b IN.
224  \param evalPointList Additional points to consider to construct barrier. -- \b IN.
225  */
226  void init(const Point& fixedVariable,
227  const EvalType& evalType,
228  const std::vector<EvalPoint>& evalPointList,

231  /**

257  void checkXFeasIsFeas(const EvalPoint &xFeas,

265  */
266  void checkXInf(const EvalPoint &xInf, const EvalType& evalType);
267 
268  /**
269  * \brief Helper function for init/setHMax.

273  void checkHMax();
