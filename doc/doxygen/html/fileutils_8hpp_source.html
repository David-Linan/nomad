143 // Write to file
144 // Use T::operator<< (examples of T: CacheSet, Mads)
145 // Will break if T::operator<< is not defined.
146 template<typename T>
147 bool write(const T &info, const std::string &filename)
148 {
149  bool writeSuccess = true;
150  std::ofstream fout;
151 
152  if (filename.empty())
153  {
154  std::cerr << "Warning: " << typeid(T).name() << ": Cannot write to file: file name is not defined.";
155  writeSuccess = false;
156  }
157 
158  if (writeSuccess)
159  {
160  fout.open(filename.c_str(), std::ofstream::out);
161  if (fout.fail())
162  {
163  std::cerr << "Warning: " << typeid(T).name() << ": Cannot write to file " + filename << std::endl;
164  writeSuccess = false;
165  fout.close();
166  }
167  }
168 
169  if (writeSuccess)
170  {
171  fout.clear();
172  fout << info;
173  fout.close();
174  }
175 
176 
177  return writeSuccess;
178 }

181 // Read from file
182 // Use T::operator>> (examples of T: CacheSet, Mads)
183 // Will break if T::operator>> is not defined.
184 template<typename T>
185 bool read(T &info, const std::string &filename)
186 {
187  bool readSuccess = true;
188  std::ifstream fin;
189 
190  if (filename.empty())
191  {
192  std::cerr << "Warning: " << typeid(T).name() << ": Cannot read file: file name is not defined.";
193  readSuccess = false;
194  }
195 
196  if (readSuccess)
197  {
198  if (!checkReadFile(filename))
199  {
200  std::cerr << "Warning: " << typeid(T).name() << ": File does not exist or cannot be read: " + filename << std::endl;
201  readSuccess = false;
202  }
203  }
204 
205  if (readSuccess)
206  {
207  fin.open(filename.c_str(), std::ifstream::out);
208  if (fin.fail())
209  {
210  std::cerr << "Warning: " << typeid(T).name() << ": Cannot read from file " + filename << std::endl;
211  readSuccess = false;
212  fin.close();
213  }
214  }
215 
216  if (readSuccess)
217  {
218  fin >> info;
219  }
220 
221  fin.close();
222 
223  return readSuccess;
224 }
