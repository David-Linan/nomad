72 std::vector<bool> NOMAD::QuadModelEvaluator::eval_block(NOMAD::Block &block,
73  const NOMAD::Double &hMax,
74  std::vector<bool> &countEval) const
75 {
76  std::vector<bool> evalOk;
77  countEval.clear();
78 
79  // Verify there is at least one point to evaluate
80  if (0 == block.size())
81  {
82  throw NOMAD::Exception(__FILE__, __LINE__, "Evaluator: eval_block called with an empty block");
83  }
84 
85  // points were sent to the evaluator in full space.
86  // Convert points to subspace, because model is in subspace.
87  for (size_t i = 0; i < block.size(); i++)
88  {
89  block[i] = std::make_shared<NOMAD::EvalPoint>(block[i]->makeSubSpacePointFromFixed(_fixedVariable));
90  }
91 
92  size_t m = block.size();
93  size_t n = block[0]->size();
94 
95  const auto bbot = _evalParams->getAttributeValue<NOMAD::BBOutputTypeList>("BB_OUTPUT_TYPE");
96 
97  size_t nbConstraints = NOMAD::getNbConstraints(bbot);
98  size_t nbModels = nbConstraints+1;
99 
100  // Init the matrices for prediction
101  // Creation of matrix for input / output of SGTELIB model
102  SGTELIB::Matrix M_predict ( "M_predict", static_cast<int>(m), static_cast<int>(nbModels));
103  SGTELIB::Matrix X_predict("X_predict", static_cast<int>(m), static_cast<int>(n));
104 
105  int j = 0;
106  for (auto it = block.begin(); it != block.end(); it++, j++)
107  {
108  if (!(*it)->isComplete())
109  {
110  throw NOMAD::Exception(__FILE__, __LINE__, "Evaluator: Incomplete point " + (*it)->display());
111  }
112 
113  std::string s = "X" + itos(j) +" =" + (*it)->display();
114  NOMAD::OutputQueue::Add(s, _displayLevel);
115 
116  // Set the input matrix
117  for (size_t i = 0; i < (*it)->size(); i++)
118  {
119  X_predict.set(j, static_cast<int>(i), (*(*it))[i].todouble());
120  }
121 
122  // Reset point outputs
123  // By default, set everything to -1
124  // Note: Currently NOMAD cannot set a bbo value by index, so we have to
125  // work around by constructing a suitable string.
126  // Note: Why set some default values on bbo?
127  NOMAD::ArrayOfString defbbo(bbot.size(), "-1");
128  (*it)->setBBO(defbbo.display(), bbot, NOMAD::EvalType::MODEL);
129 
130  }
131 
132  // ------------------------- //
133  // Output Prediction //
134  // ------------------------- //
135  NOMAD::OutputQueue::Add("Predict with quadratic formulation... ", _displayLevel);
136 
137 
138  // Unfortunately, Sgtelib is not thread-safe.
139  // For this reason we have to set part of the eval_x code to critical.
140 #ifdef _OPENMP
141 #pragma omp critical(SgtelibEvalX)
142 #endif // _OPENMP
143  {
144  _model->check_ready(__FILE__,__FUNCTION__,__LINE__);
145 
146  _model->predict(X_predict, &M_predict);
147  NOMAD::OutputQueue::Add("ok", _displayLevel);
148  }
149 
150  j = 0;
151  // Verify all points are completely defined
152  for (auto it = block.begin(); it != block.end(); it++, j++)
153  {
154  std::string s = "X" + itos(j) +": " + (*it)->display();
155  NOMAD::OutputQueue::Add(s, _displayLevel);
156  // ====================================== //
157  // Output display //
158  // ====================================== //
159  std::string sObj = "F = ";
160  std::string sCons = "C = [ ";
161  for (size_t i = 0; i < nbModels; i++)
162  {
163  if (bbot[i] != NOMAD::BBOutputType::OBJ)
164  sCons += std::to_string(M_predict.get(j,static_cast<int>(i))) + " ";
165  else
166  sObj += std::to_string(M_predict.get(j,static_cast<int>(i))) + " ";
167  }
168  s = sObj + ((nbConstraints>0 ) ? sCons+" ]":"") ;
169  NOMAD::OutputQueue::Add(s, _displayLevel);
170 
171  // ====================================== //
172  // Application of the formulation //
173  // ====================================== //
174  NOMAD::ArrayOfDouble newbbo(bbot.size(), -1);
175 
176  // ----------------- //
177  // Set BBO //
178  // ----------------- //
179  for (size_t i = 0; i < nbModels; i++)
180  {
181  newbbo[i] = M_predict.get(j,static_cast<int>(i));
182  }
183  NOMAD::ArrayOfDouble fullPrecision(bbot.size(), NOMAD::DISPLAY_PRECISION_FULL);
184  (*it)->setBBO(newbbo.display(fullPrecision), bbot, NOMAD::EvalType::MODEL);
185 
186  // ================== //
187  // Exit Status //
188  // ================== //
189  countEval.push_back( true );
190  evalOk.push_back(true);
191  (*it)->setEvalStatus(NOMAD::EvalStatusType::EVAL_OK, NOMAD::EvalType::MODEL);
192 
193  }
194 
195  // Convert points back to full space.
196  for (size_t i = 0; i < block.size(); i++)
197  {
198  block[i] = std::make_shared<NOMAD::EvalPoint>(block[i]->makeFullSpacePointFromFixed(_fixedVariable));
199  }
200 
201  return evalOk;
202 }
