89 bool NOMAD::NMIteration::runImp()
90 {
91  // Sequential run of NM steps among INITIAL, ( REFLECT, EXPANSION, INSIDE_CONTRACTION, OUTSIDE_CONTRACTION ), SHRINK
92 
93  // NMIteration cannot generate all points before evaluation
94  verifyGenerateAllPointsBeforeEval(__PRETTY_FUNCTION__, false);
95 
96  bool iterationSuccess = false;
97 
98  // Use a single NMReflect object to perform REFLECT, EXPAND, INSIDE_CONTRACTION, OUTSIDE_CONTRACTION of this iteration
99  NOMAD::NMReflective reflect( this );
100 
101  // Start with the REFLECT step
102  NMStepType stepType = NMStepType::REFLECT;
103 
104  bool nmOpt = _runParams->getAttributeValue<bool>("NM_OPTIMIZATION");
105  bool nmSearchStopOnSuccess = _runParams->getAttributeValue<bool>("NM_SEARCH_STOP_ON_SUCCESS");
106 
107  // Running an NM iteration consists in performing
108  // 1) A Reflect
109  // 2) An Expansion or an Inside contraction or an Outside contraction or Continue to next iteration (no shrink).
110  // 3) Possibly a Shrink.
111  while ( ! _stopReasons->checkTerminate() && stepType != NMStepType::CONTINUE && stepType != NMStepType::SHRINK )
112  {
113  // Need to set the current step type before starting
114  reflect.setCurrentNMStepType( stepType );
115 
116  // Create trial points and evaluate them
117  reflect.start();
118  reflect.run();
119  reflect.end();
120 
121  // The NM step type for the next pass
122  stepType = reflect.getNextNMStepType() ;
123 
124  // Update the type of success for passing to the MegaIteration
125  NOMAD::SuccessType success = reflect.getSuccessType();
126 
127  if ( success > _bestSuccess )
128  {
129  // NM Search can be stopped on success
130  if ( success == NOMAD::SuccessType::FULL_SUCCESS && !nmOpt && nmSearchStopOnSuccess )
131  {
132  auto nmStopReason = NOMAD::AlgoStopReasons<NOMAD::NMStopType>::get ( _stopReasons );
133  nmStopReason->set( NOMAD::NMStopType::NM_STOP_ON_SUCCESS );
134  }
135  iterationSuccess = true; // At least a partial success is obtained
136  _bestSuccess = success;
137  }
138  }
139 
140  // Perform SHRINK only for a standalone NM optimization
141  if ( ! _stopReasons->checkTerminate() &&
142  stepType == NMStepType::SHRINK &&
143  nmOpt )
144  {
145  // Create shrink trial points and evaluate them
146  NMShrink shrink ( this );
147  shrink.start();
148  shrink.run();
149  shrink.end();
150 
151  // Update the type of success for passing to the MegaIteration
152  NOMAD::SuccessType success = shrink.getSuccessType();
153  if ( success > _bestSuccess )
154  {
155  iterationSuccess = true; // At least a partial success is obtained
156  _bestSuccess = success;
157  }
158  }
159  // Perform SHRINK only for a standalone NM optimization ELSE stop NM
160  if ( ! _stopReasons->checkTerminate() &&
161  stepType == NMStepType::SHRINK && !nmOpt )
162  {
163  auto nmStopReason = NOMAD::AlgoStopReasons<NOMAD::NMStopType>::get ( _stopReasons );
164  nmStopReason->set( NOMAD::NMStopType::NM_STOP_NO_SHRINK );
165 
166  }
167  if ( iterationSuccess )
168  {
169  // Update MegaIteration success type with best success found.
170  getParentOfType<NOMAD::MegaIteration*>()->setSuccessType(_bestSuccess);
171  }
172 
173  // End of the iteration: iterationSuccess is true if we have a success.
174  return iterationSuccess;
175 
176 }
