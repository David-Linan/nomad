58 int main (int argc, char ** argv)
59 {
60  auto TheMainStep = std::make_unique<NOMAD::MainStep>();
61  NOMAD::OutputQueue::getInstance()->setMaxStepLevel(14);
62  std::string error;
63 
64  // Need at least parameters file.
65  if (argc < 2)
66  {
67  NOMAD::OutputQueue::getInstance()->setDisplayDegree(1);
68  TheMainStep->displayUsage(argv[0]);
69  }
70 
71  else
72  {
73 
74  if (argv[1][0] == '-')
75  {
76  // Options
77  std::string option = argv[1];
78  NOMAD::toupper(option);
79 
80  // Display usage if option '-u' has been specified
81  if (option == "-U" || option == "-USAGE" || option == "--USAGE")
82  {
83  NOMAD::OutputQueue::getInstance()->setDisplayDegree(1);
84  TheMainStep->displayUsage(argv[0]);
85  }
86 
87  // Display info and usage if option '-i' has been specified
88  else if (option == "-I" || option == "-INFO" || option == "--INFO")
89  {
90  NOMAD::OutputQueue::getInstance()->setDisplayDegree(1);
91  TheMainStep->displayInfo();
92  TheMainStep->displayUsage(argv[0]);
93  }
94 
95  // Display version if option '-v' has been specified
96  else if (option == "-V" || option == "-VERSION" || option == "--VERSION")
97  {
98  NOMAD::OutputQueue::getInstance()->setDisplayDegree(1);
99  TheMainStep->displayVersion();
100  }
101  // Display help if option '-h' has been specified
102  else if (option == "-H" || option == "-HELP" || option == "--HELP")
103  {
104  std::string helpSubject = "";
105  if (3 == argc)
106  {
107  helpSubject = argv[2];
108  NOMAD::toupper( helpSubject );
109  if ( helpSubject == "ALL" )
110  helpSubject = "";
111  }
112  TheMainStep->displayHelp ( helpSubject );
113  }
114  // Display developer help if option '-d' has been specified
115  else if (option == "-D" || option == "-DEVHELP" || option == "--DEVHELP")
116  {
117  std::string helpSubject ="";
118  if (3 == argc)
119  {
120  helpSubject = argv[2];
121  NOMAD::toupper( helpSubject );
122  if ( helpSubject == "ALL" )
123  helpSubject = "";
124  }
125  TheMainStep->displayHelp(helpSubject, true);
126  }
127  else
128  {
129  NOMAD::OutputQueue::getInstance()->setDisplayDegree(1);
130  TheMainStep->AddOutputInfo("ERROR: Unrecognized option: " +option, NOMAD::OutputLevel::LEVEL_ERROR);
131  TheMainStep->displayUsage(argv[0]);
132  }
133  }
134 
135  else
136  {
137  try
138  {
139  // Use first argument as parameters file.
140  std::string paramfilename = argv[1];
141 
142  if (!NOMAD::checkReadFile(paramfilename))
143  {
144  NOMAD::OutputQueue::getInstance()->setDisplayDegree(1);
145  error = std::string("ERROR: Could not read file \"") + argv[1] + "\"";
146  std::cerr << std::endl << error << std::endl << std::endl;
147  TheMainStep->displayUsage(argv[0]);
148  }
149  else
150  {
151  TheMainStep->setParamFileName(paramfilename);
152 
153  // Reads parameters
154  TheMainStep->start();
155 
156  // Creates the EvaluatorControl, Mads, and runs Mads.
157  TheMainStep->run();
158 
159  TheMainStep->end();
160  }
161  }
162  catch (NOMAD::Exception &e)
163  {
164  error = "ERROR: ";
165  error += e.what();
166  std::cerr << std::endl << error << std::endl << std::endl;
167  NOMAD::OutputQueue::getInstance()->setDisplayDegree(0);
168  }
169  }
170  }
171 
172  NOMAD::OutputQueue::Flush();
173 
174  return (error.empty()) ? EXIT_SUCCESS : EXIT_FAILURE;
175 }
