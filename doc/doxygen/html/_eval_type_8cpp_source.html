59 // Convert a string ("BB", "MODEL")
60 // to a NOMAD::EvalType.
61 // "UNDEFINED" throws an exception, as well as any value other than "BB", "MODEL".
62 NOMAD::EvalType NOMAD::stringToEvalType(const std::string &sConst)
63 {
64  NOMAD::EvalType ret;
65  std::string s = sConst;
66  NOMAD::toupper(s);
67 
68  if (s == "BB")
69  {
70  ret = NOMAD::EvalType::BB;
71  }
72  else if (s == "MODEL")
73  {
74  ret = NOMAD::EvalType::MODEL;
75  }
76  else
77  {
78  throw NOMAD::Exception(__FILE__, __LINE__, "Unrecognized string for NOMAD::EvalType: " + s);
79  }
80 
81  return ret;
82 }

85 // Convert a NOMAD::EvalType to a string.
86 // NOMAD::EvalType::UNDEFINED returns "UNDEFINED".
87 // An unrecognized eval type returns an exception.
88 std::string NOMAD::evalTypeToString(const NOMAD::EvalType& evalType)
89 {
90  std::string s;
91 
92  switch(evalType)
93  {
94  case NOMAD::EvalType::BB:
95  s = "BB";
96  break;
97  case NOMAD::EvalType::MODEL:
98  s = "MODEL";
99  break;
100  case NOMAD::EvalType::UNDEFINED:
101  s = "UNDEFINED";
102  break;
103  default:
104  throw NOMAD::Exception(__FILE__, __LINE__, "Unrecognized NOMAD::EvalType " + std::to_string((int)evalType));
105  break;
106  }
107 
108  return s;
109 }
