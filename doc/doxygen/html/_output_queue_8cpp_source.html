104  }

282 void NOMAD::OutputQueue::flush()
283 {
284  if (_queue.empty())
285  {
286  return;
287  }
288 
289  if (_maxOutputLevel >= NOMAD::OutputLevel::LEVEL_DEBUGDEBUG)
290  {
291  // hyper-debug
292  std::cout << "Output all " << _queue.size() << " elements." << std::endl;
293  }
294 
295 #ifdef _OPENMP
296  // Lock queue before flush
297  omp_set_lock(&_s_queue_lock);
298 #endif // _OPENMP
299 
300  // Info goes to Standard output
301  for (std::vector<NOMAD::OutputInfo>::iterator out_it = _queue.begin();
302  out_it != _queue.end(); ++out_it)
303  {
304  flushBlock(std::move(*out_it));
305  }
306  _queue.clear();
307 #ifdef _OPENMP
308  omp_unset_lock(&_s_queue_lock);
309 #endif // _OPENMP

336 void NOMAD::OutputQueue::flushBlock(const NOMAD::OutputInfo &outputInfo)
337 {
338  // Output one line for each string in the vector msg.
339 
340  NOMAD::OutputLevel outputLevel = outputInfo.getOutputLevel();
341  const NOMAD::StatsInfo* statsInfo = outputInfo.getStatsInfo();
342 
343  flushStatsToStatsFile(statsInfo);
344 
345  if (outputLevel > _maxOutputLevel)
346  {
347  // Do not display to stdout
348  return;
349  }
350 
351  NOMAD::ArrayOfString msg = outputInfo.getMsg();
352  if (outputLevel == NOMAD::OutputLevel::LEVEL_STATS)
353  {
354  flushStatsToStdout(statsInfo);
355  }
356 
357  else
358  {
359  if (outputInfo.isBlockEnd())
360  {
361  _indentLevel--;
362  if (_indentLevel < 0)
363  {
364  #ifdef _OPENMP
365  omp_unset_lock(&_s_queue_lock);
366  #endif // _OPENMP
367  throw NOMAD::Exception(__FILE__, __LINE__, "OutputQueue has more block ends than block starts.");
368  }
369  }
370 
371  // Verify step level is high enough in the tree to be displayed.
372  if (_indentLevel <= (int)_maxStepLevel)
373  {
374  for (size_t i = 0; i < msg.size(); i++)
375  {
376  indent(_indentLevel);
377  if (outputInfo.isBlockEnd())
378  {
379  endBlock();
380  }
381 
382  std::cout << msg[i];
383 
384  if (outputInfo.isBlockStart())
385  {
386  startBlock();
387  }
388  std::cout << std::endl;
389  }
390  }
391  else
392  {
393  // Show some points so the user knows something is happening.
394  // This display is not sophisticated.
395  if (_indentLevel == (int)_maxStepLevel + 1)
396  {
397  indent(_indentLevel);
398  std::cout << "........................................" << std::endl;
399  }
400  }
401 
402 
403  if (outputInfo.isBlockStart())
404  {
405  _indentLevel++;
406  }
407 
408  }

413 void NOMAD::OutputQueue::flushStatsToStdout(const NOMAD::StatsInfo *statsInfo)
414 {
415  // Early out
416  if (nullptr == statsInfo)
417  {
418  return;
419  }
420 
421  if (nullptr == _params)
422  {
423  throw NOMAD::Exception(__FILE__, __LINE__, "OutputQueue: Display Parameters are NULL");
424  }
425 
426  bool displayInfeasible = _params->getAttributeValue<bool>("DISPLAY_INFEASIBLE");
427  bool displayUnsuccessful = _params->getAttributeValue<bool>("DISPLAY_UNSUCCESSFUL");
428  bool displayAllEval = _params->getAttributeValue<bool>("DISPLAY_ALL_EVAL");
429  size_t displayHeaderFreq = _params->getAttributeValue<size_t>("DISPLAY_HEADER");
430  if ( _maxOutputLevel > NOMAD::OutputLevel::LEVEL_NORMAL)
431  {
432  // Do not display header if display is already heavy.
433  displayHeaderFreq = NOMAD::INF_SIZE_T;
434  }
435  auto displayStatsFormat = _params->getAttributeValue<NOMAD::ArrayOfString>("DISPLAY_STATS");
436  bool displayInteresting = statsInfo->alwaysDisplay(displayInfeasible, displayUnsuccessful, false);
437 
438  if (displayAllEval || displayInteresting)
439  {
440  // Set starSuccess and appendComment to false for intermediate computations.
441  // They will be set to appropriate values just before display to
442  // standard output.
443  bool starSuccess = false;
444  bool appendComment = false;
445 
446  // Update objWidth if OBJ went larger.
447  NOMAD::DisplayStatsTypeList list;
448  list.add("OBJ");
449  auto solFormat = _params->getAttributeValue<NOMAD::ArrayOfDouble>("SOL_FORMAT");
450  std::string objString = statsInfo->display(list, solFormat, _objWidth, _hWidth, starSuccess, appendComment);
451  if (objString.size() > _objWidth)
452  {
453  _objWidth = objString.size();
454  }
455  list.clear();
456  list.add("CONS_H");
457  std::string hString = statsInfo->display(list, solFormat, _objWidth, _hWidth, starSuccess, appendComment);
458  if (hString.size() > _hWidth)
459  {
460  _hWidth = hString.size();
461  }
462  // Also consider H_MAX
463  list.clear();
464  list.add("H_MAX");
465  hString = statsInfo->display(list, solFormat, _objWidth, _hWidth, starSuccess, appendComment);
466  if (hString.size() > _hWidth)
467  {
468  _hWidth = hString.size();
469  }
470 
471  // Display column titles every DISPLAY_HEADER lines.
472  if ( (displayHeaderFreq < NOMAD::INF_SIZE_T )
473  && (0 == (_statsLineCount % displayHeaderFreq)))
474  {
475  if (_statsLineCount > 0)
476  {
477  std::cout << std::endl;
478  }
479  std::cout << statsInfo->displayHeader(displayStatsFormat, solFormat, _objWidth) << std::endl;
480  }
481 
482  // To standard output
483  // Set starSuccess: On standard output, add a star for successful evaluations,
484  // when either all evaluations or all unsuccessful evaluations are shown.
485  starSuccess = displayAllEval || displayUnsuccessful;
486  appendComment = true;
487  std::cout << statsInfo->display(displayStatsFormat, solFormat, _objWidth, _hWidth, starSuccess, appendComment) << std::endl;
488  _statsLineCount++;
489  }
490 }

511 void NOMAD::OutputQueue::flushStatsToStatsFile(const NOMAD::StatsInfo *statsInfo)
512 {
513  // Early out
514  if (_statsFile.empty() || nullptr == statsInfo)
515  {
516  return;
517  }
518 
519  if (nullptr == _params)
520  {
521  throw NOMAD::Exception(__FILE__, __LINE__, "OutputQueue: Display Parameters are NULL");
522  }
523  // Display this statsInfo (to standard output and to stats file)
524  // only if parameter DISPLAY_ALL_EVAL is true, and if it
525  // is interesting to display.
526  bool displayInfeasible = _params->getAttributeValue<bool>("DISPLAY_INFEASIBLE");
527  bool displayUnsuccessful = _params->getAttributeValue<bool>("DISPLAY_UNSUCCESSFUL");
528  bool displayInteresting = statsInfo->alwaysDisplay(displayInfeasible, displayUnsuccessful, true);
529  auto n = _params->getAttributeValue<NOMAD::ArrayOfDouble>("SOL_FORMAT").size();
530  NOMAD::ArrayOfDouble solFormatStats(n, NOMAD::DISPLAY_PRECISION_FULL);
531 
532  if (displayInteresting)
533  {
534  // Add stats information.
535  _statsStream << statsInfo->display(_statsFileFormat, solFormatStats, 0, 0, false, false) << std::endl;
536  _statsWritten = true;
537  }
