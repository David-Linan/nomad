55 // Class interface for the cache.
56 /**
57  * Used by algorithm and step classes.
58  * The CacheInterface takes care of converting points from subproblems
59  to full dimension before adding them to the cache, and from
60  full dimension to subproblems when retrieving them from the cache.
61  */

133  \return The number of points found
134  */
135  size_t find(std::function<bool(const EvalPoint&)> crit,
136  std::vector<EvalPoint> &evalPointList,
137  bool findInSubspace = false ) const;

145  size_t getAllPoints(std::vector<EvalPoint> &evalPointList) const;
