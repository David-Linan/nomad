66 void NOMAD::Search::init()
67 {
68  _name = "Search";
69  verifyParentNotNull();
70 
71  auto speculativeSearch = std::make_shared<NOMAD::SpeculativeSearchMethod>(this);
72  auto userSearch = std::make_shared<NOMAD::UserSearchMethod>(this);
73  auto quadSearch = std::make_shared<NOMAD::QuadSearchMethod>(this);
74  auto sgtelibSearch = std::make_shared<NOMAD::SgtelibSearchMethod>(this);
75  auto lhSearch = std::make_shared<NOMAD::LHSearchMethod>(this);
76  auto nmSearch = std::make_shared<NOMAD::NMSearchMethod>(this);
77 
78 
79  // The search methods will be executed in the same order
80  // as they are inserted.
81  // This is the order for NOMAD 3:
82  // 1. speculative search
83  // 2. user search
84  // 3. trend matrix basic line search
85  // 4. cache search
86  // 5. Model Searches
87  // 6. VNS search
88  // 7. Latin-Hypercube (LH) search
89  // 8. NelderMead (NM) search
90 
91  _searchMethods.push_back(speculativeSearch); // 1. speculative search
92  _searchMethods.push_back(userSearch); // 2. user search
93  _searchMethods.push_back(quadSearch); // 5a. Quad Model Searches
94  _searchMethods.push_back(sgtelibSearch); // 5b. Model Searches
95  _searchMethods.push_back(lhSearch); // 7. Latin-Hypercube (LH) search
96  _searchMethods.push_back(nmSearch); // 8. NelderMead (NM) search
97 }

108 bool NOMAD::Search::runImp()
109 {
110  bool searchSuccessful = false;
111  std::string s;
112 
113  // Sanity check. The runImp function should be called only when trial points are generated and evaluated for each search method separately.
114  verifyGenerateAllPointsBeforeEval(__PRETTY_FUNCTION__, false);
115 
116  if (!isEnabled())
117  {
118  // Early out --> no found better!
119  OUTPUT_DEBUG_START
120  AddOutputDebug("Search method is disabled. Early out.");
121  OUTPUT_DEBUG_END
122  return false;
123  }
124 
125 
126  NOMAD::SuccessType bestSuccessYet = NOMAD::SuccessType::NOT_EVALUATED;
127  NOMAD::SuccessType success = NOMAD::SuccessType::NOT_EVALUATED;
128 
129  // Go through all search methods until we get a success.
130  OUTPUT_DEBUG_START
131  s = "Going through all search methods until we get a success";
132  AddOutputDebug(s);
133  OUTPUT_DEBUG_END
134  for (size_t i = 0; !searchSuccessful && i < _searchMethods.size(); i++)
135  {
136  auto searchMethod = _searchMethods[i];
137  bool enabled = searchMethod->isEnabled();
138  OUTPUT_DEBUG_START
139  s = "Search method " + searchMethod->getName() + (enabled ? " is enabled" : " not enabled");
140  AddOutputDebug(s);
141  OUTPUT_DEBUG_END
142  if (!enabled) { continue; }
143 #ifdef TIME_STATS
144  double searchStartTime = NOMAD::Clock::getCPUTime();
145  double searchEvalStartTime = NOMAD::EvcInterface::getEvaluatorControl()->getEvalTime();
146 #endif // TIME_STATS
147  searchMethod->start();
148  searchMethod->run();
149  success = searchMethod->getSuccessType();
150  searchSuccessful = (success >= NOMAD::SuccessType::FULL_SUCCESS);
151  if (success > bestSuccessYet)
152  {
153  bestSuccessYet = success;
154  }
155  searchMethod->end();
156 #ifdef TIME_STATS
157  _searchTime[i] += NOMAD::Clock::getCPUTime() - searchStartTime;
158  _searchEvalTime[i] += NOMAD::EvcInterface::getEvaluatorControl()->getEvalTime() - searchEvalStartTime;
159 #endif // TIME_STATS
160 
161 
162  if (searchSuccessful)
163  {
164  // Do not go through the other search methods if a search is
165  // successful.
166  OUTPUT_INFO_START
167  s = searchMethod->getName();
168  s += " is successful. Stop reason: ";
169  s += _stopReasons->getStopReasonAsString() ;
170 
171  AddOutputInfo(s);
172  OUTPUT_INFO_END
173  break;
174  }
175  }
176 
177  setSuccessType(bestSuccessYet);
178 
179  return searchSuccessful;
180 }

183 void NOMAD::Search::endImp()
184 {
185  // Sanity check. The endImp function should be called only when trial points are generated and evaluated for each search method separately.
186  verifyGenerateAllPointsBeforeEval(__PRETTY_FUNCTION__, false);
187 
188  if (!isEnabled())
189  {
190  // Early out
191  return;
192  }
193 
194  // Need to reset the EvalStopReason if a sub optimization is used during Search and the max bb is reached for this sub optimization
195  auto evc = NOMAD::EvcInterface::getEvaluatorControl();
196  if (evc->testIf(NOMAD::EvalMainThreadStopType::LAP_MAX_BB_EVAL_REACHED))
197  {
198  evc->setStopReason(NOMAD::getThreadNum(), NOMAD::EvalMainThreadStopType::STARTED);
199  }
200 
201 }

204 void NOMAD::Search::generateTrialPoints()
205 {
206  // Sanity check. The generateTrialPoints function should be called only when trial points are generated for all each search method. After that all trials points evaluated at once.
207  verifyGenerateAllPointsBeforeEval(__PRETTY_FUNCTION__, true);
208 
209  for (auto searchMethod : _searchMethods)
210  {
211  if (searchMethod->isEnabled())
212  {
213  searchMethod->generateTrialPoints();
214 
215  // Aggregation of trial points from several search methods.
216  // The trial points produced by a search method are already snapped on bounds and on mesh.
217  auto searchMethodPoints = searchMethod->getTrialPoints();
218  for (auto point : searchMethodPoints)
219  {
220  // NOTE trialPoints includes points from multiple SearchMethods.
221  insertTrialPoint(point);
222  }
223  }
224  }
225 }
