67 bool NOMAD::Mads::runImp()
68 {
69  size_t k = 1; // Iteration number
70  NOMAD::SuccessType megaIterSuccess = NOMAD::SuccessType::NOT_EVALUATED;
71 
72  bool successFound = false;
73 
74  if (!_termination->terminate(k))
75  {
76  std::shared_ptr<NOMAD::MeshBase> mesh;
77  std::shared_ptr<NOMAD::Barrier> barrier;
78 
79  if (nullptr != _megaIteration)
80  {
81  // Case hot restart
82  k = _megaIteration->getK();
83  barrier = _megaIteration->getBarrier();
84 
85  // Downcast from MegaIteration to MadsMegaIteration
86  mesh = (std::dynamic_pointer_cast<NOMAD::MadsMegaIteration> (_megaIteration ))->getMesh();
87  megaIterSuccess = _megaIteration->getSuccessType();
88  }
89  else
90  {
91  mesh = dynamic_cast<NOMAD::MadsInitialization*>(_initialization.get())->getMesh();
92  barrier = _initialization->getBarrier();
93  }
94 
95  // Mads member _megaIteration is used for hot restart (read and write),
96  // as well as to keep values used in Mads::end(), and may be used for _termination.
97  // Update it here.
98  _megaIteration = std::make_shared<NOMAD::MadsMegaIteration>(this, k, barrier, mesh, megaIterSuccess);
99 
100 
101  while (!_termination->terminate(k))
102  {
103  // Create an MegaIteration: manage multiple iterations on different
104  // meshes and with different frame centers at the same time.
105  NOMAD::MadsMegaIteration megaIteration(this, k, barrier, mesh, megaIterSuccess);
106  megaIteration.start();
107  megaIteration.run();
108  megaIteration.end();
109 
110  // Remember these values to construct the next MegaIteration.
111  k = megaIteration.getNextK();
112  barrier = megaIteration.getBarrier();
113  mesh = megaIteration.getMesh();
114  megaIterSuccess = megaIteration.getSuccessType();
115 
116  if (!successFound && megaIterSuccess >= NOMAD::SuccessType::FULL_SUCCESS)
117  {
118  successFound = true;
119  }
120 
121  if (_userInterrupt)
122  {
123  hotRestartOnUserInterrupt();
124  }
125  }
126  }
127 
128  _termination->start();
129  _termination->run();
130  _termination->end();
131 
132  return successFound;
133 }

136 void NOMAD::Mads::hotRestartOnUserInterrupt()
137 {
138  if (_stopReasons->checkTerminate())
139  {
140  return;
141  }
142 #ifdef TIME_STATS
143  if (isRootAlgo())
144  {
145  _totalCPUAlgoTime += NOMAD::Clock::getCPUTime() - _startTime;
146  }
147 #endif // TIME_STATS
148  hotRestartBeginHelper();
149 
150  // Reset mesh because parameters have changed.
151  std::stringstream ss;
152  auto mesh = getIterationMesh();
153  if (nullptr != mesh)
154  {
155  ss << *mesh;
156  // Reset pointer
157  mesh.reset();
158  mesh = std::make_shared<NOMAD::GMesh>(_pbParams);
159  // Get old mesh values
160  ss >> *mesh;
161  }
162 
163  hotRestartEndHelper();
164 #ifdef TIME_STATS
165  if (isRootAlgo())
166  {
167  _startTime = NOMAD::Clock::getCPUTime();
168  }
169 #endif // TIME_STATS
170 }

173 void NOMAD::Mads::readInformationForHotRestart()
174 {
175  // Restart from where we were before.
176  // For this, we need to read some files.
177  // Note: Cache file is treated independently from hot restart file.
178 
179  if (_runParams->getAttributeValue<bool>("HOT_RESTART_READ_FILES"))
180  {
181  // Verify the files exist and are readable.
182  std::string hotRestartFile = _runParams->getAttributeValue<std::string>("HOT_RESTART_FILE");
183  if (NOMAD::checkReadFile(hotRestartFile))
184  {
185  std::string s = "Read hot restart file " + hotRestartFile;
186  NOMAD::OutputQueue::Add(s, NOMAD::OutputLevel::LEVEL_NORMAL);
187 
188  // Create a GMesh and an MadsMegaIteration with default values, to be filled
189  // by istream is.
190  // NOTE: Working in full dimension
191  auto barrier = std::make_shared<NOMAD::Barrier>(NOMAD::INF, NOMAD::Point(_pbParams->getAttributeValue<size_t>("DIMENSION")), NOMAD::EvalType::BB);
192  std::shared_ptr<NOMAD::MeshBase> mesh = std::make_shared<NOMAD::GMesh>(_pbParams);
193 
194  _megaIteration = std::make_shared<NOMAD::MadsMegaIteration>(this, 0, barrier, mesh, NOMAD::SuccessType::NOT_EVALUATED);
195 
196  // Here we use Algorithm::operator>>
197  NOMAD::read<NOMAD::Mads>(*this, hotRestartFile);
198  }
199  }
