63 /// Class for handling the types of LH search.
64 /**
65  * An initial LH search can be used instead of providing X0. \n
66  * The secondary LH searches can be done at each iteration if no
67  other searches has obtained a success.
68  */
69 class LHSearchType
70 {
71 private:
72  bool _enable; ///< Flag for calling LH search.
73  size_t _lhsearch0; ///< Number of points for initial LH search.
74  size_t _lhsearch1; ///< Number of points for secondary LH searches.
75 
76 public:
77  /// Constructor
78  /**
79  \param entries The parameters for initial search and secondary searches given as a \c string -- \b IN.
80  */
81  LHSearchType(const std::string& entries = "") ;
82 
83 
84  /// Test if LH search is enabled
85  /**
86  \return \c true if LH search is enabled \c false if not.
87  */
88  bool isEnabled() const { return _enable; }
89 
90  /// Number of initial LH search points
91  /**
92  \return The number of points for initial LH search.
93  */
94  size_t getNbInitial() const { return _lhsearch0; }
95 
96  /// Number of secondary LH search points at each iteration
97  /**
98  \return The number of points for secondary LH searches.
99  */
100  size_t getNbIteration() const { return _lhsearch1; }
101 
102  /// Comparison operator \c ==.
103  /**
104  \param lhst The right-hand side object -- \b IN.
105  \return A boolean equal to \c true if \c *this \c == \c array.
106  */
107  bool operator== (const LHSearchType & lhst) const
108  {
109  return (( _enable == lhst._enable) && ( _lhsearch0 == lhst._lhsearch0 ) && ( _lhsearch1 == lhst._lhsearch1 )) ;
110  }
111 
112  /// Comparison operator \c !=.
113  /**
114  \param lhst The right-hand side object -- \b IN.
115  \return A boolean equal to \c true if \c *this \c != \c array.
116  */
117  bool operator!= (const LHSearchType &lhst ) const { return !(*this == lhst); }
118 
119 
120 };
