62 bool NOMAD::NMInitializeSimplex::runImp()
63 {
64  bool simplexCreated = false;
65  if (nullptr == _nmY)
66  {
67  throw NOMAD::Exception(__FILE__, __LINE__, "The simplex is not defined.");
68  }
69 
70  // Create a simplex from EvalPoints in Cache or in Barrier
71  if (_nmY->empty())
72  {
73  simplexCreated = createSimplex();
74  }
75  else
76  {
77  OUTPUT_INFO_START
78  AddOutputInfo("Simplex already initialized: " + std::to_string(_nmY->size()) + " points");
79  OUTPUT_INFO_END
80  simplexCreated = true;
81  }
82 
83  return simplexCreated;
84 }

88 /* Create initial sets of points for Nelder-Mead within a radius of current best */
89 /*----------------------------------------------------------------------------------*/
90 bool NOMAD::NMInitializeSimplex::createSimplex()
91 {
92  auto evalType = NOMAD::EvcInterface::getEvaluatorControl()->getEvalType();
93  auto computeType = NOMAD::EvcInterface::getEvaluatorControl()->getComputeType();
94 
95  auto iter = dynamic_cast<const NOMAD::NMIteration*>( NOMAD::Step::_parentStep );
96  if (nullptr == iter)
97  {
98  throw NOMAD::Exception(__FILE__, __LINE__, "The simplex initialization must have a NMIteration Step as parent");
99  }
100 
101  std::vector<NOMAD::EvalPoint> evalPointList;
102  std::shared_ptr<NOMAD::EvalPoint> centerPt = nullptr;
103  auto barrier = getMegaIterationBarrier();
104  if (nullptr != barrier)
105  {
106  evalPointList = barrier->getAllPoints();
107  centerPt = std::make_shared<NOMAD::EvalPoint>(evalPointList[0]);
108  }
109  if (nullptr == centerPt)
110  {
111  throw NOMAD::Exception(__FILE__, __LINE__, "A center point must be defined.");
112  }
113  if ( ! centerPt->getEval(evalType)->isBBOutputComplete() )
114  {
115  throw NOMAD::Exception(__FILE__, __LINE__, "A center point must have all output defined for functions (OBJ, PB, EB).");
116  }
117 
118  // Clear the list of NM points
119  _nmY->clear();
120 
121  // The pb params handle only variables (no need to consider fixed variables)
122  size_t n = _pbParams->getAttributeValue<size_t>("DIMENSION");
123  size_t minYSize = n + 1;
124 
125  // compute the include radius: points in Y must be at
126  // a max distance of centerPt
127  auto includeLength = _runParams->getAttributeValue<NOMAD::Double>("NM_SIMPLEX_INCLUDE_LENGTH");
128  NOMAD::ArrayOfDouble includeRectangle(n, includeLength ) ;
129 
130  NOMAD::OutputInfo dbgInfo(_name,"Insertion of potential points to include in initial Y: ", NOMAD::OutputLevel::LEVEL_DEBUG);
131 
132  // If a mesh and include factor are supplied: the max distance is include factor times Delta
133  // Else we use include length
134  auto mesh = iter->getMesh(); // The mesh can be null
135  auto includeFactor = _runParams->getAttributeValue<size_t>("NM_SIMPLEX_INCLUDE_FACTOR");
136  if ( mesh != nullptr && includeFactor > 0 && includeFactor < NOMAD::INF_SIZE_T )
137  {
138  // The max distance is NM_search_include_factor times Delta
139 
140  includeRectangle = mesh->getDeltaFrameSize() ;
141 
142  if ( ! includeRectangle.isDefined() )
143  throw NOMAD::Exception(__FILE__, __LINE__, "The frame size is not defined.");
144 
145  includeRectangle *= includeFactor ;
146 
147  OUTPUT_DEBUG_START
148  dbgInfo.addMsg("The include rectangle: " + includeRectangle.display() );
149  OUTPUT_DEBUG_END
150  }
151  if ( includeRectangle.max() == 0 )
152  throw NOMAD::Exception(__FILE__, __LINE__, "The include rectangle has no volume");
153 
154  // The set of points initially included
155  NOMAD::NMSimplexEvalPointSet T;
156 
157  if (NOMAD::EvcInterface::getEvaluatorControl()->getUseCache())
158  {
159  // browse the cache:
160  NOMAD::CacheInterface cacheInterface(this);
161  cacheInterface.getAllPoints(evalPointList);
162  }
163  else
164  {
165  if (nullptr != barrier)
166  {
167  evalPointList = barrier->getAllPoints();
168  }
169  }
170 
171  // variables used to limit display
172  const size_t maxPointsToDisplay = 4;
173  size_t nbPoints = 0;
174  for ( const auto & cur : evalPointList )
175  {
176  if ( cur.getEvalStatus(evalType) == NOMAD::EvalStatusType::EVAL_OK &&
177  cur.getX()->size() == n )
178  {
179  if ( cur.getEval(evalType)->isBBOutputComplete() )
180  {
181  // the center point has been found and put in list
182  if ( *centerPt == cur )
183  {
184  NOMAD::EvalPoint Y ( cur );
185  T.insert ( Y );
186  if (nbPoints < maxPointsToDisplay)
187  {
188  OUTPUT_DEBUG_START
189  dbgInfo.addMsg(Y.display());
190  OUTPUT_DEBUG_END
191  nbPoints++;
192  }
193  }
194  // other points must be within the include region:
195  else
196  {
197  bool include = true;
198  for (size_t i = 0 ; i < n ; i++ )
199  {
200  NOMAD::Double val = cur[i] - (*centerPt)[i];
201  if ( val.abs() > includeRectangle[i] )
202  {
203  include = false;
204  break;
205  }
206  }
207  if ( include )
208  {
209 
210  // Issue #382: make sure to evaluate f or h for points in cache (important if cache is loaded from file) see
211  NOMAD::EvalPoint Y ( cur );
212  std::pair<NMSimplexEvalPointSetIterator,bool> ret = T.insert ( Y );
213 
214 
215  if ( ! ret.second )
216  {
217  OUTPUT_DEBUG_START
218  dbgInfo.addMsg("Cannot insert a point in Y (probably tied with another point): " + Y.display() );
219  OUTPUT_DEBUG_END
220  }
221  else
222  {
223  if (nbPoints <= maxPointsToDisplay)
224  {
225  OUTPUT_DEBUG_START
226  dbgInfo.addMsg( ((nbPoints < maxPointsToDisplay) ? Y.display() : "...") );
227  OUTPUT_DEBUG_END
228  nbPoints++;
229  }
230  }
231  }
232  }
233  }
234  }
235  }
236  OUTPUT_DEBUG_START
237  NOMAD::OutputQueue::Add(std::move(dbgInfo));
238  OUTPUT_DEBUG_END
239 
240  OUTPUT_INFO_START
241  AddOutputInfo("Number of potential points to include in initial Y: " + std::to_string(T.size()) );
242  OUTPUT_INFO_END
243 
244  //
245  auto nmStopReason = NOMAD::AlgoStopReasons<NOMAD::NMStopType>::get ( getAllStopReasons() );
246 
247  // not enough points (this is not counted as an error):
248  if ( T.size() < minYSize )
249  {
250  nmStopReason->set ( NMStopType::INITIAL_FAILED );
251  OUTPUT_INFO_START
252  AddOutputInfo("Stop NM because not enough points in Y.");
253  OUTPUT_INFO_END
254  return false;
255  }
256 
257 
258 
259  // Add points in simplex to obtain dim = n+1 and simplex affinely independant
260  NOMAD::NMSimplexEvalPointSetIterator itT = T.begin();
261 
262  // For debugging
263  NOMAD::OutputInfo dbgInfo2(_name,"Proceed to simplex creation", NOMAD::OutputLevel::LEVEL_DEBUG);
264 
265  // First point is always added
266  _nmY->insert ( *itT );
267 
268  OUTPUT_DEBUG_START
269  dbgInfo2.addMsg("k=0: Point z0:" + (*itT).display() ) ;
270  dbgInfo2.addMsg(" ---> z0 KEPT in Y ");;
271  OUTPUT_DEBUG_END
272 
273  int count_feasible = 0;
274 
275  if ( (*itT).getH(evalType, computeType).isDefined()
276  && (*itT).isFeasible(evalType, computeType) )
277  count_feasible = 1 ;
278 
279  itT++;
280  int k=1;
281  while ( itT != T.end() && ! nmStopReason->checkTerminate() )
282  {
283 
284  if ( _nmY->size() == minYSize )
285  break;
286 
287  OUTPUT_DEBUG_START
288  dbgInfo2.addMsg("k=" + std::to_string(k) +": Point zk:" + (*itT).display()) ;
289  OUTPUT_DEBUG_END
290 
291  std::pair<NMSimplexEvalPointSetIterator,bool> ret = _nmY->insert ( *itT );
292 
293  if ( ! ret.second )
294  {
295  nmStopReason->set ( NMStopType::INITIAL_FAILED );
296  OUTPUT_INFO_START
297  AddOutputInfo("Stop NM because cannot insert a point in Y.");
298  OUTPUT_INFO_END
299  break;
300  }
301 
302  int rank = getRankDZ();
303 
304  if ( rank <= 0 )
305  {
306  nmStopReason->set ( NMStopType::INITIAL_FAILED );
307  OUTPUT_INFO_START
308  AddOutputInfo("Cannot get rank of DZ=[(y1-y0 (y2-y0) ...(yk-y0)].");
309  OUTPUT_INFO_END
310  break;
311  }
312 
313  // Erase last point or not
314  if ( rank != k )
315  {
316  OUTPUT_DEBUG_START
317  dbgInfo2.addMsg(" ---> zk NOT KEPT in Y ");
318  OUTPUT_DEBUG_END
319  _nmY->erase( ret.first );
320  itT++;
321  }
322  else
323  {
324  OUTPUT_DEBUG_START
325  dbgInfo2.addMsg( " ---> zk KEPT in Y " );
326  OUTPUT_DEBUG_END
327  if ( (*itT).isFeasible(evalType, computeType) )
328  count_feasible++;
329  k++;
330  itT++;
331  }
332 
333  }
334  OUTPUT_DEBUG_START
335  NOMAD::OutputQueue::Add(std::move(dbgInfo2));
336  OUTPUT_DEBUG_END
337 
338  // not enough points or insufficient rank of simplex (this is not counted as an error):
339  // Erase point
340  if ( _nmY->size() < minYSize )
341  {
342  nmStopReason->set ( NMStopType::INITIAL_FAILED );
343  OUTPUT_INFO_START
344  AddOutputInfo( "Stop NM because not enough simplex points in Y." );
345  OUTPUT_INFO_END
346  return false;
347  }
348  if ( getRankDZ() != (int)n )
349  {
350  nmStopReason->set ( NMStopType::INITIAL_FAILED );
351  OUTPUT_INFO_START
352  AddOutputInfo( "Stop NM because rank of Y < n." );
353  OUTPUT_INFO_END
354  return false;
355  }
356 
357 
358 
359  // Update simplex characteristics (volumes and diameter)
360  updateYCharacteristics();
361 
362  displayYInfo();
363 
364  return true;
365 
366 }
