62 void NOMAD::LHSearchMethod::generateTrialPointsImp()
63 {
64  if (nullptr == _iterAncestor)
65  {
66  throw NOMAD::Exception(__FILE__,__LINE__,"LHSearchMethod: must have an iteration ancestor");
67  }
68  auto mesh = _iterAncestor->getMesh();
69  if (nullptr == mesh)
70  {
71  throw NOMAD::Exception(__FILE__,__LINE__,"LHSearchMethod: must have a mesh");
72  }
73 
74  // The frame center is only used to compute bounds, if they are not defined.
75  // Use the first available point.
76  auto barrier = getMegaIterationBarrier();
77  if (nullptr == barrier)
78  {
79  throw NOMAD::Exception(__FILE__,__LINE__,"LHSearchMethod: must have a MadsMegaIteration ancestor with a barrier");
80  }
81  auto frameCenter = barrier->getFirstPoint();
82 
83  auto lhSearch = _runParams->getAttributeValue<NOMAD::LHSearchType>("LH_SEARCH");
84  size_t n = _pbParams->getAttributeValue<size_t>("DIMENSION");
85  size_t p = (0 == _iterAncestor->getK()) ? lhSearch.getNbInitial() : lhSearch.getNbIteration();
86  auto lowerBound = _pbParams->getAttributeValue<NOMAD::ArrayOfDouble>("LOWER_BOUND");
87  auto upperBound = _pbParams->getAttributeValue<NOMAD::ArrayOfDouble>("UPPER_BOUND");
88 
89  NOMAD::ArrayOfDouble deltaFrameSize = mesh->getDeltaFrameSize();
90  NOMAD::Double scaleFactor = sqrt(-log(NOMAD::DEFAULT_EPSILON));
91  // Apply Latin Hypercube algorithm (provide frameCenter, deltaFrameSize, and scaleFactor for updating bounds)
92  NOMAD::LHS lhs(n, p, lowerBound, upperBound, frameCenter, deltaFrameSize, scaleFactor);
93  auto pointVector = lhs.Sample();
94 
95  // Insert the point. Projection on mesh and snap to bounds is done in SearchMethod
96  for (auto point : pointVector)
97  {
98  // Insert point (if possible)
99  NOMAD::EvalPoint evalPoint(point);
100  evalPoint.setPointFrom(std::make_shared<NOMAD::EvalPoint>(frameCenter), NOMAD::SubproblemManager::getInstance()->getSubFixedVariable(this));
101  evalPoint.setGenStep(getName());
102  insertTrialPoint(evalPoint);
103  }
104 }
