79  NOMAD::EvcInterface::getEvaluatorControl()->setLapMaxBbEval(NOMAD::INF_SIZE_T);

81 
82  if ( ! _stopReasons->checkTerminate() )
83  {
84  foundBetter = evalTrialPoints(this);
85  }
86 
87  // Update MegaIteration success type with best success found.

100 // Generate new points to evaluate from Poll and Search
101 void NOMAD::MegaSearchPoll::generateTrialPoints()
102 {
103  verifyGenerateAllPointsBeforeEval(__PRETTY_FUNCTION__, true);
104  OUTPUT_INFO_START
105  AddOutputInfo("Generate points for " + _name, true, false);
106  OUTPUT_INFO_END
107 
108  NOMAD::EvalPointSet trialPoints;
109 
110  // Generate trial points for Search (all enabled search methods) and Poll.
111  // Note: Search and Poll generateTrialPoints() methods both
112  // take care of verifying that the generated are on mesh, and also
113  // update the "PointFrom" with the frame center.
114  NOMAD::Search search(this);
115  search.generateTrialPoints();
116  auto trialPointsSearch = search.getTrialPoints();
117 
118  NOMAD::Poll poll(this);
119  poll.generateTrialPoints();
120  auto trialPointsPoll = poll.getTrialPoints();
121 
122  // Merge two sets and remove duplicates
123  // Naive implementation. Easier to understand - I could not make std::merge,
124  // std::unique or std::set_union work fine.
125  // Caveat: Multiple EvalPoints copy.
126  for (auto point : trialPointsSearch)
127  {
128  insertTrialPoint(point);
129  }
130  for (auto point : trialPointsPoll)
131  {
132  insertTrialPoint(point);
133  }
134 
135  OUTPUT_INFO_START
136  AddOutputInfo("Generated " + NOMAD::itos(getTrialPointsCount()) + " points");
137  AddOutputInfo("Generate points for " + _name, false, true);
138  OUTPUT_INFO_END
139 
140 }
