94 void NOMAD::SgtelibModelInitialization::validateX0s() const
95 {
96  auto x0s = _pbParams->getAttributeValue<NOMAD::ArrayOfPoint>("X0");
97  size_t n = _pbParams->getAttributeValue<size_t>("DIMENSION");
98  bool validX0available = false;
99  std::string err;
100 
101  for (size_t x0index = 0; x0index < x0s.size(); x0index++)
102  {
103  auto x0 = x0s[x0index];
104  if (!x0.isComplete() || x0.size() != n)
105  {
106  err += "Initialization: eval_x0s: Invalid X0 " + x0.display() + ".";
107  }
108  else
109  {
110  validX0available = true;
111  }
112  }
113  if (validX0available)
114  {
115  if (!err.empty())
116  {
117  // Show invalid X0s
118  AddOutputWarning(err);
119  }
120  }
121  else
122  {
123  // No valid X0 available. Throw exception.
124  size_t cacheSize = NOMAD::CacheBase::getInstance()->size();
125  if (cacheSize > 0)
126  {
127  err += " Hint: Try not setting X0 so that the cache is used (";
128  err += std::to_string(cacheSize) + " points)";
129  }
130  else
131  {
132  err += ". Cache is empty.";
133  }
134  throw NOMAD::Exception(__FILE__, __LINE__, err);
135  }
136 
137 }

140 // Eval X0s, using blackbox.
141 // If we are here, it means we are in standalone mode. Either X0s were provided,
142 // or the best points were found in the cache, in MainStep, and put in parameter
143 // X0.
144 // Method is copied from MadsInitialization.
145 bool NOMAD::SgtelibModelInitialization::eval_x0s()
146 {
147  bool evalOk = false;
148 
149  auto x0s = _pbParams->getAttributeValue<NOMAD::ArrayOfPoint>("X0");
150 
151  validateX0s();
152 
153  // Add X0s that need evaluation to eval queue
154  NOMAD::CacheInterface cacheInterface(this);
155  NOMAD::EvcInterface evcInterface(this);
156  auto evc = evcInterface.getEvaluatorControl();
157  evc->lockQueue();
158 
159  NOMAD::EvalPointSet evalPointSet;
160  for (size_t x0index = 0; x0index < x0s.size(); x0index++)
161  {
162  auto x0 = x0s[x0index];
163  NOMAD::EvalPoint evalPoint_x0(x0);
164  evalPointSet.insert(evalPoint_x0);
165  }
166 
167  // Add points to the eval queue.
168  // Convert to full dimension if needed.
169  // Note: Queue is already locked - it needs to be locked to add points.
170  evcInterface.keepPointsThatNeedEval(evalPointSet, false); // false: no mesh
171 
172  // Enforce no opportunism.
173  auto previousOpportunism = evc->getOpportunisticEval();
174  evc->setOpportunisticEval(false);
175 
176  evc->unlockQueue(false); // false: do not sort eval queue
177 
178  // Evaluate all x0s. Ignore returned success type.
179  // Note: EvaluatorControl would not be able to compare/compute success since there is no barrier.
180  evcInterface.startEvaluation();
181 
182  // Reset opportunism to previous values.
183  evc->setOpportunisticEval(previousOpportunism);
184 
185  auto evalPointList = evcInterface.retrieveAllEvaluatedPoints();
186  for (auto x0 : x0s)
187  {
188  if (_stopReasons->checkTerminate())
189  {
190  break;
191  }
192  NOMAD::EvalPoint evalPoint_x0(x0);
193  cacheInterface.find(x0, evalPoint_x0, NOMAD::EvalType::BB);
194  // To evaluate X0, use blackbox, not sgtelib model.
195  if (evalPoint_x0.isEvalOk(NOMAD::EvalType::BB))
196  {
197  // evalOk is true if at least one evaluation is Ok
198  evalOk = true;
199  AddOutputInfo("Using X0: " + evalPoint_x0.displayAll());
200  }
201  else
202  {
203  AddOutputError("X0 evaluation failed for X0 = " + x0.display());
204  }
205  }
206 
207  if (evalOk)
208  {
209  // Construct barrier using x0s
210  auto hMax = _runParams->getAttributeValue<NOMAD::Double>("H_MAX_0");
211  _barrier = std::make_shared<NOMAD::Barrier>(hMax,
212  NOMAD::SubproblemManager::getInstance()->getSubFixedVariable(this),
213  evc->getEvalType(), evc->getComputeType(), evalPointList);
214  }
215  else
216  {
217  auto sgtelibModelStopReason = NOMAD::AlgoStopReasons<NOMAD::ModelStopType>::get(_stopReasons);
218  sgtelibModelStopReason->set(NOMAD::ModelStopType::X0_FAIL);
219  }
220 
221  NOMAD::OutputQueue::Flush();
222 
223  return evalOk;
224 }
