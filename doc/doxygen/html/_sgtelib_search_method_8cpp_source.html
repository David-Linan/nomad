60 void NOMAD::SgtelibSearchMethod::init()
61 {
62  setName("Sgtelib Search Method");
63  //setComment("(SgtelibModel)");
64  verifyParentNotNull();
65 
66  const auto parentSearch = getParentStep()->getParentOfType<NOMAD::SgtelibSearchMethod*>(false);
67  setEnabled((nullptr == parentSearch) && _runParams->getAttributeValue<bool>("SGTELIB_MODEL_SEARCH"));
68 #ifndef USE_SGTELIB
69  if (isEnabled())
70  {
71  OUTPUT_INFO_START
72  AddOutputInfo(_name + " cannot be performed because NOMAD is compiled without sgtelib library");
73  OUTPUT_INFO_END
74  setEnabled(false);
75  }
76 #endif
77 
78 #ifdef USE_SGTELIB
79  // Check that there is exactly one objective
80  if (isEnabled())
81  {
82  const auto bbot = NOMAD::SgtelibModel::getBBOutputType();
83  auto nbObj = NOMAD::getNbObj(bbot);
84  if (0 == nbObj)
85  {
86  OUTPUT_INFO_START
87  AddOutputInfo(_name + " not performed when there is no objective function");
88  OUTPUT_INFO_END
89  setEnabled(false);
90  }
91  else if (nbObj > 1)
92  {
93  OUTPUT_INFO_START
94  AddOutputInfo(_name + " not performed on multi-objective function");
95  OUTPUT_INFO_END
96  setEnabled(false);
97  }
98 
99  auto modelDisplay = _runParams->getAttributeValue<std::string>("SGTELIB_MODEL_DISPLAY");
100  _displayLevel = modelDisplay.empty()
101  ? NOMAD::OutputLevel::LEVEL_DEBUGDEBUG
102  : NOMAD::OutputLevel::LEVEL_INFO;
103 
104  // Create the SgtelibModel algorithm with its own stop reasons
105  auto stopReasons = std::make_shared<NOMAD::AlgoStopReasons<NOMAD::ModelStopType>>();
106  auto barrier = getMegaIterationBarrier();
107  const NOMAD::MadsIteration* iteration = getParentOfType<NOMAD::MadsIteration*>();
108  auto mesh = iteration->getMesh();
109  _modelAlgo = std::make_shared<NOMAD::SgtelibModel>(this, stopReasons,
110  barrier, _runParams,
111  _pbParams, mesh);
112  _modelAlgo->setEndDisplay(false);
113  }
114 #endif
115 }

156  /*----------------*/
157  oraclePoints = _modelAlgo->createOraclePoints();
158 
159  if (0 == oraclePoints.size())
160  {
161  OUTPUT_INFO_START
162  s = "Failed generating points. Stop " + _name;
163  AddOutputInfo(s, _displayLevel);
164  OUTPUT_INFO_END

172  }
173  else
