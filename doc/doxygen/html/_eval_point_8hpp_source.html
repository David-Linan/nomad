354 /// Class for eval point compare.
355 /**
356  * Compare the Point parts only.
357  * Trying to insert an EvalPoint in a set that has already a Point defined
358  * for that EvalPoint will return false.
359  */
360 class EvalPointCompare
361 {
362 public:
363  bool operator() (const EvalPoint& lhs, const EvalPoint& rhs) const
364  {
365  return Point::weakLess(*(lhs.getX()),*(rhs.getX()));
366  }
367 };
