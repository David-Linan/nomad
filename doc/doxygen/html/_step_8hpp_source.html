240  /// Template function to get the parent of given type.
241  /**
242  * Starting with parent of current Step, and going through ancestors,
243  get first Step that is of type T.
244  * By default, stop if an Algorithm is found. Returned Step could be
245  irrelevant otherwise. To go further up than an Algorithm, set optional
246  parameter stopAtAlgo to false.
247  */
248  template<typename T>
249  T getParentOfType(const bool stopAtAlgo = true) const
250  {
251  Step* retStep = nullptr;
252 
253  Step* step = const_cast<Step*>(_parentStep);
254  while (nullptr != step)
255  {
256  if (nullptr != dynamic_cast<T>(step))
257  {
258  retStep = step;
259  break;
260  }
261  else if (stopAtAlgo && step->isAnAlgorithm())
262  {
263  break;
264  }
265  step = const_cast<Step*>(step->getParentStep());
266  }
267 
268  return dynamic_cast<T>(retStep);
269  }
