106 void NOMAD::QuadSearchMethod::generateTrialPointsImp()
107 {
108 #ifdef USE_SGTELIB
109  // The trial points are generated for a feasible frame center and an infeasible one.
110  if ( ! _stopReasons->checkTerminate() )
111  {
112  auto madsIteration = getParentOfType<MadsIteration*>();
113 
114  // MegaIteration's barrier member is already in sub dimension.
115  auto bestXFeas = madsIteration->getMegaIterationBarrier()->getFirstXFeas();
116  auto bestXInf = madsIteration->getMegaIterationBarrier()->getFirstXInf();
117 
118  auto evalType = NOMAD::EvcInterface::getEvaluatorControl()->getEvalType();
119  auto computeType = NOMAD::EvcInterface::getEvaluatorControl()->getComputeType();
120  if (nullptr != bestXFeas
121  && bestXFeas->getF(evalType, computeType).isDefined()
122  && bestXFeas->getF(evalType, computeType) < MODEL_MAX_OUTPUT)
123  {
124  NOMAD::QuadModelSinglePass singlePassFeas(this, bestXFeas, madsIteration->getMesh());
125 
126  // Generate the trial points
127  singlePassFeas.generateTrialPoints();
128 
129  // Pass the generated trial pts to this
130  auto trialPtsSinglePassFeas = singlePassFeas.getTrialPoints();
131  for (auto evalPoint : trialPtsSinglePassFeas)
132  {
133  evalPoint.setPointFrom(bestXFeas, NOMAD::SubproblemManager::getInstance()->getSubFixedVariable(this));
134  insertTrialPoint(evalPoint);
135  }
136  }
137  if (nullptr != bestXInf
138  && bestXInf->getF(evalType, computeType).isDefined()
139  && bestXInf->getF(evalType, computeType) < MODEL_MAX_OUTPUT
140  && bestXInf->getH(evalType, computeType).isDefined()
141  && bestXInf->getH(evalType, computeType) < MODEL_MAX_OUTPUT)
142  {
143  NOMAD::QuadModelSinglePass singlePassInf(this, bestXInf, madsIteration->getMesh());
144 
145  // Generate the trial points
146  singlePassInf.generateTrialPoints();
147 
148  // Pass the generated trial pts to this
149  auto trialPtsSinglePassInf = singlePassInf.getTrialPoints();
150  for (auto evalPoint : trialPtsSinglePassInf)
151  {
152  evalPoint.setPointFrom(bestXInf, NOMAD::SubproblemManager::getInstance()->getSubFixedVariable(this));
153  insertTrialPoint(evalPoint);
154  }
155  }
156  }
157 #endif
158 }
