67 // Update the SGTELIB::TrainingSet and SGTELIB::Surrogate contained in the
68 // ancestor QuadModel (modelAlgo).
69 //
70 // 1- Get relevant points in cache, around current frame center.
71 // 2- Add points to training set, and build new model.
72 // 3- Assess if model is ready. Update its bounds.
73 //
74 // Note: Update uses blackbox values
75 bool NOMAD::QuadModelUpdate::runImp()
76 {
77  std::string s; // Used for output
78  bool updateSuccess = false;
79 
80  const NOMAD::QuadModelIteration * iter = getParentOfType<NOMAD::QuadModelIteration*>();
81 
82  if (nullptr == iter)
83  {
84  throw NOMAD::Exception(__FILE__,__LINE__,"Update must have a Iteration among its ancestors.");
85  }
86 
87  auto n = _pbParams->getAttributeValue<size_t>("DIMENSION");
88  const auto bbot = NOMAD::QuadModelAlgo::getBBOutputType();
89  size_t nbConstraints = NOMAD::getNbConstraints(bbot);
90  size_t nbModels = nbConstraints+1; /// Constraint plus a single objective
91 
92  // row_X and row_Z are one-liner matrices to stock current point.
93  SGTELIB::Matrix row_X("row_X", 1, static_cast<int>(n));
94  SGTELIB::Matrix row_Z("row_Z", 1, static_cast<int>(nbModels));
95 
96  // Matrices to stock all points to send to the model
97  auto add_X = std::make_shared<SGTELIB::Matrix>("add_X", 0, static_cast<int>(n));
98  auto add_Z = std::make_shared<SGTELIB::Matrix>("add_Z", 0, static_cast<int>(nbModels));
99 
100  auto model=iter->getModel();
101  auto trainingSet=iter->getTrainingSet();
102 
103 
104  // Go through cache points
105  AddOutputInfo("Review of the cache", _displayLevel);
106  NOMAD::Double v;
107 
108  _radiuses = NOMAD::ArrayOfDouble(n,INF);
109  //
110  // 1- Get relevant points in cache, around current frame centers.
111  //
112  std::vector<NOMAD::EvalPoint> evalPointList;
113  if (NOMAD::EvcInterface::getEvaluatorControl()->getUseCache())
114  {
115  _frameCenter = iter->getFrameCenter()->getX();
116  NOMAD::EvalPoint frameCenterEvalPoint;
117  NOMAD::CacheInterface cacheInterface(this);
118  if (0 == cacheInterface.find(*_frameCenter, frameCenterEvalPoint))
119  {
120  // In the context of using the cache, ensure the frame center is in it.
121  cacheInterface.smartInsert(*iter->getFrameCenter());
122  }
123 
124  // Select valid points that are close enough to frame centers
125  // Compute distances that must not be violated for each variable.
126  // Get a Delta (frame size) if available.
127 
128  if (nullptr != iter->getMesh())
129  {
130  _radiuses = iter->getMesh()->getDeltaFrameSize();
131 
132  // Multiply by radius parameter
133  auto radiusFactor = _runParams->getAttributeValue<NOMAD::Double>("QUAD_MODEL_RADIUS_FACTOR");
134  _radiuses *= radiusFactor;
135  }
136  OUTPUT_DEBUG_START
137  if (nullptr != iter->getMesh())
138  {
139  s = "Mesh size: " + iter->getMesh()->getdeltaMeshSize().display();
140  AddOutputInfo(s);
141  s = "Frame size: " + iter->getMesh()->getDeltaFrameSize().display();
142  AddOutputInfo(s);
143  }
144  s = "Radiuses: " + _radiuses.display();
145  AddOutputInfo(s);
146  OUTPUT_DEBUG_END
147 
148  // Get valid points: notably, they have a BB evaluation.
149  // Use CacheInterface to ensure the points are converted to subspace
150  auto crit = [&](const NOMAD::EvalPoint& evalPoint){return this->isValidForIncludeInModel(evalPoint);};
151  cacheInterface.find(crit, evalPointList, true /*find in subspace*/);
152  }
153 
154  // Minimum and maximum number of valid points to build a model
155  const size_t minNbPoints = _runParams->getAttributeValue<size_t>("SGTELIB_MIN_POINTS_FOR_MODEL");
156  // not using SGTELIB_MAX_POINTS_FOR_MODEL, see justification below.
157  //const size_t maxNbPoints = _runParams->getAttributeValue<size_t>("SGTELIB_MAX_POINTS_FOR_MODEL");
158 
159 
160  size_t nbValidPoints = evalPointList.size();
161 
162  /*
163  // This is not a safe way to select a subset of points.
164  // First, we should ensure that the frame center is part of that subset.
165  // Second, how to select the other points? The points closest to
166  // the frame center seem like an interesting option.
167  if (nbValidPoints > maxNbPoints)
168  {
169 
170  OUTPUT_INFO_START
171  s = "QuadModel found " + std::to_string(nbValidPoints);
172  s += " valid points to build model. Keep only ";
173  s += std::to_string(maxNbPoints);
174  AddOutputInfo(s);
175  OUTPUT_INFO_END
176 
177  // First, sort evalPointList by dominance
178  std::sort(evalPointList.begin(), evalPointList.end());
179  // Next, get first maxNbPoints points
180  std::vector<NOMAD::EvalPoint> evalPointListShort;
181  for (size_t i = 0; i < maxNbPoints; i++)
182  {
183  evalPointListShort.push_back(evalPointList[i]);
184  }
185  evalPointList = evalPointListShort;
186  nbValidPoints = evalPointList.size();
187  }
188  */
189 
190  if (nbValidPoints < minNbPoints)
191  {
192  // If no points available, it is impossible to build a model.
193  auto stopReason = NOMAD::AlgoStopReasons<NOMAD::ModelStopType>::get(_stopReasons);
194  stopReason->set(NOMAD::ModelStopType::NOT_ENOUGH_POINTS );
195 
196  OUTPUT_INFO_START
197  AddOutputInfo("QuadModel has not enough points to build model");
198  OUTPUT_INFO_END
199  }
200  OUTPUT_INFO_START
201  s = "Found " + std::to_string(nbValidPoints);
202  s += " point";
203  if (nbValidPoints > 1)
204  {
205  s += "s";
206  }
207  s += " in cache of size " + std::to_string(NOMAD::CacheBase::getInstance()->size());
208  AddOutputInfo(s, NOMAD::OutputLevel::LEVEL_INFO);
209  OUTPUT_INFO_END
210 
211  for (auto evalPoint : evalPointList)
212  {
213  NOMAD::Point x = *evalPoint.getX();
214  s = "xNew = " + evalPoint.display();
215  AddOutputInfo(s, _displayLevel);
216 
217  for (size_t j = 0; j < n; j++)
218  {
219  // X
220  row_X.set(0, static_cast<int>(j), x[j].todouble());
221  }
222  add_X->add_rows(row_X);
223 
224  // Objective
225  // Update uses blackbox values
226  if (!evalPoint.getF(NOMAD::EvalType::BB, NOMAD::ComputeType::STANDARD).isDefined())
227  {
228  s = "Error: In QuadModelUpdate, this point should have a BB Eval: ";
229  s += evalPoint.displayAll();
230  throw NOMAD::Exception(__FILE__,__LINE__,s);
231  }
232  row_Z.set(0, 0, evalPoint.getF(NOMAD::EvalType::BB, NOMAD::ComputeType::STANDARD).todouble()); // 1st column: constraint model
233 
234  NOMAD::ArrayOfDouble bbo = evalPoint.getEval(NOMAD::EvalType::BB)->getBBOutput().getBBOAsArrayOfDouble();
235  // Constraints
236  int k = 1;
237  for (size_t j = 0; j < bbo.size(); j++)
238  {
239  if (NOMAD::isConstraint(bbot[j]))
240  {
241  row_Z.set(0, k, bbo[j].todouble());
242  k++;
243  }
244  }
245  add_Z->add_rows(row_Z);
246 
247  } // Done going through valid points
248 
249  //
250  // 2- Add points to training set, and build new model.
251  //
252  if (add_X->get_nb_rows() > 0)
253  {
254  // Build the model
255  AddOutputInfo("Add points to training set...", _displayLevel);
256 
257  trainingSet->partial_reset_and_add_points(*add_X, *add_Z);
258 
259  AddOutputInfo("OK.", _displayLevel);
260  AddOutputInfo("Build model from training set...", _displayLevel);
261 
262  model->build();
263  AddOutputInfo("OK.", _displayLevel);
264  }
265 
266  //
267  // 3- Assess if model is ready. Update its bounds.
268  //
269  // Check if the model is ready
270  if ( model->is_ready() )
271  {
272  updateSuccess = true;
273  }
274  else
275  {
276  updateSuccess = false;
277  }
278 
279  OUTPUT_INFO_START
280  s = "New nb of points: " + std::to_string(trainingSet->get_nb_points());
281  AddOutputInfo(s, NOMAD::OutputLevel::LEVEL_INFO);
282  s = "Ready: " + NOMAD::boolToString(model->is_ready());
283  AddOutputInfo(s, NOMAD::OutputLevel::LEVEL_INFO);
284  OUTPUT_INFO_END
285 
286  return updateSuccess;
287 }

290 bool NOMAD::QuadModelUpdate::isValidForIncludeInModel(const NOMAD::EvalPoint& evalPoint) const
291 {
292  if ( ! _frameCenter->NOMAD::ArrayOfDouble::isComplete() || ! _radiuses.isComplete() )
293  {
294  std::cerr << "QuadModelUpdate : isValidForIncludeInModel : frameCenter or radiuses not defined " << std::endl;
295  }
296 
297  return (isValidForUpdate(evalPoint) && NOMAD::ArrayOfDouble(*evalPoint.getX() - *_frameCenter).abs() <= _radiuses ) ;
298 
299 }

302 bool NOMAD::QuadModelUpdate::isValidForUpdate(const NOMAD::EvalPoint& evalPoint) const
303 {
304  // Verify that the point is valid
305  // - Not a NaN
306  // - Not a fail
307  // - All outputs defined
308  // - Blackbox OBJ available (Not MODEL)
309  bool validPoint = true;
310  NOMAD::ArrayOfDouble bbo;
311 
312  auto evalType = NOMAD::EvcInterface::getEvaluatorControl()->getEvalType();
313  auto eval = evalPoint.getEval(evalType);
314  if (NOMAD::EvalType::BB != evalType)
315  {
316  validPoint = false;
317  }
318  else if (nullptr == eval)
319  {
320  // Eval must be available
321  validPoint = false;
322  }
323  else
324  {
325  // Note: it could be discussed if points that have h > hMax should still be used
326  // to build the model (Nomad 3). We validate them to comply with Nomad 3.
327  // If f or h greater than MODEL_MAX_OUTPUT (default=1E10) the point is not valid (same as Nomad 3)
328  if ( ! eval->isBBOutputComplete()
329  || !(NOMAD::EvalStatusType::EVAL_OK == eval->getEvalStatus())
330  || !eval->getF(NOMAD::ComputeType::STANDARD).isDefined()
331  || !eval->getH(NOMAD::ComputeType::STANDARD).isDefined()
332  || eval->getF(NOMAD::ComputeType::STANDARD) > NOMAD::MODEL_MAX_OUTPUT
333  || eval->getH(NOMAD::ComputeType::STANDARD) > NOMAD::MODEL_MAX_OUTPUT)
334  {
335  validPoint = false;
336  }
337  }
338 
339  return validPoint;
340 }
