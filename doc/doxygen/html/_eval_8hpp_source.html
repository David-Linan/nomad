75 /// Type for an evaluation status
76 /**
77  * \todo These statuses should be made 100% clear.
78  * Add something in the name that says if we can re-evaluate the point or not.
79  * Even add some User Cases.
80  */
81 enum class EvalStatusType
82 {
83  EVAL_NOT_STARTED, ///< Evaluation has not been done yet. Initial status.
84  EVAL_FAILED, ///< Evaluation failure. Do not re-submit.
85  EVAL_ERROR, ///< Evaluation did not proceed normally. May be submitted again.
86  EVAL_USER_REJECTED, ///< Evaluation was rejected by user. May be submitted again
87  EVAL_OK, ///< Correct evaluation
88  EVAL_IN_PROGRESS, ///< Evaluation in progress
89  EVAL_WAIT, ///< Evaluation in progress for another instance of the same point: Wait for evaluation to be done.
90  EVAL_STATUS_UNDEFINED ///< Undefined evaluation status
91 };

210  /// Dominance

214  * The feasible point x dominates the feasible point y \n
215  when f(x) < f(y).
216  * The infeasible point x dominates the infeasible point y \n
217  when f(x) <= f(y) and h(x) <= h(y),
218  with at least one strict inequality.
219  * Otherwise, return false
