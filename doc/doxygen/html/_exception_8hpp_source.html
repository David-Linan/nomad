60 /// Custom class for exceptions.
61 /**
62  * NOMAD uses this type of exceptions.
63  * It indicates the file and line number at which a throw is made.
64  *
65  * \b Example
66  *
67  * \code
68  * throw Exception(__FILE__, __LINE__, "an error message");
69  * \endcode
70  */
71 class Exception : public std::exception
72 {
73 
74 private:
75 
76  mutable std::string _what; ///< Error message.
77  std::string _file; ///< File where the exception is thrown.
78  size_t _line; ///< Line number at which the exception is thrown.
79 
80 protected:
81  std::string _typeMsg; ///< Basic exception message indicating the type of exception
82 
83 public:
84 
85  /// Constructor.
86  /**
87  \param file A string corresponding to the file where the
88  exception is thrown -- \b IN
89  \param line An integer corresponding to the line number
90  at which the exception is thrown -- \b IN.
91  \param msg A string corresponding to the error message -- \b IN.
92  */
93  Exception(const std::string& file, const size_t line, const std::string & msg)
94  : _what(msg),
95  _file(file),
96  _line(line),
97  _typeMsg("")
98  {}
99 
100  /// Destructor.
101  virtual ~Exception() throw() {}
102 
103  /// Access to the error message.
104  /**
105  \return A string with the error message.
106  */
107  const char * what() const throw();
108 };
