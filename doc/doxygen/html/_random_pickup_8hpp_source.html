65 /// Class for randomly picking up integers.
66 /**
67  - The integers are chosen in [0;n-1] and are distinct.
68  - Example displaying 5 different integers in [0;4]:
69  \code
70  NOMAD::RandomPickup rp(5);
71  for (size_t i = 0 ; i < 5 ; ++i)
72  {
73  std::cout << rp.pickup() << std::endl;
74  }
75  \endcode
76 */
77 class RandomPickup : private Uncopyable {
78 
79 private:
80  size_t _n0; ///< Initial value of \c n.
81  size_t _n; ///< Current value of \c n.
82  size_t *_elems; ///< Elements that have not been chosen yet.
83 
84 public:
85  /// Constructor.
86  /**
87  \param n -- The unsigned integer \c n defining the range
88  of values that can be picked up -- \b IN.
89  */
90  explicit RandomPickup(const size_t n);
91 
92  /// Destructor.
93  virtual ~RandomPickup() { delete [] _elems; }
94 
95  /// Get number of remaining values
96  size_t getN() const { return _n; }
97 
98  /// Reset.
99  void reset();
100 
101  /// Randomly pick up an element in [0;n-1].
102  /**
103  \return The element.
104  */
105  size_t pickup();
106 
107 };
