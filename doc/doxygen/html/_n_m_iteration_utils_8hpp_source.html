72 /// Class of utils for NM iterations.
73 /**
74  - Manage the simplex: update the characteristics (diameter, volume and normalized volume). The diameter is max(distance(y_i,y_j)). The volume is det(y_k-y_0)/!n (k=1,..n). The normalized volume is volume/diameter^n. \n
75 
76  - Hold a variable NMIterationUtils::_currentStepType for ::NMStepType (phase of Nelder Mead algorithm).
77  - Calculate the rank of DZ=[y_i-y_0] using NMIterationUtils::_rankEps as trigger (see ::getRank function)
78 
79  */

129  : IterationUtils(parentStep),

133  _simplexDiamPt1(nullptr),

135  _Delta(ArrayOfDouble()),
136  _rankEps(DEFAULT_EPSILON),
137  _currentStepType(NMStepType::UNSET),
138  _nmY(nullptr)
139  {
140  auto iter = dynamic_cast<const NMIteration*>(_iterAncestor);
141  if ( nullptr != iter )
142  {
143  _nmY = iter->getY();
144 
145  // If a Mads mesh is available, initialize the frame size.
146  auto madsMesh = iter->getMesh();
147  if ( nullptr !=
148  madsMesh )
149  {
150  _Delta = madsMesh->getDeltaFrameSize();
151  }
152  }
153  }
154 
155 
156 };
