62 void NOMAD::PSDMads::init(const std::shared_ptr<NOMAD::Evaluator>& evaluator,
63  const std::shared_ptr<NOMAD::EvaluatorControlParameters>& evalContParams)
64 {
65  _name = "PSD-MADS";
66  verifyParentNotNull();
67 
68  // Instanciate MadsInitialization member
69  _initialization = std::make_unique<NOMAD::MadsInitialization>(this);
70 
71  // Initialize all the main threads we will need.
72  // The main threads will be the ones with thread numbers 0-(nbMainThreads-1).
73  // Main thread 0 is already added to EvaluatorControl at its creation.
74  size_t nbMainThreads = _runParams->getAttributeValue<size_t>("PSD_MADS_NB_SUBPROBLEM");
75  auto evc = NOMAD::EvcInterface::getEvaluatorControl();
76  for (size_t mainThreadNum = 1; mainThreadNum < nbMainThreads; mainThreadNum++)
77  {
78  auto evalStopReason = std::make_shared<NOMAD::StopReason<NOMAD::EvalMainThreadStopType>>();
79  auto subProblemEvalContParams = std::unique_ptr<NOMAD::EvaluatorControlParameters>(new NOMAD::EvaluatorControlParameters(*evalContParams));
80  subProblemEvalContParams->checkAndComply();
81  evc->addMainThread(mainThreadNum, evalStopReason, evaluator, std::move(subProblemEvalContParams));
82  }
83 
84  _randomPickup.reset();
85 
86  omp_init_lock(&_psdMadsLock);
87 }

106 void NOMAD::PSDMads::startImp()
107 {
108  NOMAD::Algorithm::startImp();
109 
110  if (0 == NOMAD::getThreadNum())
111  {
112  // Initializations that are run in thread 0 only
113  size_t k = 0;
114  _psdMainMesh = dynamic_cast<NOMAD::MadsInitialization*>(_initialization.get())->getMesh();
115  _barrier = _initialization->getBarrier(); // Initialization was run in Algorithm::startImp()
116  _megaIteration = std::make_shared<NOMAD::MadsMegaIteration>(this, k, _barrier, _psdMainMesh, NOMAD::SuccessType::NOT_EVALUATED);
117  }
118  else
119  {
120  // Other threads must wait for thread 0 to initialize _barrier.
121  waitForBarrier();
122  }
123 }

126 bool NOMAD::PSDMads::runImp()
127 {
128  auto evc = NOMAD::EvcInterface::getEvaluatorControl();
129  const NOMAD::EvalType evalType = evc->getEvalType();
130  const NOMAD::ComputeType computeType = evc->getComputeType();
131  std::string s;
132 
133  bool isPollster = (0 == NOMAD::getThreadNum()); // Pollster is thread 0, which is always a main thread.
134  size_t k = _megaIteration->getK();
135 
136  while (!_termination->terminate(k))
137  {
138  // Create a PSDMadsMegaIteration to manage the pollster worker and the regular workers.
139  auto bestEvalPoint = (_barrier->nbXFeas() > 0) ? _barrier->getFirstXFeas() : _barrier->getFirstXInf();
140  NOMAD::Point fixedVariable(_pbParams->getAttributeValue<size_t>("DIMENSION"));
141  if (!isPollster)
142  {
143  fixedVariable = *(bestEvalPoint->getX());
144  generateSubproblem(fixedVariable);
145  }
146 
147  NOMAD::PSDMadsMegaIteration psdMegaIteration(this, k, _barrier,
148  _psdMainMesh, _megaIteration->getSuccessType(),
149  *bestEvalPoint, fixedVariable);
150  psdMegaIteration.start();
151  bool madsSuccessful = psdMegaIteration.run(); // One Mads (on pollster or subproblem) is run per PSDMadsMegaIteration.
152 
153  if (madsSuccessful)
154  {
155  auto madsOnSubPb = psdMegaIteration.getMads();
156  OUTPUT_INFO_START
157  s = madsOnSubPb->getName() + " was successful. Barrier is updated.";
158  AddOutputInfo(s);
159  OUTPUT_INFO_END
160 
161  omp_set_lock(&_psdMadsLock);
162  _lastMadsSuccessful = !isPollster; // Ignore pollster for this flag
163  auto evalPointList = madsOnSubPb->getMegaIterationBarrier()->getAllPoints();
164  NOMAD::convertPointListToFull(evalPointList, fixedVariable);
165  _barrier->updateWithPoints(evalPointList,
166  evalType,
167  computeType,
168  _runParams->getAttributeValue<bool>("FRAME_CENTER_USE_CACHE"));
169  omp_unset_lock(&_psdMadsLock);
170  }
171 
172  psdMegaIteration.end();
173 
174  if (isPollster)
175  {
176  if (doUpdateMesh())
177  {
178  // Reset values
179  // Note: In the context of PSD-Mads, always reset RandomPickup variable.
180  #pragma omp critical(randomPickup)
181  _randomPickup.reset();
182  _lastMadsSuccessful = false;
183 
184  // MegaIteration manages the mesh and barrier
185  // MegaIteration will not be run.
186  NOMAD::MadsUpdate update(_megaIteration.get());
187 
188  // Update will take care of enlarging or refining the mesh,
189  // based on the current _barrier.
190  update.start();
191  update.run();
192  update.end();
193 
194  _psdMainMesh->checkMeshForStopping(_stopReasons);
195  }
196 
197  // Update iteration number
198  _megaIteration->setK(k++);
199  }
200 
201  if (_userInterrupt)
202  {
203  hotRestartOnUserInterrupt();
204  }
205  }
206 
207  return true;
208 }

211 void NOMAD::PSDMads::endImp()
212 {
213  // To be run in thread 0 only.
214  #pragma omp master
215  {
216  // PSD-Mads is done. Ensure other threads are not stuck waiting for points to be evaluated.
217  NOMAD::CacheBase::getInstance()->setStopWaiting(true);
218  _termination->start();
219  _termination->run();
220  _termination->end();
221  }
222 
223  NOMAD::Algorithm::endImp();
224 }

227 // Update mesh:
228 // - If PSD_MADS_ORIGINAL is true, like this is done in NOMAD 3.
229 // - If a percentage of variables been covered
230 // - If the last mads on a subproblem was successful.
231 //
232 // - Not implemented: if a certain number of mads has been executed.
233 bool NOMAD::PSDMads::doUpdateMesh() const
234 {
235  if (_runParams->getAttributeValue<bool>("PSD_MADS_ORIGINAL"))
236  {
237  // Behave like the NOMAD 3 version of PSD-Mads, i.e, always update mesh.
238  return true;
239  }
240 
241  bool doUpdate = false;
242  auto coverage = _runParams->getAttributeValue<NOMAD::Double>("PSD_MADS_SUBPROBLEM_PERCENT_COVERAGE");
243  coverage /= 100.0;
244 
245  int nbRemaining = 0;
246  #pragma omp critical(randomPickup)
247  nbRemaining = _randomPickup.getN();
248 
249  if (_lastMadsSuccessful && _runParams->getAttributeValue<bool>("PSD_MADS_ITER_OPPORTUNISTIC"))
250  {
251  doUpdate = true;
252  OUTPUT_INFO_START
253  AddOutputInfo("Last subproblem iteration was successful. Update mesh.");
254  OUTPUT_INFO_END
255  }
256  // Coverage % of variables must have been covered.
257  else if (nbRemaining < (1.0-coverage.todouble()) * _pbParams->getAttributeValue<size_t>("DIMENSION"))
258  {
259  OUTPUT_INFO_START
260  NOMAD::Double pctCov = 100.0 - ((100.0 * nbRemaining) / _pbParams->getAttributeValue<size_t>("DIMENSION"));
261  std::string s = pctCov.tostring() + "% of variables were covered by subproblems. Update mesh.";
262  AddOutputInfo(s);
263  OUTPUT_INFO_END
264  doUpdate = true;
265  }
266 
267  return doUpdate;
268 }

272 {

276  const auto nbVariablesInSubproblem = _runParams->getAttributeValue<size_t>("PSD_MADS_NB_VAR_IN_SUBPROBLEM");
277  #pragma omp critical(randomPickup)
278  {
279  for (size_t k = 0; k < nbVariablesInSubproblem; k++)
280  {
281  fixedVariable[_randomPickup.pickup()] = NOMAD::Double();
282  }
