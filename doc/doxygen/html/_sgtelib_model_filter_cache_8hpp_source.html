54 // Methods available for greedy selection
55 enum class FilterSelectionMethod
56 {
57  METHOD_BEST = 0, // Method 0: Select the best point
58  METHOD_MOST_DISTANT, // Method 1: Select the most distant point
59  METHOD_BEST_MIN_DIST, // Method 2: Select the best point but with a minimum distance to points already selected
60  METHOD_BEST_GOOD_HMAX, // Method 3: Select the best points with a good enough value of hmax
61  METHOD_HIGHEST_ISOLATION, // Method 4: Select point with highest isolation number
62  METHOD_HIGHEST_DENSITY, // Method 5: Select point with highest density number

64 };
