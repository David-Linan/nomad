113 /// Read and interpret BBOutputType
114 inline std::ostream& operator<<(std::ostream& os, const BBOutputType &bbot)
115 {
116  switch (bbot)
117  {
118  case BBOutputType::OBJ:
119  os << "OBJ";
120  break;
121  case BBOutputType::PB:
122  os << "PB";
123  break;
124  case BBOutputType::EB:
125  os << "EB";
126  break;
127  case BBOutputType::CNT_EVAL:
128  os << "CNT_EVAL";
129  break;
130  case BBOutputType::BBO_UNDEFINED:
131  default:
132  return os << "BBO_UNDEFINED";
133  break;
134  }
135 
136  return os;
137 }
