309  // This template function implementation must be in the header to be available in the library
310  /**
311  Insert a new attribute by its name and type (a duplicate attribute triggers exception). Call for the AttributeFactor::Create function to create an Attribute with a templated type. Register the type (as a string) and the name of an Attribute into Parameters::_typeOfAttributes.
312  See Parameters::_definition for details.
313  */
314  template<typename T, typename ... ARGS>
315  void registerAttribute( std::string name,
316  T default_value,
317  bool algoCompatibilityCheck,
318  bool restartAttribute,
319  bool uniqueEntry,
320  ARGS&&... infoArgs)
321  {
322 
323  if ( std::is_reference<T>::value
324  || std::is_pointer<T>::value
325  || std::is_const<T>::value )
326  {
327  std::string err = "Attribute " + name;
328  err += " must be of regular type (no pointer or reference).";
329  throw Exception(__FILE__,__LINE__, err);
330  }
331 
332  NOMAD::toupper(name);
333 
334  auto attribute = AttributeFactory{}.Create<T>(name, default_value,
335  algoCompatibilityCheck, restartAttribute, uniqueEntry,
336  std::forward<ARGS>(infoArgs)...);
337 
338  auto ret = _attributes.insert(attribute);
339 
340  if ( !ret.second )
341  {
342  std::string err = "Attribute " + name + " is already in set of attributes.";
343  throw Exception(__FILE__,__LINE__, err);
344  }
345 
346  std::string typeName = typeid(T).name();
347  auto nameTypePair = std::pair<std::string, std::string>(name, typeName);
348  auto ret_t = _typeOfAttributes.insert(nameTypePair);
349 
350  if ( !ret_t.second )
351  {
352  // Attribute is already in the map of attribute types.
353  // Verify the type is the same.
354  if ( _typeOfAttributes[name] != typeName )
355  {
356  std::string err = "Trying to add attribute " + name;
357  err += " with type " + typeName;
358  err += " which is different from registered type " + _typeOfAttributes[name];
359  throw Exception(__FILE__,__LINE__, err);
360  }
361  }
362 
363  }

375  // getSpValue: value is of correct type for parameter name.
376  template<typename T> const T&
377  getSpValue(const std::string &name, bool flagCheckException, bool flagDefault = false) const
378  {
379  // Get attribute from which to get value
380  SPtrAtt att;
381  att = getAttribute(name);
382 
383  if (nullptr == att)
384  {
385  // At this point, Verify att is non-null.
386  std::string err = "getAttributeValue: attribute " + name + " does not exist";
387  throw Exception(__FILE__, __LINE__, err);
388  }
389 
390  // Verify attribute type
391  // Must use map access with "at" (not []) because the function is const
392  const std::string &typeTName = typeid(T).name();
393  if (typeTName != _typeOfAttributes.at(name))
394  {
395  std::string err = "In getAttributeValue<T> the attribute ";
396  err += name + " is not of type T = " + typeTName;
397  throw Exception(__FILE__,__LINE__, err);
398  }
399 
400  // Note: we use getAttributeValue in init() and checkAndComply().
401  // We cannot verify toBeChecked() here. It has to be verified at
402  // another level.
403 
404  // Get value from attribute
405  // Dynamic cast to the selected TypeAttribute
406  std::shared_ptr<TypeAttribute<T>> sp = std::dynamic_pointer_cast<TypeAttribute<T>>(att);
407 
408  if (flagDefault)
409  {
410  // Get initial value
411  return sp->getInitValue();
412  }
413  else
414  {
415  // All attributes except DIMENSION must be checked before accessing the value
416  if ( _toBeChecked && flagCheckException && name != "DIMENSION" )
417  {
418  std::string err = "In getAttributeValue<T> the attribute ";
419  err += name + " has not been checked";
420  throw ParameterToBeChecked(__FILE__,__LINE__, err);
421  }
422  return sp->getValue();
423  }
424  }

443  /**
444  Template specialization when getting attribute value for Point.
445  This handles the special case where the type is an ArrayOfPoint, but user asks to
446  return a Point. So we get the "value" for an ArrayOfPoint and return its first element.
447  */
448  template<typename T> const Point&
449  getAttributeValueProtected(const std::string &name, type<Point>, bool flagCheckException, bool flagDefault = false) const
450  {
451  auto namecaps = name;
452  NOMAD::toupper(namecaps);
453  if (typeid(ArrayOfPoint).name() == _typeOfAttributes.at(namecaps))
454  {
455  // Special case: Attribute type is an ArrayOfPoint, but user asks to
456  // return a Point.
457  // Get the ArrayOfPoint and return its first element.
458  const ArrayOfPoint & aop = getSpValue<ArrayOfPoint>(namecaps, flagCheckException, flagDefault);
459  if (aop.size() >= 1)
460  {
461  return aop[0];
462  }
463  else
464  {
465  std::string err = "In getAttributeValue<Point> : the attribute " + name;
466  err += " contains no point.";
467  throw Exception(__FILE__,__LINE__, err);
468  }
469  }
470 
471  // Default behaviour
472  return getSpValue<T>(namecaps, flagCheckException, flagDefault);
473  }

560  /**
561  Function to manage different set cases.
562  Default: value is of the correct type for parameter designed by name.

565  */

570

572  {
573  // At this point, Verify att is non-null.
574  std::string err = "setSpValueDefault: attribute " + name + " does not exist";
575  throw Exception(__FILE__, __LINE__, err);
576  }
577  // Dynamic cast to the selected TypeAttribute
578  std::shared_ptr<TypeAttribute<T>> sp = std::dynamic_pointer_cast<TypeAttribute<T>>( att );
579 
580  const std::string &typeTName = typeid(T).name();
581  if (typeTName != _typeOfAttributes[name])
582  {
583  std::string err = "setSpValueDefault<T> : the attribute " + name;
584  err += " is of type " + _typeOfAttributes[name];
585  err += " and not of type T = " + typeTName;
586  throw Exception(__FILE__,__LINE__, err);
587  }
588  if (!sp->uniqueEntry())
589  {
590  if (typeid(ArrayOfString).name() == _typeOfAttributes.at(name))
591  {
592  // Special case for DISPLAY parameter
593  ArrayOfString* valueAsAos = (ArrayOfString*)(&value);
594  ArrayOfString* aos = (ArrayOfString*)(&sp->getValue());
595  for (size_t i = 0; i < valueAsAos->size(); i++)
596  {
597  aos->add((*valueAsAos)[i]);
598  }
599  // Convert back to T
600  value = *((T*)(aos));
601  }
602  }
603  sp->setValue(value);
604 
605  // Keep a track of all the non default set (used by runner to display attributes)
606  if ( ! sp->isDefaultValue() )
607  {
608  _streamedAttribute << " [ ";
609  sp->display( _streamedAttribute , false /* no short info*/ );
610  _streamedAttribute << " ] " ;
611  }
